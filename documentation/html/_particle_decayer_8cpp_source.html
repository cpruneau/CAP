33void ParticleDecayer::decay2(ParticleType & parentType,
34 TLorentzVector & parentMomentum,
35 TLorentzVector & parentPosition,
36 ParticleType & childType1,
37 TLorentzVector & p1,
38 TLorentzVector & r1,
39 ParticleType & childType2,
40 TLorentzVector & p2,
41 TLorentzVector & r2)
42{
43 double e, vx, vy, vz, mp, mWidth;
44 double m1, m2, minMass,tau, taup, gamma, lifeTime;
45 mp = parentMomentum.M();
46 m1 = childType1.getMass();
47 m2 = childType2.getMass();
48 minMass = m1 + m2;
49 if (mp < minMass)
50 {
51 //cout << "<W> ParticleDecayer::decay2(...) Parent too light: M = " << mp << ", m1 = " << m1 << ", m2 = " << m2 << endl;
52 mp = 1.02*minMass;
53 }
54 e = parentMomentum.E();
55 vx = parentMomentum.Px()/e;
56 vy = parentMomentum.Py()/e;
57 vz = parentMomentum.Pz()/e;
58 gamma = parentMomentum.Gamma();
59 mWidth = parentType.getWidth();
60 tau = parentType.getLifeTime();
61 taup = gamma*tau;
62 lifeTime = -3.0E23 * taup * log(random->Rndm()); // fm
63 r1 = parentPosition;
64 r2 = parentPosition;
65 TLorentzVector shift;
66 shift.SetXYZT(vx*lifeTime,vy*lifeTime,vz*lifeTime,lifeTime);
67 r1 += shift;
68 r2 += shift;
69 double temp = mp*mp - m1*m1 - m2*m2;
70
71// cout << " mp: " << mp << endl;
72// cout << " m1: " << m1 << endl;
73// cout << " m2: " << m2 << endl;
74// cout << " minMass: " << minMass << endl;
75// cout << " Px: " << parentMomentum.Px() << endl;
76// cout << " Py: " << parentMomentum.Py() << endl;
77// cout << " Pz: " << parentMomentum.Pz() << endl;
78// cout << " beta: " << parentMomentum.Beta() << endl;
79// cout << " vx: " << vx << endl;
80// cout << " vy: " << vy << endl;
81// cout << " vz: " << vz << endl;
82// cout << " gamma: " << gamma << endl;
83// cout << " mWidth: " << mWidth << endl;
84// cout << " tau: " << tau << endl;
85// cout << " taup: " << taup << endl;
86// cout << " lifeTime: " << lifeTime << endl;
87// cout << " temp: " << temp << endl;
88
89 double p_lrf = sqrt(temp*temp - 4*m1*m1*m2*m2)/(2*mp);
90 // randomly pick emission angle of particle 1
91 double phi = twoPi*random->Rndm();
92 double cosPhi = cos(phi);
93 double sinPhi = sin(phi);
94 double cosTheta = 2.*random->Rndm() - 1.0;
95 double sinTheta = sqrt(1.00 - cosTheta*cosTheta);
96 // compute daughter particles' energy and momentum in the parent rest frame
97 double e1_lrf = sqrt(p_lrf*p_lrf + m1*m1);
98 double p1_lrf_x = p_lrf*sinTheta*cosPhi;
99 double p1_lrf_y = p_lrf*sinTheta*sinPhi;
100 double p1_lrf_z = p_lrf*cosTheta;
101 double e2_lrf = sqrt(p_lrf*p_lrf + m2*m2);
102 double p2_lrf_x = -p1_lrf_x;
103 double p2_lrf_y = -p1_lrf_y;
104 double p2_lrf_z = -p1_lrf_z;
105 p1.SetPxPyPzE(p1_lrf_x,p1_lrf_y,p1_lrf_z,e1_lrf);
106 p2.SetPxPyPzE(p2_lrf_x,p2_lrf_y,p2_lrf_z,e2_lrf);
107 // boost in "lab frame"
108 p1.Boost(vx,vy,vz);
109 p2.Boost(vx,vy,vz);
110}

112void ParticleDecayer::decay3(ParticleType & parentType,
113 TLorentzVector & parentMomentum,
114 TLorentzVector & parentPosition,
115 ParticleType & childType1,
116 TLorentzVector & p1,
117 TLorentzVector & r1,
118 ParticleType & childType2,
119 TLorentzVector & p2,
120 TLorentzVector & r2,
121 ParticleType & childType3,
122 TLorentzVector & p3,
123 TLorentzVector & r3)
124{
125 double e, vx, vy, vz, mp, mWidth;
126 double m1, m2, m3, minMass,tau, taup, gamma, lifeTime;
127 double e1_lrf, e2_lrf, e3_lrf, p1_lrf, p2_lrf, cos12_lrf;
128 double v2;
129
130 mp = parentMomentum.M();
131 m1 = childType1.getMass();
132 m2 = childType2.getMass();
133 m3 = childType3.getMass();
134 minMass = m1 + m2 + m3;
135 if (mp < minMass)
136 {
137 //cout << "<E> ParticleDecayer::decay3(...) Parent too light: M = " << mp << ", m1 = " << m1 << ", m2 = " << m2 << ", m3 = " << m3 << endl;
138 mp = mp = 1.02*minMass;
139 }
140 e = parentMomentum.E();
141 vx = parentMomentum.Px()/e;
142 vy = parentMomentum.Py()/e;
143 vz = parentMomentum.Pz()/e;
144 v2 = vx*vx + vy*vy + vz*vz;
145 gamma = parentMomentum.Gamma();
146 mWidth = parentType.getWidth();
147 tau = parentType.getLifeTime();
148 taup = gamma*tau;
149 lifeTime = -3.0E23 * taup * log(random->Rndm()); // fm
150 r1 = parentPosition;
151 r2 = parentPosition;
152 r3 = parentPosition;
153 TLorentzVector shift;
154 shift.SetXYZT(vx*lifeTime,vy*lifeTime,vz*lifeTime,lifeTime);
155 r1 += shift;
156 r2 += shift;
157 r3 += shift;
158 //
159 double deltaM = mp - m1 - m2 - m3;
160
161 do {
162 do {
163 e1_lrf = random->Rndm()*deltaM + m1;
164 e2_lrf = random->Rndm()*deltaM + m2;
165 } while (e1_lrf + e2_lrf > mp);
166 p1_lrf = sqrt(e1_lrf*e1_lrf - m1*m1);
167 p2_lrf = sqrt(e2_lrf*e2_lrf - m2*m2);
168 e3_lrf = mp - e1_lrf - e2_lrf;
169 cos12_lrf = (e3_lrf*e3_lrf - p1_lrf*p1_lrf - p2_lrf*p2_lrf - m3*m3)/(2.*p1_lrf*p2_lrf);
170 } while (cos12_lrf < - 1.0 || cos12_lrf > 1.0);
171
172 // compute the momentum of decay daughters
173 double tp2_lrf_x = p2_lrf*sqrt(1. - cos12_lrf*cos12_lrf);
174 double tp2_lrf_z = p2_lrf*cos12_lrf;
175 double tp3_lrf_x = - tp2_lrf_x;
176 double tp3_lrf_z = - (p1_lrf + tp2_lrf_z);
177 double phi = twoPi*random->Rndm();
178 double ksi = twoPi*random->Rndm();
179 double cos_theta = 2.0*random->Rndm() - 1.0;
180 double sin_phi = sin(phi);
181 double cos_phi = cos(phi);
182 double sin_ksi = sin(ksi);
183 double cos_ksi = cos(ksi);
184 double sin_theta = sqrt(1. - cos_theta*cos_theta);
185
186 double p1_lrf_x = - p1_lrf*sin_theta*cos_ksi;
187 double p1_lrf_y = p1_lrf*sin_theta*sin_ksi;
188 double p1_lrf_z = p1_lrf*cos_theta;
189 e1_lrf = sqrt(m1*m1 + p1_lrf_x*p1_lrf_x + p1_lrf_y*p1_lrf_y + p1_lrf_z*p1_lrf_z);
190 double p2_lrf_x = (tp2_lrf_x*(cos_phi*cos_theta*cos_ksi - sin_phi*sin_ksi) - tp2_lrf_z*sin_theta*cos_ksi);
191 double p2_lrf_y = (tp2_lrf_x*(-cos_phi*cos_theta*sin_ksi - sin_phi*cos_ksi) + tp2_lrf_z*sin_theta*sin_ksi);
192 double p2_lrf_z = tp2_lrf_x*(cos_phi*sin_theta) + tp2_lrf_z*cos_theta;
193 e2_lrf = sqrt(m2*m2 + p2_lrf_x*p2_lrf_x + p2_lrf_y*p2_lrf_y + p2_lrf_z*p2_lrf_z);
194 double p3_lrf_x = (tp3_lrf_x*(cos_phi*cos_theta*cos_ksi - sin_phi*sin_ksi) - tp3_lrf_z*sin_theta*cos_ksi);
195 double p3_lrf_y = (tp3_lrf_x*(-cos_phi*cos_theta*sin_ksi - sin_phi*cos_ksi) + tp3_lrf_z*(sin_theta*sin_ksi));
196 double p3_lrf_z = tp3_lrf_x*cos_phi*sin_theta + tp3_lrf_z*cos_theta;
197 e3_lrf = sqrt(m3*m3 + p3_lrf_x*p3_lrf_x + p3_lrf_y*p3_lrf_y + p3_lrf_z*p3_lrf_z);
198
199 double gamma_m_1 = gamma - 1.;
200 double vp1 = vx*p1_lrf_x + vy*p1_lrf_y + vz*p1_lrf_z;
201 double vp2 = vx*p2_lrf_x + vy*p2_lrf_y + vz*p2_lrf_z;
202 double vp3 = vx*p3_lrf_x + vy*p3_lrf_y + vz*p3_lrf_z;
203
204 p1.SetPxPyPzE(p1_lrf_x + (gamma_m_1*vp1/v2 + gamma*e1_lrf)*vx,
205 p1_lrf_y + (gamma_m_1*vp1/v2 + gamma*e1_lrf)*vy,
206 p1_lrf_z + (gamma_m_1*vp1/v2 + gamma*e1_lrf)*vz,
207 gamma*(e1_lrf + vp1));
208 p2.SetPxPyPzE(p2_lrf_x + (gamma_m_1*vp2/v2 + gamma*e2_lrf)*vx,
209 p2_lrf_y + (gamma_m_1*vp2/v2 + gamma*e2_lrf)*vy,
210 p2_lrf_z + (gamma_m_1*vp2/v2 + gamma*e2_lrf)*vz,
211 gamma*(e2_lrf + vp2));
212 p3.SetPxPyPzE(p3_lrf_x + (gamma_m_1*vp3/v2 + gamma*e3_lrf)*vx,
213 p3_lrf_y + (gamma_m_1*vp3/v2 + gamma*e3_lrf)*vy,
214 p3_lrf_z + (gamma_m_1*vp3/v2 + gamma*e3_lrf)*vz,
215 gamma*(e3_lrf + vp3));
216}
