78void RootTreeReader::initialize()
79{
80
81 if (reportStart(__FUNCTION__))
82 ;
83 Task::initialize();
84 if (reportStart("RootTreeReader",getName(),"initialize()"))
85 ;
86
87 const Configuration & configuration = getConfiguration();
88 dataInputPath = configuration.getValueString("dataInputPath");
89 dataInputFileName = configuration.getValueString("dataInputFileName");
90 dataInputTreeName = configuration.getValueString("dataInputTreeName");
91 firstFile = configuration.getValueInt("dataInputFileMinIndex");
92 lastFile = configuration.getValueInt("dataInputFileMaxIndex");
93 clonesMaxArraySize = configuration.getValueInt("clonesMaxArraySize");
94 randomizeEventPlane = configuration.getValueBool("randomizeEventPlane");
95
96 inputRootChain = new TChain(dataInputTreeName);
97 if (!inputRootChain)
98 {
99 if (reportFatal()) cout << "Chain is a null pointer" << endl;
100 postTaskFatal();
101 return;
102 }
103
104 // If the dataInputFileName is equal to "folder", then the use has requested
105 // the data be read from the folder identified by the dataInputPath parameter
106 // and using IncludePattern and ExcludePattern to extract the data files to be
107 // used in this particular analysis.
108 vector<TString> selectedFileNames;
109 if (dataInputFileName.Contains("FOLDER"))
110 {
111 selectedFileNames = getSelectedFileNamesFrom(dataInputPath);
112 }
113 else
114 {
115 selectedFileNames.push_back(dataInputPath+dataInputFileName);
116 }
117 if (selectedFileNames.size()<1)
118 {
119 if (reportError(__FUNCTION__)) cout << "No root data file selected for input" << endl;
120 postTaskError();
121 return;
122 }
123
124 // If the parameters firstFile and lastFile are less than zero, unspecified by the user,
125 // we assume the user wants to read all files in the selection.
126 if (firstFile < 0) firstFile = 0;
127 if (lastFile < 0) lastFile = selectedFileNames.size();
128 if (lastFile > int(selectedFileNames.size())) lastFile = selectedFileNames.size();
129 for(int iFile=firstFile; iFile<lastFile; iFile++)
130 {
131 TString fileName = selectedFileNames[iFile];
132 if (!fileName.EndsWith(".root")) fileName += ".root";
133 if (reportInfo(__FUNCTION__)) cout << "Adding input file:" << fileName << endl;
134 inputRootChain->Add(fileName);
135 }
136 initInputTreeMapping();
137 setInputRootTreeIndex(-1);
138 entryIndex = 0;
139 nEntries = inputRootChain->GetEntriesFast();
140 if (nEntries < 1)
141 {
142 if (reportError(__FUNCTION__)) cout << "No data found: nEntries < 1. Abort job!" << endl;
143 postTaskFatal();
144 return;
145 }
146 else
147 {
148 if (reportInfo(__FUNCTION__)) cout << "nEntries: " << nEntries << endl;
149 }
150 nBytes = 0;
151 nb = 0;
152 if (reportEnd(__FUNCTION__))
153 ;
154}

181Long64_t RootTreeReader::LoadTree(Long64_t entry)
182{
183 // Set the environment to read one entry
184 if (!inputRootChain) return -5;
185 Long64_t centry = inputRootChain->LoadTree(entry);
186 if (centry < 0) return centry;
187 if (inputRootChain->GetTreeNumber() != inputRootTreeIndex)
188 {
189 inputRootTreeIndex = inputRootChain->GetTreeNumber();
190 }
191 return centry;
192}
