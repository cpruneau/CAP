142 LogLevel debugLevel = getReportLevel();
143 TString bn = getName();
144 bn += "_";
145 unsigned int nEventFilters = eventFilters.size();
146 unsigned int nParticleFilters = particleFilters.size();
147
148 Histograms * histos;
149 if (reportInfo(__FUNCTION__))
150 {
151 cout << endl;

153 cout << " nEventFilters: " << nEventFilters << endl;
154 cout << " nParticleFilters: " << nParticleFilters << endl;
155 cout << endl;
156 }
157 for (unsigned int iEventFilter=0; iEventFilter<nEventFilters; iEventFilter++ )
158 {
159 TString efn = eventFilters[iEventFilter]->getName();
160 if (reportDebug(__FUNCTION__)) cout << "Event filter:" << efn << endl;
161 // singles
162 for (unsigned int iParticleFilter1=0; iParticleFilter1<nParticleFilters; iParticleFilter1++ )
163 {
164 TString pfn1 = particleFilters[iParticleFilter1]->getName();
165 if (reportDebug(__FUNCTION__)) cout << "Particle with filter: " << pfn1 << endl;
166 histos = new ParticleDerivedHistos(createHistogramName(bn,efn,pfn1),configuration,debugLevel);
167 histos->createHistograms();
168 derivedSingleHistograms.push_back(histos);
169 }
170 // pairs
171 for (unsigned int iParticleFilter1=0; iParticleFilter1<nParticleFilters; iParticleFilter1++ )
172 {
173 TString pfn1 = particleFilters[iParticleFilter1]->getName();
174 for (unsigned int iParticleFilter2=0; iParticleFilter2<nParticleFilters; iParticleFilter2++ )
175 {
176 TString pfn2 = particleFilters[iParticleFilter2]->getName();
177 if (reportDebug(__FUNCTION__)) cout << "Particle pairs with filter: " << pfn1 << " & " << pfn2 << endl;
178 histos = new ParticlePairDerivedHistos(createPairHistogramName(bn,efn,pfn1,pfn2),configuration,debugLevel);
179 histos->createHistograms();
180 derivedPairHistograms.push_back(histos);

182 }

252 if (reportStart(__FUNCTION__))
253 ;
254 incrementTaskExecuted();
255 const Configuration & config = getConfiguration();
256 unsigned int nEventFilters = eventFilters.size();
257 unsigned int nParticleFilters = particleFilters.size();
258 if (reportInfo(__FUNCTION__))
259 {
260 cout << endl;
261 cout << "Computing derived histograms for: " << endl;
262 cout << " nEventFilters: " << nEventFilters << endl;
263 cout << " nParticleFilters: " << nParticleFilters << endl;
264 cout << " nSingleHistos: " << getNBaseSingleHistograms() << endl;

269 ParticleHistos * bSingleHistos2;
270 ParticlePairHistos * bPairHistos;

272 ParticleDerivedHistos * dSingleHistos2;
273 ParticlePairDerivedHistos * dPairHistos;
274
275 for (unsigned int iEventFilter=0; iEventFilter<nEventFilters; iEventFilter++ )
276 {
277 if (reportDebug(__FUNCTION__)) cout << " iEventFilter:" << iEventFilter << endl;
278 unsigned int index;
279 unsigned int baseSingle = iEventFilter*nParticleFilters;
280 unsigned int basePair = iEventFilter*nParticleFilters*nParticleFilters;
281 unsigned int baseDerivedPair = basePair;
282
284 for (unsigned int iParticleFilter1=0; iParticleFilter1<nParticleFilters; iParticleFilter1++)
285 {
286 TString pfn1 = particleFilters[iParticleFilter1]->getName();
287 index = baseSingle+iParticleFilter1;
288 if (reportDebug(__FUNCTION__)) cout << " (1) iParticleFilter1:" << iParticleFilter1 << " named " << pfn1 << " with index:" << index << endl;
289 bSingleHistos1 = (ParticleHistos *) baseSingleHistograms[index];

291 dSingleHistos1 = (ParticleDerivedHistos *) derivedSingleHistograms[index];
292 if (reportDebug(__FUNCTION__)) cout << " (2) iParticleFilter1:" << iParticleFilter1 << " named " << pfn1 << " with index:" << index << endl;
293 dSingleHistos1->calculateDerivedHistograms(bSingleHistos1);
294 if (reportDebug(__FUNCTION__)) cout << " (3) iParticleFilter1:" << iParticleFilter1 << " named " << pfn1 << " with index:" << index << endl;
295 }
296
298 for (unsigned int iParticleFilter1=0; iParticleFilter1<nParticleFilters; iParticleFilter1++)
299 {
300 TString pfn1 = particleFilters[iParticleFilter1]->getName();
301 index = baseSingle+iParticleFilter1;
302 bSingleHistos1 = (ParticleHistos *) baseSingleHistograms[index];
303 dSingleHistos1 = (ParticleDerivedHistos *) derivedSingleHistograms[index];
304
305 for (unsigned int iParticleFilter2=0; iParticleFilter2<nParticleFilters; iParticleFilter2++)
306 {
307 TString pfn2 = particleFilters[iParticleFilter2]->getName();
308 if (reportDebug(__FUNCTION__))
309 cout << " iParticleFilter1:" << iParticleFilter1 << " named " << pfn1 << "; iParticleFilter2:" << iParticleFilter2<< " named " << pfn2 << endl;
310 index = baseSingle+iParticleFilter2;

312 dSingleHistos2 = (ParticleDerivedHistos *) derivedSingleHistograms[index];
313
314 index = basePair+iParticleFilter1*nParticleFilters+iParticleFilter2;
315 bPairHistos = (ParticlePairHistos *) basePairHistograms[index];
316 dPairHistos = (ParticlePairDerivedHistos *) derivedPairHistograms[index];
317 dPairHistos->calculatePairDerivedHistograms(*bSingleHistos1,*bSingleHistos2,*dSingleHistos1,*dSingleHistos2,*bPairHistos,
318 configuration.getValueDouble("binCorrPP") );
