52void EventCountHistos::loadHistograms(TFile * inputFile)
53{
54
55 if (reportStart(__FUNCTION__))
56 ;
57 if (!inputFile)
58 {
59 if (reportFatal()) cout << "Attempting to load EventCountHistos from an invalid file pointer" << endl;
60 return;
61 }
62 TString bn = getHistoBaseName();
63 TString histoName;
64
65 histoName = bn+TString("taskExecuted");
66 h_taskExecutedReset = loadH1(inputFile,histoName);
67 if (!h_taskExecutedReset && reportError()) cout << "Could not load histogram: " << histoName << endl;
68 histoName = bn+TString("eventAccepted");
69 h_eventAcceptedReset = loadH1(inputFile,histoName);
70 if (!h_eventAcceptedReset && reportError()) cout << "Could not load histogram: " << histoName << endl;
71 histoName = bn+TString("partilceAccepted");
72 h_partilceAcceptedReset = loadH1(inputFile,histoName);
73 if (!h_partilceAcceptedReset && reportError()) cout << "Could not load histogram: " << histoName << endl;
74 if (reportEnd(__FUNCTION__))
75 ;
76}

79void EventCountHistos::fill(long nTaskExecutedReset,
80 vector<long> & nEventsAcceptedReset,
81 vector<long> & nParticleAcceptedReset)
82{
83
84 if (reportStart(__FUNCTION__))
85 ;
86 h_taskExecutedReset->Fill(0.0, double(nTaskExecutedReset));
87 for (int iEventFilter=0; iEventFilter<nEventFilters; iEventFilter++)
88 {
89 h_eventAcceptedReset->Fill(double(iEventFilter),double(nEventsAcceptedReset[iEventFilter]));
90 for (int iParticleFilter=0; iParticleFilter<nParticleFilters; iParticleFilter++)
91 {
92 int index = iEventFilter*nParticleFilters + iParticleFilter;
93 h_partilceAcceptedReset->Fill(double(index),double(nParticleAcceptedReset[index]));
94 }
95 }
96 if (reportEnd(__FUNCTION__))
97 ;
98}
