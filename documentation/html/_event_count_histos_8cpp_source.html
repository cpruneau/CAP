31void EventCountHistos::loadHistograms(TFile * inputFile)
32{
33 if (!inputFile)
34 {
35 if (reportFatal()) cout << "Attempting to load EventCountHistos from an invalid file pointer" << endl;
36 return;
37 }
38 TString bn = getHistoBaseName();
39 TString histoName = bn+TString("eventCount");
40 h_eventCounts = loadH1(inputFile,histoName);
41 if (!h_eventCounts)

43 if (reportError()) cout << "Could not load histogram: " << histoName << endl;
