97void Cluster::boostZ(double boost)
98{
99 double rapidity, mt, px, py, pz, e;
100 // boost the momentum of the cluster
101 rapidity = p.Rapidity() + boost;
102 mt = p.Mt();
103 px = p.Px();
104 py = p.Py();
105 pz = mt * sinh(rapidity);
106 e = mt * cosh(rapidity);
107 p.SetPxPyPzE(px,py,pz,e);
108 for (int iPart=0; iPart<nParticles; iPart++)
109 {
110 particles[iPart]->boostRapidity(boost);
111 }
112}
