282// Generate an event
284void BWModelEventGenerator::execute()
285{
286 /* event multiplicity */
287 int multiplicity = int(gRandom->Gaus(totalMultiplicityMean,totalMultiplicitySigma));
288 int netcharge = int(gRandom->Gaus(netChargeMean,netChargeSigma));
289 int ntogenplus = int(multiplicity/2.0) + netcharge;
290 int ntogenminus = multiplicity - ntogenplus;
291 double reactionplane = TMath::TwoPi()*gRandom->Rndm();
292 if (useAllKinds)
293 {
294 azimuthalAngleAllKinds->SetParameter(0,reactionplane);
295 }
296 else
297 {
298 azimuthalAnglePions->SetParameter(0,reactionplane);
299 azimuthalAngleKaons->SetParameter(0,reactionplane);
300 azimuthalAngleProtons->SetParameter(0,reactionplane);
301 }
302
303 /* generate particles */
304 int nGenerated = 0;
305 int nGeneratedPlus = 0;
306 int nGeneratedMinus = 0;
307 TParticle pion = TParticle();
308 pion.SetPdgCode(211);
309 TParticle kaon = TParticle();
310 kaon.SetPdgCode(321);
311 TParticle proton = TParticle();
312 proton.SetPdgCode(2212);
313
314 Event & event = * eventStreams[0];
315 /* longitudinal flow */
316 double lbeta = longitudinalFlow;
317 for (int iparticle = 0; iparticle < multiplicity; iparticle++)
318 {
319 /* pseudo-rapidity */
320 double eta = gRandom->Uniform(eta_Min,eta_Max);
321 /* charge */
322 int charge = 0;
323 if (double(ntogenplus)/double(multiplicity) < gRandom->Rndm()) {
324 charge = -1;
325 nGeneratedMinus++;
326 }
327 else {
328 charge = 1;
329 nGeneratedPlus++;
330 }
331
332 /* momentum and azimuthal angle */
333 double phi;
334 double pT;
335 double mass;
336 double pid;
337 if (useAllKinds)
338 {
339 pT = ptSpectraAllKinds->GetRandom();
340 phi = azimuthalAngleAllKinds->GetRandom();
341 mass = pion.GetMass();
342 pid = pion.GetPdgCode();
343 }
344 else
345 {
346 double species = gRandom->Rndm();
347 if (species < pionFraction) {
348 pT = ptSpectraPions->GetRandom();
349 phi = azimuthalAnglePions->GetRandom();
350 mass = pion.GetMass();
351 pid = pion.GetPdgCode();
352 }
353 else if (species < pionFraction + kaonFraction)
354 {
355 pT = ptSpectraKaons->GetRandom();
356 phi = azimuthalAngleKaons->GetRandom();
357 mass = kaon.GetMass();
358 pid = kaon.GetPdgCode();
359 }
360 else
361 {
362 pT = ptSpectraProtons->GetRandom();
363 phi = azimuthalAngleProtons->GetRandom();
364 mass = proton.GetMass();
365 pid = proton.GetPdgCode();
366 }
367 }
368 Particle *particle = particleFactory->getNextObject();
369 double px = pT*TMath::Cos(phi);
370 double py = pT*TMath::Sin(phi);
371 double pz = pT*TMath::SinH(eta);
372 double E = TMath::Sqrt(mass*mass+px*px+py*py+pz*pz);
373 particle->setPidPxPyPzE( pid , px,py,pz,E);
374 particle->boost(0.0,0.0,(pz < 0) ? -lbeta : lbeta);
375 event.add(particle);
376 nGenerated++;
377 if (reportDebug()) particle->printProperties(cout);
378 }
379 if (reportDebug("BWModelEventGenerator",getName(),"execute()"))
380 {
381 cout << "BWModelEventGenerator::execute() " << nGenerated << " particles, " << nGeneratedPlus << " positive, " << nGeneratedMinus << " negative" << endl;
382 }
383}

392double BWModelEventGenerator::BGBlastWaveIntegrand(double *x, double *par)
393{
394 // integrand for boltzman-gibbs blast wave
395 // x[0] -> r (radius)
396 // p[0] -> mass
397 // p[1] -> pT (transverse momentum)
398 // p[2] -> beta_avg (avg radial velocity)
399 // p[3] -> T (freezout temperature)
400 // p[4] -> n (velocity profile)
401 double x0 = x[0];
402 double mass = par[0];
403 double pT = par[1];
404 double beta_avg = par[2];
405 double temp = par[3];
406 double n = par[4];
407
408 // Keep beta within reasonable limits
409 double beta_max = beta_avg*(n+2.0)*0.5;
410 double beta = beta_max * TMath::Power(x0, n);
411 if (beta > 0.9999999999999999) beta = 0.9999999999999999;
412 double mT = TMath::Sqrt(mass*mass+pT*pT);
413 double rho0 = TMath::ATanH(beta);
414 double arg00 = pT*TMath::SinH(rho0)/temp;
415 if (arg00 > 700.) arg00 = 700.; // avoid FPE
416 double arg01 = mT*TMath::CosH(rho0)/temp;
417 double f0 = x0*mT*TMath::BesselI0(arg00)*TMath::BesselK1(arg01);
418 return f0;
419}

422Double_t BWModelEventGenerator::BGBlastWavedNdPt_FN(double *x, double *par) {
423
424 // implementation of BGBW (dNdpt)
425
426 double pT = x[0];
427 double mass = par[0];
428 double beta_avg = par[1];
429 double temp = par[2];
430 double n = par[3];
431 static TF1 fIntBG = TF1 ("fBGBWIntegrand", BGBlastWaveIntegrand, 0, 1, 5);
432 fIntBG.SetParameters(mass, pT, beta_avg, temp,n);
433 double result = fIntBG.Integral(0,1);
434 return pT*result;
435}
