282// Generate an event
284void BWModelEventGenerator::execute()
285{
286 incrementTaskExecuted();
287 /* event multiplicity */
288 int multiplicity = int(gRandom->Gaus(totalMultiplicityMean,totalMultiplicitySigma));
289 int netcharge = int(gRandom->Gaus(netChargeMean,netChargeSigma));
290 int ntogenplus = int(multiplicity/2.0) + netcharge;
291 int ntogenminus = multiplicity - ntogenplus;
292 double reactionplane = TMath::TwoPi()*gRandom->Rndm();
293 if (useAllKinds)
294 {
295 azimuthalAngleAllKinds->SetParameter(0,reactionplane);
296 }
297 else
298 {
299 azimuthalAnglePions->SetParameter(0,reactionplane);
300 azimuthalAngleKaons->SetParameter(0,reactionplane);
301 azimuthalAngleProtons->SetParameter(0,reactionplane);
302 }
303
304 /* generate particles */
305 int nGenerated = 0;
306 int nGeneratedPlus = 0;
307 int nGeneratedMinus = 0;
308 TParticle pion = TParticle();
309 pion.SetPdgCode(211);
310 TParticle kaon = TParticle();
311 kaon.SetPdgCode(321);
312 TParticle proton = TParticle();
313 proton.SetPdgCode(2212);
314
315 Event & event = * eventStreams[0];
316 /* longitudinal flow */
317 double lbeta = longitudinalFlow;
318 for (int iparticle = 0; iparticle < multiplicity; iparticle++)
319 {
320 /* pseudo-rapidity */
321 double eta = gRandom->Uniform(eta_Min,eta_Max);
322 /* charge */
323 int charge = 0;
324 if (double(ntogenplus)/double(multiplicity) < gRandom->Rndm()) {
325 charge = -1;
326 nGeneratedMinus++;
327 }
328 else {
329 charge = 1;
330 nGeneratedPlus++;
331 }
332
333 /* momentum and azimuthal angle */
334 double phi;
335 double pT;
336 double mass;
337 double pid;
338 if (useAllKinds)
339 {
340 pT = ptSpectraAllKinds->GetRandom();
341 phi = azimuthalAngleAllKinds->GetRandom();
342 mass = pion.GetMass();
343 pid = pion.GetPdgCode();
344 }
345 else
346 {
347 double species = gRandom->Rndm();
348 if (species < pionFraction) {
349 pT = ptSpectraPions->GetRandom();
350 phi = azimuthalAnglePions->GetRandom();
351 mass = pion.GetMass();
352 pid = pion.GetPdgCode();
353 }
354 else if (species < pionFraction + kaonFraction)
355 {
356 pT = ptSpectraKaons->GetRandom();
357 phi = azimuthalAngleKaons->GetRandom();
358 mass = kaon.GetMass();
359 pid = kaon.GetPdgCode();
360 }
361 else
362 {
363 pT = ptSpectraProtons->GetRandom();
364 phi = azimuthalAngleProtons->GetRandom();
365 mass = proton.GetMass();
366 pid = proton.GetPdgCode();
367 }
368 }
369 Particle *particle = particleFactory->getNextObject();
370 double px = pT*TMath::Cos(phi);
371 double py = pT*TMath::Sin(phi);
372 double pz = pT*TMath::SinH(eta);
373 double E = TMath::Sqrt(mass*mass+px*px+py*py+pz*pz);
374 particle->setPidPxPyPzE( pid , px,py,pz,E);
375 particle->boost(0.0,0.0,(pz < 0) ? -lbeta : lbeta);
376 event.add(particle);
377 nGenerated++;
378 if (reportDebug(__FUNCTION__)) particle->printProperties(cout);
379 }
380 if (reportDebug("BWModelEventGenerator",getName(),"execute()"))
381 {
382 cout << "BWModelEventGenerator::execute() " << nGenerated << " particles, " << nGeneratedPlus << " positive, " << nGeneratedMinus << " negative" << endl;
383 }
384}

393double BWModelEventGenerator::BGBlastWaveIntegrand(double *x, double *par)
394{
395 // integrand for boltzman-gibbs blast wave
396 // x[0] -> r (radius)
397 // p[0] -> mass
398 // p[1] -> pT (transverse momentum)
399 // p[2] -> beta_avg (avg radial velocity)
400 // p[3] -> T (freezout temperature)
401 // p[4] -> n (velocity profile)
402 double x0 = x[0];
403 double mass = par[0];
404 double pT = par[1];
405 double beta_avg = par[2];
406 double temp = par[3];
407 double n = par[4];
408
409 // Keep beta within reasonable limits
410 double beta_max = beta_avg*(n+2.0)*0.5;
411 double beta = beta_max * TMath::Power(x0, n);
412 if (beta > 0.9999999999999999) beta = 0.9999999999999999;
413 double mT = TMath::Sqrt(mass*mass+pT*pT);
414 double rho0 = TMath::ATanH(beta);
415 double arg00 = pT*TMath::SinH(rho0)/temp;
416 if (arg00 > 700.) arg00 = 700.; // avoid FPE
417 double arg01 = mT*TMath::CosH(rho0)/temp;
418 double f0 = x0*mT*TMath::BesselI0(arg00)*TMath::BesselK1(arg01);
419 return f0;
420}

423Double_t BWModelEventGenerator::BGBlastWavedNdPt_FN(double *x, double *par) {
424
425 // implementation of BGBW (dNdpt)
426
427 double pT = x[0];
428 double mass = par[0];
429 double beta_avg = par[1];
430 double temp = par[2];
431 double n = par[3];
432 static TF1 fIntBG = TF1 ("fBGBWIntegrand", BGBlastWaveIntegrand, 0, 1, 5);
433 fIntBG.SetParameters(mass, pT, beta_avg, temp,n);
434 double result = fIntBG.Integral(0,1);
435 return pT*result;
436}
