111 incrementNEventsAccepted(iEventFilter);

113
114 if (reportDebug(__FUNCTION__)) cout << " cggh: " << cggh << endl;
115
116
117 unsigned int nParticles = event.getNParticles();
118 for (unsigned int iParticle=0; iParticle<nParticles; iParticle++)
119 {
120 if (reportDebug(__FUNCTION__)) cout << " iParticle: " << iParticle << endl;
121
122 Particle & particle = * event.getParticleAt(iParticle);
123 if (particle.isLive() || particle.isInteraction() )
124 {
125 if (reportDebug(__FUNCTION__)) cout << " iParticle: " << iParticle << " loop" << endl;
126
127 TLorentzVector & position = particle.getPosition();
128 rx = position.X(); // units are fm
129 ry = position.Y();
130 if (reportDebug(__FUNCTION__))
131 cout << " iParticle: " << iParticle << " loop2" << endl
132 << " rx: " << rx << endl
133 << " ry: " << ry << endl;
134 cggh->getRadiusAndGradient(rx,ry, r,gx,gy);
135 if (reportDebug(__FUNCTION__))
136 cout << " iParticle: " << iParticle << " loop3" << endl;
137// phi = 0.0;
138 phi = TMath::ATan2(gy,gx);
139 if (phi<0) phi += TMath::TwoPi();
140 beta = param_a * TMath::Power(r, param_b);
141 if (beta > betaMaximum) beta = betaMaximum;
142 //cout << " gx:" << gx << " gy:" << gy << " phi:" << phi*180.0/3.1415927 << endl;
143 double g = sqrt(gx*gx+gy*gy);
144 betax = beta * gx/g;
145 betay = beta * gy/g;
146 RadialBoostHistos * histos = (RadialBoostHistos *) histograms[0];
147 histos->fill(rx,ry,r,phi,beta,1.0);
148 particle.boost(betax,betay,0.0);
149 }
150 }
151 }
152 if (reportEnd(__FUNCTION__))
