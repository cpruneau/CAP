151void HadronGasHistograms::fillBf(int trig, int sameSignAssoc, int oppSignAssoc, int bfIndex)
152{
153 if (reportStart("HadronGasHistograms",getName(),"fillBf(...)"))
154 ;
155 double sameSignAssocYield = h_rho1->GetBinContent(sameSignAssoc);
156 double oppSignAssocYield = h_rho1->GetBinContent(oppSignAssoc);
157 double sameSignPair = h_R2->GetBinContent(trig,sameSignAssoc);
158 double oppSignPair = h_R2->GetBinContent(trig,oppSignAssoc);
159 double bf = oppSignAssocYield*oppSignPair - sameSignAssocYield*sameSignPair;
160 h_BF->SetBinContent(bfIndex, bf);
161 h_BF->SetBinError(bfIndex, 0.0);

166void HadronGasHistograms::fillRho1VsP(HadronGas & hadronGas, double volume)
167{
168 if (reportStart("HadronGasHistograms",getName(),"fillRho1VsP(...)"))
169 ;
170 const Configuration & config = getConfiguration();
171 int nP = config.getValueInt("nP");
172 double minP = config.getValueDouble("minP");
173 double maxP = config.getValueDouble("maxP");
174 double deltaP = (maxP - minP)/double(nP);
175
176 double zero = 0.0;
177 double density;
178 double temperature = hadronGas.getTemperature();
179 ParticleTypeCollection & particleTypes = hadronGas.getParticleTypes();
180 for (unsigned int iSpecies=0; iSpecies<particleTypes.size(); iSpecies++)
181 {
182 ParticleType & particleType = *(particleTypes[iSpecies]);
183 double g = particleType.getSpinFactor();
184 double mass = particleType.getMass();
185 double sign = particleType.getStatistics();
186 double mu = hadronGas.chemicalPotentials[iSpecies];
187 double p = 0.0;
188 for (int iP=1; iP<=nP; iP++)
189 {
190 density = hadronGas.computeThermalDensityVsP(temperature, mass, mu, g, sign, p);
191 h_rho1ThVsP[iSpecies]->SetBinContent(iP, density);
192 h_rho1ThVsP[iSpecies]->SetBinError(iP,zero);
193 p += deltaP;
194 }
195 }
196 if (reportEnd("HadronGasHistograms",getName(),"fillRho1VsP(...)"))
197 ;

200void HadronGasHistograms::calculateDerivedHistograms()
201{
202 if (reportStart("HadronGasHistograms",getName(),"calculateDerivedHistograms()"))
203 ;
204 double zero = 0.0;
205 if (!h_rho1) { cout << "h_rho1 Does not exist" << endl; return;}
206 if (!h_rho1Th) { cout << "h_rho1Th Does not exist" << endl; return;}
207 if (!h_rho1ToRho1Th) { cout << "h_rho1ToRho1Th Does not exist" << endl; return;}
208 if (!h_rho2Corr) { cout << "h_rho2Corr Does not exist" << endl; return;}
209 if (!h_rho2Uncorr) { cout << "h_rho2Uncorr Does not exist" << endl; return;}
210 if (!h_rho1rho1) { cout << "h_rho1rho1 Does not exist" << endl; return;}
211 if (!h_rho1thrho1th) { cout << "h_rho1thrho1th Does not exist" << endl; return;}
212 if (!h_rho2) { cout << "h_rho2 Does not exist" << endl; return;}
213 if (!h_C2) { cout << "h_C2 Does not exist" << endl; return;}
214 if (!h_R2) { cout << "h_R2 Does not exist" << endl; return;}
215 if (!h_BF) { cout << "h_BF Does not exist" << endl; return;}
216 h_rho1ToRho1Th->Divide(h_rho1,h_rho1Th);
217 int n = h_rho1Th->GetNbinsX();
218 for (int i1=1; i1<= n; i1++)
219 {
220 double rho1Th_1 = h_rho1Th->GetBinContent(i1);
221 double rho1_1 = h_rho1->GetBinContent(i1);
222
223 for (int i2=1; i2<= n; i2++)
224 {
225 double rho1Th_2 = h_rho1Th->GetBinContent(i2);
226 double rho1_2 = h_rho1->GetBinContent(i2);
227 double c2 = h_rho2Corr->GetBinContent(i1,i2);
228 double c1c1 = h_rho2Uncorr->GetBinContent(i1,i2);
229 double rho2 = c2 + rho1Th_1*rho1_2 + rho1_1*rho1Th_2- rho1Th_1*rho1Th_2 +c1c1;
230 double rho1rho1 = rho1_1*rho1_2;
231 double rho1Thrho1Th = rho1Th_1*rho1Th_2;
232 double C2 = rho2 - rho1rho1;
233 double R2 = (rho1rho1>0) ? C2/rho1rho1 - 1.0 : 0.0;
234 h_rho1rho1->SetBinContent(i1,i2,rho1rho1); h_rho1rho1->SetBinError(i1,i2,zero);
235 h_rho1thrho1th->SetBinContent(i1,i2,rho1Thrho1Th); h_rho1thrho1th->SetBinError(i1,i2,zero);
236 h_rho2->SetBinContent(i1,i2,rho2); h_rho2->SetBinError(i1,i2,zero);
237 h_C2 ->SetBinContent(i1,i2,C2); h_C2->SetBinError(i1,i2,zero);
238 h_R2 ->SetBinContent(i1,i2,R2); h_R2->SetBinError(i1,i2,zero);
239 }
240 if (reportEnd("HadronGasHistograms",getName(),"calculateDerivedHistograms()"))
241 ;
242 }
243
244 for (int k=1;k<=18;k++)
245 {
246 h_BF->SetBinError(k,zero);
247 }
248 // pi- | pi+
249 fillBf(2, 2, 3, 1);
250 // K- | pi+
251 fillBf(2, 4, 5, 2);
252 // p- | pi+
253 fillBf(2, 8, 9, 3);
254 // pi+ | pi-
255 fillBf(3, 3, 2, 4);
256 // K+ | pi-

262 // K- | K+
263 fillBf(4, 4, 5, 8);
264 // p- | K+
265 fillBf(4, 8, 9, 9);
266 // pi+ | K-
267 fillBf(5, 3, 2, 10);
268 // K+ | K-

320void HadronGasHistograms::fill(HadronGas & hadronGas)
321{
322 if (reportStart("HadronGasHistograms",getName(),"fill(HadronGas & hadronGas)"))
323 ;
324 double rho1th1, rho1_1, rho1_2, rho_2_12c, rho_2_12u, mass, g;
325 double zero = 0.0;
326 int pdgCode1;
327 ParticleTypeCollection & particleTypes = hadronGas.getParticleTypes();
328 ParticleTypeCollection & stableParticleTypes = hadronGas.getStableParticleTypes();
329 vector<double> & particleDensities = hadronGas.particleDensities;
330 vector<double> & stableParticleDensities = hadronGas.stableParticleDensities;
331 unsigned int nSpecies = particleTypes.size();
332 unsigned int nStableSpecies = stableParticleTypes.size();
333 for (unsigned int iSpecies1=0; iSpecies1<nSpecies; iSpecies1++)
334 {
335 ParticleType * ptr = particleTypes.getParticleType(iSpecies1);
336 if (ptr==nullptr)
337 {
338 if (reportError("HadronGasHistograms",getName(),"fill(HadronGas & hadronGas)"))
339 {
340 cout << "Null pointer at iSpecies1:" << iSpecies1 << endl;
341 exit(1);
342 }
343 }
344 ParticleType & particleType = *ptr;
345 g = particleType.getSpinFactor();
346 mass = particleType.getMass();
347 rho1th1 = particleDensities[iSpecies1];
348 h_rho1All->SetBinContent(iSpecies1+1, rho1th1);
349 h_rho1All->SetBinError (iSpecies1+1, 0.0);
350 h_rho1AllVsMass->Fill(mass,rho1th1/g);
351 }
352 for (unsigned int iStableSpecies1=0; iStableSpecies1<nStableSpecies; iStableSpecies1++)
353 {
354 ParticleType & type1 = *(stableParticleTypes[iStableSpecies1]);
355 g = type1.getSpinFactor();
356 mass = type1.getMass();
357 pdgCode1 = type1.getPdgCode();
358 int iSpecies1 = particleTypes.findIndexForPdgCode(pdgCode1);
359 rho1_1 = stableParticleDensities[iStableSpecies1];
360 rho1th1 = particleDensities[iSpecies1];
361 h_rho1Th->SetBinContent(iStableSpecies1+1, rho1th1); h_rho1Th->SetBinError(iStableSpecies1,zero);
362 h_rho1 ->SetBinContent(iStableSpecies1+1, rho1_1); h_rho1 ->SetBinError(iStableSpecies1,zero);
363 h_rho1ThVsMass->Fill(mass,rho1th1/g);
364 h_rho1VsMass->Fill(mass,rho1_1);
365 for (unsigned int iStableSpecies2=0; iStableSpecies2<nStableSpecies; iStableSpecies2++)
366 {
367 rho1_2 = stableParticleDensities[iStableSpecies2];
368 rho_2_12c = hadronGas.stableParticlePairDensities[iStableSpecies1][iStableSpecies2];
369 rho_2_12u = hadronGas.stableParticlePairUncorrelatedDensities[iStableSpecies1][iStableSpecies2];
370 h_rho1rho1->SetBinContent (iStableSpecies1+1, iStableSpecies2+1, rho1_1*rho1_2);
371 h_rho1rho1->SetBinError (iStableSpecies1+1, iStableSpecies2+1, zero);
372 h_rho2Corr->SetBinContent (iStableSpecies1+1, iStableSpecies2+1, rho_2_12c);
373 h_rho2Corr->SetBinError (iStableSpecies1+1, iStableSpecies2+1, zero);
374 h_rho2Uncorr->SetBinContent(iStableSpecies1+1, iStableSpecies2+1, rho_2_12u);
375 h_rho2Uncorr->SetBinError (iStableSpecies1+1, iStableSpecies2+1, zero);
376 }
377 }
378 if (reportEnd("HadronGasHistograms",getName(),"fill(HadronGas & hadronGas)"))
379 ;
