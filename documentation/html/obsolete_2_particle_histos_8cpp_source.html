95void ParticleHistos::loadHistograms(TFile * inputFile)
96{
97 if (!inputFile)
98 {
99 if (reportFatal()) cout << "Attempting to load ParticleHistos from an invalid file pointer" << endl;
100 return;
101 }
102 ParticleAnalyzerConfiguration & ac = *(ParticleAnalyzerConfiguration*) getConfiguration();
103 TString bn = getHistoBaseName();
104 h_n1 = loadH1(inputFile,bn+TString("n1"));
105 if (!h_n1)
106 {
107 if (reportError()) cout << "Could not load histogram: " << bn+TString("n1") << endl;
108 return;
109 }
110 h_n1_eTotal = loadH1(inputFile,bn+TString("n1_eTotal"));
111 h_n1_pt = loadH1(inputFile,bn+TString("n1_pt"));
112 h_n1_ptXS = loadH1(inputFile,bn+TString("n1_ptXS"));
113
114
115 if (ac.fillEta)
116 {
117 h_n1_phi = loadH1(inputFile,bn+TString("n1_phi"));
118 h_n1_eta = loadH1(inputFile,bn+TString("n1_eta"));
119 h_n1_ptEta = loadH2(inputFile,bn+TString("n1_ptEta"));
120 h_n1_phiEta = loadH2(inputFile,bn+TString("n1_phiEta"));
121
122 h_spt_phi = loadH1(inputFile,bn+TString("sumpt1_phi"));
123 h_spt_phiEta = loadH2(inputFile,bn+TString("sumpt1_phiEta"));
124 h_spt_eta = loadH1(inputFile,bn+TString("sumpt1_eta"));
125
126 h_pt_eta = loadH1(inputFile,bn+TString("pt1_eta"));
127 h_pt_phi = loadH1(inputFile,bn+TString("pt1_phi"));
128 h_pt_phiEta = loadH2(inputFile,bn+TString("pt1_phiEta"));
129 }
130 if (ac.fillY)
131 {
132 h_n1_y = loadH1(inputFile,bn+TString("n1_y"));
133 h_n1_ptY = loadH2(inputFile,bn+TString("n1_ptY"));
134 h_n1_phiY = loadH2(inputFile,bn+TString("n1_phiY"));
135 h_spt_phiY = loadH2(inputFile,bn+TString("sumpt1_phiY"));
136 h_spt_y = loadH1(inputFile,bn+TString("sumpt1_y"));
137 h_pt_phiY = loadH2(inputFile,bn+TString("pt1_phiY"));
138 h_pt_y = loadH1(inputFile,bn+TString("pt1_y"));
139 }

149void ParticleHistos::fill(TLorentzVector & p, double weight)
150{
151 ParticleAnalyzerConfiguration & ac = *(ParticleAnalyzerConfiguration*)getConfiguration();
152
153 double pt = p.Pt();
154 double phi = p.Phi(); if (phi<0) phi += TMath::TwoPi();
155 h_n1_pt ->Fill(pt, weight);
156 h_n1_ptXS ->Fill(pt, weight/pt);
157 h_n1_ptPhi ->Fill(phi, pt, weight/pt);
158 if (ac.fillEta)
159 {
160 double eta = p.Eta();
161 h_n1_ptEta ->Fill(eta, pt, weight);
162 h_n1_phiEta ->Fill(eta, phi, weight);
163 h_spt_phiEta ->Fill(eta, phi, pt*weight);
164 }
165 if (ac.fillY)
166 {
167 double y = p.Rapidity();
168 h_n1_ptY ->Fill(y, pt, weight);
169 h_n1_phiY ->Fill(y, phi, weight);
170 h_spt_phiY ->Fill(y, phi, pt*weight);
171 }
172}

182void ParticleHistos::completeFill()
183{
184 ParticleAnalyzerConfiguration & ac = *(ParticleAnalyzerConfiguration*)getConfiguration();
185
186
187 if (ac.fillEta)
188 {
189 TH1* h0 = h_n1_phiEta->ProjectionY();
190 h_n1_phi->Reset();
191 h_n1_phi->Add(h0);
192 delete h0;
193 TH1 * h1 = h_n1_phiEta->ProjectionX();
194 h_n1_eta->Reset();
195 h_n1_eta->Add(h1);
196 delete h1;
197 TH1 *h2 = h_spt_phiEta->ProjectionY();
198 h_spt_phi->Reset();
199 h_spt_phi->Add(h2);
200 delete h2;
201 TH1 * h3 = h_spt_phiEta->ProjectionX();
202 h_spt_eta->Reset();
203 h_spt_eta->Add(h3);
204 delete h3;
205 }
206
207 if (ac.fillY)
208 {
209 TH1* h0 = h_n1_phiY->ProjectionY();
210 h_n1_phi->Reset();
211 h_n1_phi->Add(h0);
212 delete h0;
213 TH1 * h1 = h_n1_phiY->ProjectionX();
214 h_n1_y->Reset();
215 h_n1_y->Add(h1);
216 delete h1;
217 TH1 *h2 = h_spt_phiY->ProjectionY();
218 h_spt_phi->Reset();
219 h_spt_phi->Add(h2);
220 delete h2;
221 TH1 * h3 = h_spt_phiY->ProjectionX();
222 h_spt_y->Reset();
223 h_spt_y->Add(h3);
224 delete h3;
225 }

236 }

240 if (h_spt_phiY) calculateAveragePtH2(h_spt_phiY, h_n1_phiY, h_pt_phiY);
