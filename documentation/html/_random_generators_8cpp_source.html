169 void VectorRandomGenerator::generate(double &theta, double &phi, double & ax, double & ay,double & az)
170 {
171 double cosTh = -1.0 + 2.0 * randomGenerator->Rndm();
172 double sinTh = sqrt(1.0 - cosTh*cosTh);
173 theta = acos(cosTh);
174 phi = 2.0*3.1415927*randomGenerator->Rndm();
175 ax = sinTh * cos(phi);
176 ay = sinTh * sin(phi);
177 az = cosTh;
178 }
