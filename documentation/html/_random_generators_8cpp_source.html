174 void VectorRandomGenerator::generate(double &theta, double &phi, double & ax, double & ay,double & az)
175 {
176 double cosTh = -1.0 + 2.0 * randomGenerator->Rndm();
177 double sinTh = sqrt(1.0 - cosTh*cosTh);
178 theta = acos(cosTh);
179 phi = 2.0*3.1415927*randomGenerator->Rndm();
180 ax = sinTh * cos(phi);
181 ay = sinTh * sin(phi);
182 az = cosTh;
