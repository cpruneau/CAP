118void HadronGas::calculateParticleDecayProbability()
119{
120 if (reportStart("HadronGas", getName(), "calculateParticleDecayProbability()" ) )
121 ;
122 int nPart = particleTypes->size();
123 int nStablePart = stableParticleTypes->size();
124 if (reportInfo("HadronGas", getName(), "calculateParticleDecayProbability()" ) )
125 {
126 cout << "==================================================================================" << std::endl;
127 cout << "<I> HadronGas::calculateParticleDecayProbability()" << endl;
128 cout << " nPart: " << nPart << endl;
129 cout << " nStablePart: " << nStablePart << endl;
130 cout << "==================================================================================" << std::endl;
131 }
132 decayProbabilities.clear();
133 decayPairProbabilities.clear();
134 // Setup arrays
135 for (int iPart = 0; iPart < nPart; iPart++)
136 {
137 vector<double> temp(nStablePart,0.0);
138 vector< vector<double> > temp2; // (nStablePart,temp);
139 for (int k=0;k<nStablePart;k++) temp2.push_back(temp);
140
141 ParticleType & particleType = * (particleTypes->getParticleType(iPart));
142 if (particleType.isStable() && !particleType.isPhoton())
143 {
144 int indexStable = stableParticleTypes->findIndexForPdgCode(particleType.getPdgCode());
145 if (indexStable<0 || indexStable>=nStablePart)
146 {
147 if (reportError("HadronGas", getName(), "calculateParticleDecayProbability()" ) )
148 {
149 cout << "==================================================================================" << std::endl;
150 cout << "iPart: " << iPart << " indexStable:" << indexStable << " --- LOGIC ERROR - ABORT" << endl;
151 cout << "Terminate task." << endl;
152 cout << "==================================================================================" << std::endl;
153 exit(1);
154 }
155 }
156 temp[indexStable] = 1.0;
157 }
158 decayProbabilities.push_back(temp);
159 decayPairProbabilities.push_back(temp2);
160 }
161 // Calculate decayProbabilities
162 for (int iPart = 0; iPart < nPart; iPart++)
163 {
164 vector<double> temp(nStablePart,0.0);
165 vector< vector<double> > temp2; // (nStablePart,temp);
166 for (int k=0;k<nStablePart;k++) temp2.push_back(temp);
167
168 vector<int> indexOfDecayParts(5,0);
169 ParticleType & particleType = * (particleTypes->getParticleType(iPart));
170 for (int jChannel = 0; jChannel < particleType.getNDecayModes(); jChannel++)
171 {
172 ParticleDecayMode & decayMode = particleType.getDecayMode(jChannel);
173 double branchingRatio = decayMode.getBranchingRatio();
174 int nChildren = decayMode.getNChildren();
175 for (int kParticle = 0; kParticle < nChildren; kParticle++)
176 {
177 int pdgCode = decayMode.getChildPdgCode(kParticle);
178 int index = particleTypes->findIndexForPdgCode(pdgCode);
179 if (index<0 || index > iPart)
180 {
181 if (reportError("HadronGas", getName(), "calculateParticleDecayProbability()" ) )
182 {
183 cout << "==================================================================================" << std::endl;
184 cout << "lPart1=" << index << " -- Could not find PDG code: " << pdgCode << endl;
185 cout << "Terminate task." << endl;
186 cout << "==================================================================================" << std::endl;
187 exit(1);
188 }
189 }
190 indexOfDecayParts[kParticle] = index;
191 }
192 // now we have a local list of the pdg ids iPart decays into (w/o photons)
193 // must look at singles and then at pairs
194 // if there are no particleTypes left, just skip this decay channel...
195 double prob1, prob2;
196 for (int kParticle1 = 0; kParticle1 < nChildren; kParticle1++)
197 {
198 int lPart1 = indexOfDecayParts[kParticle1];
199 for (int mStable1 = 0; mStable1 < nStablePart; mStable1++)
200 {
201 prob1 = decayProbabilities[lPart1][mStable1];
202 temp[mStable1] += branchingRatio*prob1;
203
204 //cout << "kParticle1:" << kParticle1 << "mStable1" << mStable1 << endl;
205
206 for (int kParticle2 = 0; kParticle2 < nChildren; kParticle2++)
207 {
208 if (kParticle1 != kParticle2)
209 {
210 int lPart2 = indexOfDecayParts[kParticle2];
211 //cout << "lPart2:" << lPart2 << endl;
212
213 //cout << " kParticle1:" << kParticle1 << " lPart1: " << lPart1 << " kParticle2:" << kParticle2 << " lPart2: " << lPart2 << endl;
214 for (int mStable2 = 0; mStable2 < nStablePart; mStable2++)
215 {
216 //cout << "mStable2:" << mStable2 << endl;
217
218 prob2 = decayProbabilities[lPart2][mStable2];
219 temp2[mStable1][mStable2] += branchingRatio*prob1*prob2;
220 }
221 }
222 }
223
224 for (int mStable3 = 0; mStable3 < nStablePart; mStable3++)
225 {
226 //cout << "mStable3:" << mStable3 << endl;
227
228 prob1 = decayProbabilities[mStable1][mStable3];
229 //if (branchingRatio * prob1>0.0) cout << " NON ZERO HAPPENS lPart1: " << lPart1 << " mStable3:" << mStable3<< endl;
230 temp2[mStable1][mStable3] += branchingRatio *prob1;
231 }
232 }
233 }
234 }
235 // cout << "Store results." << endl;
236 for (int iStable=0;iStable<nStablePart;iStable++)
237 {
238 decayProbabilities[iPart][iStable] = temp[iStable];
239 //cout << "iPart:" << iPart << " iStable:" << iStable << " decayProbabilities[iPart][iStable]:" << decayProbabilities[iPart][iStable] << endl;
240 for (int jStable=0;jStable<nStablePart;jStable++)
241 {
242 decayPairProbabilities[iPart][iStable][jStable] = temp2[iStable][jStable];
243 //cout << "iPart:" << iPart << " iStable:" << iStable << " jStable:" << jStable << " decayPairProbabilities[iPart][iStable][jStable] :" << decayPairProbabilities[iPart][iStable][jStable] << endl;
244 }
245 }
246// cout << "Next Particle." << endl;
247
248 }
249 if (reportEnd("HadronGas", getName(), "calculateParticleDecayProbability()" ) )
250 ;

253void HadronGas::calculateParticleDecays()
254{
255 if (reportStart("HadronGas", getName(), "calculateParticleDecays()" ) )
256 ;
257 int nPart = particleTypes->size();
258 int nStablePart = stableParticleTypes->size();
259
260 if (stableParticleDensities.size()>0)
261 {
262 stableParticleDensities.clear();
263 stableParticlePairDensities.clear();
264 stableParticlePairCorrelatedDensities.clear();
265 stableParticlePairUncorrelatedDensities.clear();
266 }
267 vector<double> temp(nStablePart,0.0);
268 vector< vector<double> > temp2(nStablePart,temp);
269 stableParticleDensities = temp;
270 stableParticlePairDensities = temp2;
271 stableParticlePairCorrelatedDensities = temp2;
272 stableParticlePairUncorrelatedDensities = temp2;
273
274 double stableDensity;
275 for (int jStable = 0; jStable < nStablePart;jStable++)
276 {
277 stableDensity = 0.0;
278 for (int kPart = 0; kPart < nPart;kPart++)
279 {
280 stableDensity += decayProbabilities[kPart][jStable]*particleDensities[kPart];
281 }
282 stableParticleDensities[jStable] = stableDensity;
283 }
284 for (int jStable1 = 0; jStable1 < nStablePart; jStable1++)
285 {
286 for (int jStable2 = 0; jStable2 < nStablePart; jStable2++)
287 {
288 stableDensity = 0.0;
289 for (int kPart = 0; kPart < nPart;kPart++)
290 {
291 stableDensity += decayPairProbabilities[kPart][jStable1][jStable2]*particleDensities[kPart];
292 }
293 stableParticlePairCorrelatedDensities[jStable1][jStable2] = stableDensity;
294
295 stableDensity = 0.0;
296 for (int kPart1 = 0; kPart1 < nPart;kPart1++)
297 {
298 double v1 = decayProbabilities[kPart1][jStable1]*particleDensities[kPart1];
299 for (int kPart2 = 0; kPart2 < nPart;kPart2++)
300 {
301 if (kPart1 == kPart2) continue;
302 double v2 = decayProbabilities[kPart2][jStable2]*particleDensities[kPart2];
303 stableDensity += v1*v2;
304 }
305 stableParticlePairUncorrelatedDensities[jStable1][jStable2] = stableDensity;
306 }
307
308 stableParticlePairDensities[jStable1][jStable2] = stableParticleDensities[jStable1]*stableParticleDensities[jStable2];
309 }
310 }
311 if (reportEnd("HadronGas", getName(), "calculateParticleDecays()" ) )
312 ;
313}

342// ========================================================================================
343// Compute all properties of this gas at the given temperature
344// and chemical potentials
345// ========================================================================================
346void HadronGas::calculateAllProperties(double _temp, double _muB, double _muS)
347{
348 if (reportStart("HadronGas", getName(), "calculateAllProperties(double _temp, double _muB, double _muS)" ) )
349 ;
350 temperature = _temp;
351 muB = _muB;
352 muS = _muS;
353
354 chemicalPotentials.clear();
355 particleDensities.clear();
356 deltaN_bulk_term1s.clear();
357 deltaN_bulk_term2s.clear();
358 deltaN_qmu_term1s.clear();
359 deltaN_qmu_term2s.clear();
360 particleEnergies.clear();
361 particlePressures.clear();
362 particleEntropies.clear();
363 particleDpOverTmu.clear();
364 particleDeOverTDmu.clear();
365 particleDnDmu.clear();
366 particleDsDmu.clear();
367
368
369 double mu;
370 double numberDensity;
371 double energyDensity;
372 double entropyDensity;
373 double pressure;
374
375 double deltaN_bulk_term1;
376 double deltaN_bulk_term2;
377 double deltaN_qmu_term1;
378 double deltaN_qmu_term2;
379
380 double dsdmu;
381 double dndmu;
382 double deoverTdmu;
383 double dPoverTdmu;
384 double gSpin;
385 double arg;
386 double lambda;
387
388 double beta = 1.0/temperature;
389 double t2 = temperature*temperature;
390 double sign;
391 double mass;
392 double mass2;
393 double mass3;
394 double mass4;
395 double mbeta;
396 double mbeta2;
397 double mbeta3;
398 double mbeta4;
399 double hbarC = 0.19733;
400 double hbarC2 = hbarC*hbarC;
401 double hbarC3 = hbarC2*hbarC;
402 double twiceMPi2 = 2*M_PI*M_PI;
403 double theta;
404 double fugacity;
405 double besselKn1;
406 double besselKn2;
407 double doubleJ;
408
409 thermalNumberDensity = 0.0;
410 thermalEnergyDensity = 0.0;
411 thermalEntropyDensity = 0.0;
412 thermalPressure = 0.0;
413 thermalNetBaryonDensity = 0.0;
414 thermalNetStrangenessDensity = 0.0;
415 thermalNetChargeDensity = 0.0;
416
417 unsigned int nPart = particleTypes->size();
418 for (unsigned int iPart=0; iPart < nPart; iPart++)
419 {
420 ParticleType & particleType = *(particleTypes->getParticleType(iPart));
421 mass = particleType.getMass();
422
423 if (mass>0.0)
424 {
425 gSpin = particleType.getSpinFactor();
426 sign = particleType.getStatistics();
427 mass2 = mass*mass;
428 mass3 = mass2*mass;
429 mass4 = mass3*mass;
430 mbeta = mass*beta;
431 mbeta2 = mbeta*mbeta;
432 mbeta3 = mbeta2*mbeta;
433 mbeta4 = mbeta3*mbeta;
434 mu = muB*particleType.getBaryon() + muS*particleType.getStrange();
435
436 numberDensity = 0.0;
437 deltaN_bulk_term1 = 0.0;
438 deltaN_bulk_term2 = 0.0;
439 deltaN_qmu_term1 = 0.0;
440 energyDensity = 0.0;
441 pressure = 0.0;
442 dPoverTdmu = 0.0;
443 deoverTdmu = 0.0;
444 dndmu = 0.0;
445 for (int j = 1; j < trunOrder; j++)
446 {
447 doubleJ = double(j);
448 arg = doubleJ*mbeta; // argument inside bessel functions
449 lambda = exp(beta*mu); // fugacity factor
450 theta = pow(-sign, j-1);
451 fugacity = pow(lambda, j);
452 besselKn1 = TMath::BesselK1(arg);
453 besselKn2 = TMath::BesselK(2,arg);
454
455 numberDensity += theta*fugacity*besselKn2/doubleJ;
456 deltaN_bulk_term1 += theta*fugacity*(mass*beta*besselKn1 + 3.0/doubleJ*besselKn2);
457 deltaN_bulk_term2 += theta*fugacity*besselKn1;
458 deltaN_qmu_term1 += theta*fugacity*besselKn2/doubleJ; // baryon diffusion contribution
459 double I_1_1 = exp(-arg)/arg*(2./(arg*arg) + 2.0/arg - 0.5);
460 double I_1_2 = 3./8.0*ROOT::Math::expint_n(2, arg);
461 double I_1_n = I_1_1 + I_1_2;
462 double double_factorial = 1.; // record (2k-5)!!
463 double factorial = 2.; // record k! start with 2!
464 double factor_2_to_k_power = 4.; // record 2^k start with 2^2
465 for (int k = 3; k < trunOrder; k++)
466 {
467 double_factorial *= (2*k - 5);
468 factorial *= k;
469 factor_2_to_k_power *= 2;
470 double rrr = ROOT::Math::expint_n(2*k-2, arg);
471 double I_1_k = (3.*double_factorial/factor_2_to_k_power/factorial*rrr);
472 I_1_n += I_1_k;
473 }
474 I_1_n = -mbeta3*I_1_n;
475 deltaN_qmu_term2 += doubleJ*theta*fugacity*I_1_n;
476 energyDensity += theta*fugacity*(3.*besselKn2/(arg*arg) + besselKn1/arg);
477 pressure += theta*fugacity*besselKn2/doubleJ/doubleJ;
478 dPoverTdmu += theta*fugacity*besselKn2/doubleJ;
479 deoverTdmu += theta*fugacity*doubleJ*(3*besselKn2/(arg*arg)+ besselKn1/arg);
480 dndmu += theta*fugacity*besselKn2;
481 }
482 numberDensity = numberDensity*gSpin*mass2*temperature/twiceMPi2/hbarC3;
483 deltaN_bulk_term1 = mass2*temperature*deltaN_bulk_term1;
484 deltaN_bulk_term2 = 0.33333*mass3*deltaN_bulk_term2;
485 deltaN_qmu_term1 = mass2*deltaN_qmu_term1/(beta*beta);
486 deltaN_qmu_term2 = deltaN_qmu_term2/(3.0*beta*beta*beta);
487 energyDensity = energyDensity*gSpin*mass4/twiceMPi2/hbarC3;
488 pressure = pressure*gSpin*mass2*t2/twiceMPi2/hbarC3;
489 entropyDensity = (energyDensity + pressure - mu*numberDensity)*beta; // unit : 1/fm^3
490 dPoverTdmu = dPoverTdmu*gSpin*mass2/twiceMPi2/hbarC3;
491 deoverTdmu = deoverTdmu*gSpin*mass4/twiceMPi2/hbarC3;
492 dndmu = dndmu*gSpin*mass/twiceMPi2;
493 dsdmu = dPoverTdmu - numberDensity - mu*dndmu + deoverTdmu;
494
495 chemicalPotentials.push_back(mu);
496 particleDensities.push_back(numberDensity);
497 deltaN_bulk_term1s.push_back(deltaN_bulk_term1);
498 deltaN_bulk_term2s.push_back(deltaN_bulk_term2);
499 deltaN_qmu_term1s.push_back(deltaN_qmu_term1);
500 deltaN_qmu_term2s.push_back(deltaN_qmu_term2);
501 particleEnergies.push_back(energyDensity);
502 particlePressures.push_back(pressure);
503 particleEntropies.push_back(entropyDensity);
504 particleDpOverTmu.push_back(dPoverTdmu);
505 particleDeOverTDmu.push_back(deoverTdmu);
506 particleDnDmu.push_back(dndmu);
507 particleDsDmu.push_back(dsdmu);
508
509 thermalNumberDensity += numberDensity;
510// cout << setw(5) << "iPart:" << setw(5)<< iPart;
511// cout << setw(25) << "NumberDensity:";
512// cout << scientific << setw(15)<< setprecision(5) << numberDensity;
513// cout << setw(25) << "ThermalNumberDensity:";
514// cout << scientific << setw(15)<< setprecision(5) << thermalNumberDensity << endl;
515
516 thermalEnergyDensity += energyDensity;
517 thermalEntropyDensity += entropyDensity;
518 thermalPressure += pressure;
519 if (particleType.isBaryon()) thermalNetBaryonDensity += numberDensity * particleType.getBaryon();
520 if (particleType.isCharged()) thermalNetChargeDensity += numberDensity * particleType.getCharge();
521 if (particleType.isStrange()) thermalNetStrangenessDensity += numberDensity * particleType.getStrange();
522 }
523 else
524 {
525 double zero = 0.0;
526 chemicalPotentials.push_back(zero);
527 particleDensities.push_back(zero);
528 deltaN_bulk_term1s.push_back(zero);
529 deltaN_bulk_term2s.push_back(zero);
530 deltaN_qmu_term1s.push_back(zero);
531 deltaN_qmu_term2s.push_back(zero);
532 particleEnergies.push_back(zero);
533 particlePressures.push_back(zero);
534 particleEntropies.push_back(zero);
535 particleDpOverTmu.push_back(zero);
536 particleDeOverTDmu.push_back(zero);
537 particleDnDmu.push_back(zero);
538 particleDsDmu.push_back(zero);
539
540 }
541 }
542 calculateParticleDecays();
543 setupDecayGenerator();
544 if (reportEnd("HadronGas", getName(), "calculateAllProperties(double _temp, double _muB, double _muS)" ) )
545 ;
546}

564// ======================================================================
565// computes dn/dp at given temperature, mass, mu, g, sign, and p
566// ======================================================================
567double HadronGas::computeThermalDensityVsP(double temperature, double mass, double mu, double g, double sign, double p)
568{
569 if (reportStart("HadronGas", getName(), "computeThermalDensityVsP(...)" ) )
570 ;
571 double e = sqrt(p*p+mass*mass);
572 double prefactor = g/twoPiCube;
573 double arg = (e-mu)/temperature;
574 double density = prefactor*(p/e)/(exp(arg)+sign);
575 if (reportEnd("HadronGas", getName(), "computeThermalDensityVsP(...)" ) )
576 ;
577 return density;
578}
