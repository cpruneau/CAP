20// Compute Total Xsect vs sqrt(s)
21// x[0] : sqrt(s)
22// Gauron parameterization is "s"
23double GauronModelF1(double *xx, double *pp)
24{
25 double s = xx[0]*xx[0];
26 double Z = pp[0];
27 double X = pp[1];
28 double eps = pp[2];
29 double Y1 = pp[3];
30 double Y2 = pp[4];
31 double alpha = pp[5]-1.0;
32 return Z + X*TMath::Power(s,eps) + (Y1 - Y2)*TMath::Power(s,alpha);
33}

35double GauronGammaModelF1(double *xx, double *pp)
36{
37 double s = xx[0]*xx[0];
38 double delta = pp[0];
39 double Z = pp[1];
40 double X = pp[2];
41 double eps = pp[3];
42 double Y1 = pp[4];
43 double alpha = pp[5]-1;
44 return delta*Z + delta*X*TMath::Power(s,eps) + Y1*TMath::Power(s,alpha);
45}
