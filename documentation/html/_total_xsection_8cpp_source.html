15// Compute Total Xsect vs sqrt(s)
16// x[0] : sqrt(s)
17// Gauron parameterization is "s"
18double GauronModelF1(double *xx, double *pp)
19{
20 double s = xx[0]*xx[0];
21 double Z = pp[0];
22 double X = pp[1];
23 double eps = pp[2];
24 double Y1 = pp[3];
25 double Y2 = pp[4];
26 double alpha = pp[5]-1.0;
27 return Z + X*TMath::Power(s,eps) + (Y1 - Y2)*TMath::Power(s,alpha);
28}

30double GauronGammaModelF1(double *xx, double *pp)
31{
32 double s = xx[0]*xx[0];
33 double delta = pp[0];
34 double Z = pp[1];
35 double X = pp[2];
36 double eps = pp[3];
37 double Y1 = pp[4];
38 double alpha = pp[5]-1;
39 return delta*Z + delta*X*TMath::Power(s,eps) + Y1*TMath::Power(s,alpha);
40}
