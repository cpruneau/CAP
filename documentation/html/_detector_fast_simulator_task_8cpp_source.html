49void DetectorFastSimulatorTask::execute()
50{
51 incrementEventProcessed();
52 ParticleFilter & particleFilterPrimary = * particleFilters[0];
53 ParticleFilter & particleFilterSecondary = * particleFilters[1];
54 Event & event = *eventStreams[0]; // source -- unsmeared
55 Event & eventDet = *eventStreams[1]; // target -- smeared.
56
57 unsigned int nParticles = event.getNParticles();
58 for (unsigned int iParticle=0; iParticle<nParticles; iParticle++)
59 {
60 Particle & particle = * event.getParticleAt(iParticle);
61 if (particle.isLive())
62 {
63 if (particle.isPrimary())
64 {
65 if (particleFilterPrimary.accept(particle))
66 {
67 Particle & particleDet = *particleFactory->getNextObject();
68 particleDet = particle;
69 particleDet.setTruth(&particle);
70 particleSmearers[0]->smear(particle.getMomentum(),particleDet.getMomentum());
71 eventDet.add(&particleDet);
72 }
73 else if (particleFilterSecondary.accept(particle))
74 {
75 Particle & particleDet = *particleFactory->getNextObject();
76 particleDet = particle;
77 particleDet.setTruth(&particle);
78 particleSmearers[1]->smear(particle.getMomentum(),particleDet.getMomentum());
79 eventDet.add(&particleDet);
80 }
81 }
82 }
83 }
