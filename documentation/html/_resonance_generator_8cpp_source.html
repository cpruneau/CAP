78void ResonanceGenerator::execute()
79{
80 if (reportStart("ResonanceGenerator",getName(),"execute()"))
81 ;
82 incrementTaskExecuted();
83 Event & event = *eventStreams[0];
84 EventProperties & ep = * eventStreams[0]->getEventProperties();
85
86 Particle * interaction;
87 if (standaloneMode)
88 {
89 // In this mode, we generate one PYTHIA (pp) collision per event. One interaction vertex is
90 // inserted in the event stream and PYTHIA is called to carry out the particle generation.
91 // factory and event resets done by the task iterator to avoid repetition.
92 event.reset();
93 particleFactory->reset();
94 // resetParticleCounters();
95 interaction = particleFactory->getNextObject();
96 interaction->reset();
97 interaction->setType( ParticleType::getInteractionType());
98 interaction->setXYZT(0.0, 0.0, 0.0, 0.0);
99 event.add(interaction);
100 event.setNucleusA(1.0,1.0);
101 event.setNucleusB(1.0,1.0);
102 generate(interaction);
103 ep.zProjectile = 1; // atomic number projectile
104 ep.aProjectile = 1; // mass number projectile
105 ep.nPartProjectile = 1; // number of participants projectile
106 ep.zTarget = 1; // atomic number target
107 ep.aTarget = 1; // mass number target
108 ep.nPartTarget = 1; // number of participants target
109 ep.nParticipantsTotal = 2; // total number of participants
110 ep.nBinaryTotal = 1; // total number of binary collisions
111 ep.impactParameter = -99999; // nucleus-nucleus center distance in fm
112 ep.fractionalXSection = -99999; // fraction cross section value
113// ep.referenceMultiplicity = getNParticlesAccepted(); // nominal multiplicity in the reference range
114// ep.particlesCounted = getNParticlesCounted();
115// ep.particlesAccepted = getNParticlesAccepted();
116 }
117 else
118 {
119 // In this mode, we generate several PYTHIA collisions per event. Interaction vertices and their locations
120 // are assumed to be already loaded in the event stream and we produce as many PYTHIA events as there
121 // are nucleon-nucleon interactions in the stream. Objects nucleusA and nucleusB are assumed defined by
122 // an earlier task and the structure EventProperties is assumed filled by that earlier task.He we only fill the number of
123 // particle generated and accepted (in addition to storing the generated particles in the event)
124
125 // May skip this event if it does not satisfy the event cut.
126 if (!eventFilters[0]->accept(event)) return;
127 if (event.getParticleCount() < 1)
128 {
129 return;
130 }
131 vector<Particle*> interactions = event.getNucleonNucleonInteractions();
132
133 unsigned int n = interactions.size();
134 // if (reportWarning("ResonanceGenerator",getName(),"execute()"))
135 // cout << "Size of interactions:" << n << endl;
136 for (unsigned int kInter=0; kInter<n; kInter++)
137 {
138 generate(interactions[kInter]);
139 }
140// ep.referenceMultiplicity = getNParticlesAccepted(); // nominal multiplicity in the reference range
141// ep.particlesCounted = getNParticlesCounted();
142// ep.particlesAccepted = getNParticlesAccepted();
143 }
144 if (reportDebug("ResonanceGenerator",getName(),"execute()"))
145 {
146 ep.printProperties(cout);
147 }
