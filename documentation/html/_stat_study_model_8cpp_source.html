40void StatStudyModel::generate(double & nPlus, double & nMinus, double & nPlusEff, double & nMinusEff)
41{
42 double nPlusPlus;
43 double nPlusMinus;
44 double nMinusMinus;
45
46 nPlus = int( gRandom->Gaus(nPlusAvg,sqrt(nPlusAvg)) +0.5 );
47 nMinus = int( gRandom->Gaus(nMinusAvg,sqrt(nMinusAvg)) +0.5 );
48 nPlusPlus = int( gRandom->Gaus(nPlusPlusAvg,sqrt(nPlusPlusAvg)) +0.5 );
49 nPlusMinus = int( gRandom->Gaus(nPlusMinusAvg,sqrt(nPlusMinusAvg))+0.5 );
50 nMinusMinus = int( gRandom->Gaus(nMinusMinusAvg,sqrt(nMinusMinusAvg))+0.5);
51 nPlus += 2.0*nPlusPlus;
52 nPlus += nPlusMinus;
53 nMinus += 2.0*nMinusMinus;
54 nMinus += nPlusMinus;
55 nPlusEff = binomialLoss(nPlus,plusEfficiency);

57}
