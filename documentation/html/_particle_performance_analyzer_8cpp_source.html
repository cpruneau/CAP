162void ParticlePerformanceAnalyzer::execute()
163{
164
165 if (reportStart(__FUNCTION__))
166 ;
167 incrementTaskExecuted();
168 Event & recoEvent = *eventStreams[1];
169 //cout<< "ParticlePerformanceAnalyzer::execute() -- 2 --" << endl;
170 unsigned int nEventFilters = eventFilters.size();
171 unsigned int nParticleFilters = particleFilters.size();
172 unsigned long nParticles = recoEvent.getParticleCount();
173 //cout<< "ParticlePerformanceAnalyzer::execute() -- 3 --" << endl;
174
175 for (unsigned int iEventFilter=0; iEventFilter<nEventFilters; iEventFilter++ )
176 {
177 //cout<< "ParticlePerformanceAnalyzer::execute() -- 4 --" << endl;
178 if (!eventFilters[iEventFilter]->accept(recoEvent)) continue;
179 //cout<< "ParticlePerformanceAnalyzer::execute() -- 5 --" << endl;
180 incrementNEventsAccepted(iEventFilter);
181 ParticlePerformanceHistos * histos = (ParticlePerformanceHistos *) baseSingleHistograms[0];
182 for (unsigned int iParticle=0; iParticle<nParticles; iParticle++)
183 {
184 //cout<< "ParticlePerformanceAnalyzer::execute() -- 6 --" << endl;
185 Particle & recoParticle = * recoEvent.getParticleAt(iParticle);
186 for (unsigned int iParticleFilter=0; iParticleFilter<nParticleFilters; iParticleFilter++ )
187 {
188 //cout<< "ParticlePerformanceAnalyzer::execute() -- 7 --" << endl;
189 if (particleFilters[iParticleFilter]->accept(recoParticle))
190 {
191 //cout<< "ParticlePerformanceAnalyzer::execute() -- 8 --" << endl;
192 histos->fill(recoParticle,1.0);
193 } // particle accepted by filter
194 } // filter loop
195 } //particle loop
196 } // event filter loop
197}
