167void CollisionGeometryGenerator::execute()
168{
169 if (reportStart("CollisionGeometryGenerator",getName(),"execute()"))
170 ;
171 incrementEventProcessed();
172 Event & event = *eventStreams[0];
173 //event.reset();
174 Nucleus & nucleusA = event.getNucleusA();
175 Nucleus & nucleusB = event.getNucleusB();
176 double rr = gRandom->Rndm();
177 double b = sqrt(minBSq + rr*(maxBSq-minBSq));
178 event.setImpactParameter(b);
179 nucleusGeneratorA->generate(nucleusA, -b/2.0);
180 nucleusGeneratorB->generate(nucleusB, b/2.0);
181 Particle* interaction;
182 for (unsigned int i1=0; i1<nucleusA.getNNucleons(); i1++)
183 {
184 Particle * nucleonA = nucleusA.getNucleonAt(i1);
185 for (unsigned int i2=0; i2<nucleusB.getNNucleons(); i2++)
186 {
187 Particle * nucleonB = nucleusB.getNucleonAt(i2);
188 double dSq = nucleonA->distanceXYSq(nucleonB);
189 if (dSq<maxNNDistanceSq)
190 {
191 //cout << " A: x= " << nucleonA->getPosition().X() << " B: x= " << nucleonB->getPosition().X() << " I: x= " << interaction.getPosition().X() << endl;
192 interaction = event.addInteraction(nucleonA,nucleonB);
193 if (!nucleonA->isWounded())
194 {
195 TLorentzVector & positionA = nucleonA->getPosition();
196 event.getParticipantMoments().fill(positionA.X() ,positionA.Y());
197 nucleonA->setWounded(true);
198 }
199 if (!nucleonB->isWounded())
200 {
201 TLorentzVector & positionB = nucleonA->getPosition();
202 event.getParticipantMoments().fill(positionB.X() ,positionB.Y());
203 nucleonB->setWounded(true);
204 }
205 TLorentzVector & positionInt = interaction->getPosition();
206 event.getBinaryMoments().fill(positionInt.X(),positionInt.Y());
207 }
208 }
209 }
210 event.getBinaryMoments().calculate();
211 event.getBinaryMoments().calculate();
