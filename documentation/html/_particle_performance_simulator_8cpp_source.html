231void ParticlePerformanceSimulator::smearMomentum(const TLorentzVector &in, TLorentzVector & out)
232{
233 double smearedPt = 0.0;
234 double smearedEta = 0.0;
235 double smearedPhi = 0.0;
236 double pt = in.Pt();
237 double phi = in.Phi();

244void ParticlePerformanceSimulator::smearMomentum(double pt, double eta, double phi,
245 double &smearedPt, double &smearedEta, double &smearedPhi)
246{
247 double bias;
248 double rms;
249 double zeroRms = 0.0001;
250
251 switch (resolutionOption)
252 {
253 default:
254 case 0:
255 bias = 0.0;
256 rms = zeroRms;
257 smearedPt = pt;
258 smearedEta = eta;
259 smearedPhi = phi;
260 break;
261
262 case 1:
263 smearFromFunction(pt, eta, phi, ptFunction, bias, rms);
264 smearedPt = gRandom->Gaus(pt+bias,rms);
265 //cout << "pt:"<< pt<< " eta:" << eta << " phi:" << phi << " bias:" << bias << " rms:" << rms << " smearedPt" << smearedPt << endl;
266 smearFromFunction(pt, eta, phi, etaFunction, bias, rms);
267 smearedEta = gRandom->Gaus(eta+bias,rms);
268 smearFromFunction(pt, eta, phi, phiFunction, bias, rms);
269 smearedPhi = gRandom->Gaus(phi+bias,rms);
270 break;
271
272 case 2:
273 smearFromHisto(pt, eta, phi, biasPtHistogram, rmsPtHistogram, bias, rms);
274 smearedPt = gRandom->Gaus(pt+bias,rms);
275 smearFromHisto(pt, eta, phi, biasEtaHistogram, rmsEtaHistogram, bias, rms);
276 smearedEta = gRandom->Gaus(eta+bias,rms);
277 smearFromHisto(pt, eta, phi, biasPhiHistogram, rmsPhiHistogram, bias, rms);
278 smearedPhi = gRandom->Gaus(phi+bias,rms);
279 break;
280 }
281}
