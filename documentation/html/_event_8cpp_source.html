198Event * Event::getEventStream(unsigned int index)
199{
200 //cout << "Event::getEventStream(unsigned int index) w/ index:" << index << endl;
201 unsigned int nStreams = eventStreamsStore.size();
202 Event * event;
203 if (nStreams == 0)
204 {
205 // none exist, create one
206 cout << "Event::getEventStream(unsigned int index) nStreams == 0; Create an event" << endl;
207 event = new Event();
208 event->setStreamIndex(index);
209 eventStreamsStore.push_back(event);
210 return event;
211 }
212 // streams already exist, find the one requested
213 //cout << "Event::getEventStream(unsigned int index) nStreams: " << nStreams << " Find it" << endl;
214
215 for (unsigned int k=0; k<nStreams; k++)
216 {
217 event = eventStreamsStore[k];
218 if (event->getStreamIndex() == index) return event;
219 }
220 //cout << "Event::getEventStream(unsigned int index) stream index " << index << " not found. Create it" << endl;
221
222 // stream does not exist, create it
223 event = new Event();
224 event->setStreamIndex(index);
225 eventStreamsStore.push_back(event);
226 return event;
227}
