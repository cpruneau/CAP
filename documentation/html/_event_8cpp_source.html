194Event * Event::getEventStream(unsigned int index)
195{
196 //cout << "Event::getEventStream(unsigned int index) w/ index:" << index << endl;
197 unsigned int nStreams = eventStreamsStore.size();
198 Event * event;
199 if (nStreams == 0)
200 {
201 // none exist, create one
202 cout << "Event::getEventStream(unsigned int index) nStreams == 0; Create an event" << endl;
203 event = new Event();
204 event->setStreamIndex(index);
205 eventStreamsStore.push_back(event);
206 return event;
207 }
208 // streams already exist, find the one requested
209 //cout << "Event::getEventStream(unsigned int index) nStreams: " << nStreams << " Find it" << endl;
210
211 for (unsigned int k=0; k<nStreams; k++)
212 {
213 event = eventStreamsStore[k];
214 if (event->getStreamIndex() == index) return event;
215 }
216 //cout << "Event::getEventStream(unsigned int index) stream index " << index << " not found. Create it" << endl;
217
218 // stream does not exist, create it
219 event = new Event();
220 event->setStreamIndex(index);
221 eventStreamsStore.push_back(event);
222 return event;
223}
