116void ParticlePairHistos::fill(Particle & particle1, Particle & particle2, double weight)
117{
118 ParticlePairAnalyzerConfiguration & ac = * (ParticlePairAnalyzerConfiguration*) getConfiguration();
119 TLorentzVector & momentum1 = particle1.getMomentum();
120 TLorentzVector & momentum2 = particle2.getMomentum();
121 double pt1 = momentum1.Pt();
122 double pt2 = momentum2.Pt();
123 h_n2_ptPt->Fill(pt1, pt2, weight);
124 if (ac.fillEta)
125 {
126 int iPhiEta1 = particle1.ixEtaPhi;
127 int iPhiEta2 = particle2.ixEtaPhi;
128 bool wrongix = iPhiEta1 < 0 || iPhiEta2 < 0;
129 if (!wrongix)
130 {
131 int binno = (iPhiEta2+1)*(h_n2_phiEtaPhiEta->GetNbinsX()+2)+(iPhiEta1+1);
132 double nentries = h_n2_phiEtaPhiEta->GetEntries() + 1;
133 h_n2_phiEtaPhiEta->AddBinContent(binno, weight);
134 h_n2_phiEtaPhiEta->SetEntries(nentries);
135 if (ac.fillP2)
136 {
137 h_ptn_phiEtaPhiEta ->AddBinContent(binno, weight*pt1);
138 h_npt_phiEtaPhiEta ->AddBinContent(binno, weight*pt2);
139 h_ptpt_phiEtaPhiEta->AddBinContent(binno, weight*pt1*pt2);
140 h_ptn_phiEtaPhiEta ->SetEntries(nentries);
141 h_npt_phiEtaPhiEta ->SetEntries(nentries);
142 h_ptpt_phiEtaPhiEta->SetEntries(nentries);
143 }
144 }
145 }
146 if (ac.fillY)
147 {
148 int iPhiY1 = particle1.ixYPhi;
149 int iPhiY2 = particle2.ixYPhi;
150 bool wrongix = iPhiY1 < 0 || iPhiY2 < 0;
151 if (!wrongix)
152 {
153 int binno = (iPhiY2+1)*(h_n2_phiYPhiY->GetNbinsX()+2)+(iPhiY1+1);
154 double nentries = h_n2_phiEtaPhiEta->GetEntries() + 1;
155 h_n2_phiYPhiY->AddBinContent(binno, weight);
156 h_n2_phiYPhiY->SetEntries(nentries);
157 if (ac.fillP2)
158 {
159 h_ptn_phiYPhiY ->AddBinContent(binno, weight*pt1);
160 h_npt_phiYPhiY ->AddBinContent(binno, weight*pt2);
161 h_ptpt_phiYPhiY->AddBinContent(binno, weight*pt1*pt2);
162 h_ptn_phiYPhiY->SetEntries(nentries);
163 h_npt_phiYPhiY->SetEntries(nentries);
164 h_ptpt_phiYPhiY->SetEntries(nentries);
165 }
166 }
167 }
