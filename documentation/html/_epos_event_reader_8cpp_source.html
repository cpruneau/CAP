101// Copy the event into Event for convenience...
103void EposEventReader::execute()
104{
105 if (reportDebug()) cout << "EposEventReader::execute() Started" << endl;
106
107 if (!fChain)
108 {
109 if (reportFatal()) cout << " EposEventReader::execute() no TChain available" << endl;
110 postTaskFatal();
111 return;
112 }
113
114 Factory<Particle> * particleFactory = Particle::getFactory();
115
116 bool seekingEvent = true;
117 while (seekingEvent)
118 {
119 // load another event from the root file/TTree
120 Long64_t ientry = LoadTree(jentry++);
121
122 // returning a null point is an indication that
123 // there are no more events in the file or stack of files.
124 if (ientry < 0)
125 {
126 postTaskEod(); // end of data
127 return;
128 }
129 nb = fChain->GetEntry(jentry); nbytes += nb;
130 //cout<<"sumit : "<<Mult<<endl;
131 // check if the event is acceptable.
132 event->impactParameter = impact;
133 if (eventFilter->accept(*event)) seekingEvent = false;
134 }
135
136 if (mult > arraySize)
137 {
138 if (reportError()) cout<< "EposEventReader::execute() n particles is " << mult << " and exceeds capacity " << arraySize << endl;
139 postTaskError();
140 return;
141 }
142 if (reportDebug()) cout<< "EposEventReader::execute() Impact par: "<<impact<<" "<<mult<<endl;
143
144 int thePid;
145 double charge, p_x, p_y, p_z, p_e, mass;
146 Particle * particle;
147 int particleAccepted = 0;
148
149 //------------------- Randomizing the particle phi --------------Starts
150 double eventAngle= TMath::TwoPi() * gRandom->Rndm();
151 double cosPhi = cos(eventAngle);
152 double sinPhi = sin(eventAngle);
153
154 // load particles from TTree and copy those that are selected into
155 // event
156 int iParticle=0;
157 bool readingEvent = true;
158 while (readingEvent)
159 {
160 particle = particleFactory->getNextObject();
161 bool seekParticle = true;
162 while (seekParticle)
163 {
164 thePid = pid[iParticle];
165 if ( thePid == 211 || thePid == 321 || thePid ==2212 )
166 charge = 1;
167 else if (thePid ==-211 || thePid ==-321 || thePid==-2212)
168 charge = -1;
169 else
170 charge = 0;
171 // only accept charged particles
172 if (charge==0) continue;
173 p_x = cosPhi*px[iParticle] - sinPhi*py[iParticle];
174 p_y = sinPhi*px[iParticle] + cosPhi*py[iParticle];
175 p_z = pz[iParticle];
176 p_e = e[iParticle];;
177 particle->setPidPxPyPzE(thePid, charge, p_x,p_y,p_z,p_e);
178 if (particleFilter->accept(*particle)) seekParticle = false;
179 if (reportDebug()) particle->printProperties(cout);
180 particleAccepted++;
181 iParticle++;
182 if (iParticle>=mult)
183 {
184 seekParticle = false;
185 readingEvent = false;
186 }
187 }
188 }
189 event->nParticles = particleAccepted;
190 if (reportDebug()) cout << "EposEventReader::execute() No of accepted Particles : "<< particleAccepted<<endl;
191 if (reportDebug()) cout << "EposEventReader::execute() event completed!" << endl;
192}
