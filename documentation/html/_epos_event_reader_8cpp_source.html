33void EposEventReader::execute()
34{
35 // if (reportDebug("EposEventReader",getName(),"execute()"))
36 // ;
37 incrementTaskExecuted();
38 EventFilter & eventFilter = * eventFilters[0];
39 ParticleFilter & particleFilter = * particleFilters[0];
40 incrementTaskExecuted();
41 Event & event = * eventStreams[0];
42 event.reset();
43 particleFactory->reset();
44 // resetParticleCounters();
45 Particle * parentInteraction;
46 parentInteraction = particleFactory->getNextObject();
47 parentInteraction->reset();
48 parentInteraction->setType( ParticleType::getInteractionType());
49 parentInteraction->setXYZT(0.0, 0.0, 0.0, 0.0);
50 event.add(parentInteraction);
51 Long64_t ientry = LoadTree(entryIndex);
52 if (ientry < 0)
53 {
54 postTaskEod(); return;
55 }
56 nb = rootInputTreeChain()->GetEntry(entryIndex++);
57 nBytes += nb;
58 if (nParticles > arraySize)
59 {
60 if (reportError("AmptEventReader",getName(),"execute()"))
61 cout<< "nParticles: " << nParticles << " exceeds capacity " << arraySize << endl;
62 postTaskFatal();
63 exit(1);
64 }
65
66 double eventPhi;
67 double cosPhi;
68 double sinPhi;
69 if (useRandomizeEventPlane())
70 {
71 eventPhi = getRandomEventPlaneAngle();
72 cosPhi = cos(eventPhi);
73 sinPhi = sin(eventPhi);
74 }
75
76 TLorentzVector & sourcePosition = parentInteraction->getPosition();
77 double r_x = sourcePosition.X();
78 double r_y = sourcePosition.Y();
79 double r_z = sourcePosition.Z();
80 double r_t = sourcePosition.T();
81 ParticleType * type;
82
83 for (int iParticle=0; iParticle<nParticles; iParticle++)
84 {
85 int pdgCode = pid[iParticle];
86 type = particleTypeCollection->findPdgCode(pdgCode);
87 if (type==nullptr)
88 {
89 if (reportWarning("AmptEventReader",getName(),"execute()")) cout << "Encountered unknown pdgCode: " << pdgCode << " Particle not added to event." << endl;
90 continue;
91 }
92
93 Particle * particle = particleFactory->getNextObject();
94 double p_x, p_y, p_z, p_e;
95 double mass = type->getMass();
96 if (useRandomizeEventPlane())
97 {
98 p_x = cosPhi*px[iParticle] - sinPhi*py[iParticle];
99 p_y = sinPhi*px[iParticle] + cosPhi*py[iParticle];
100 }
101 else
102 {
103 p_x = px[iParticle];
104 p_y = py[iParticle];
105 }
106 p_z = pz[iParticle];
107 p_e = sqrt(p_x*p_x + p_y*p_y + p_z*p_z + mass*mass);
108 particle->set(type,p_x,p_y,p_z,p_e,r_x,r_y,r_z,r_t,true);
109 if (!particleFilter.accept(*particle)) continue;
110 incrementNParticlesAccepted();
111 event.add(particle);
112 }
113 event.setEventNumber(events);
114 EventProperties & eventProperties = * event.getEventProperties();
115 eventProperties.zProjectile = 0;
116 eventProperties.aProjectile = 0;
117 eventProperties.nPartProjectile = 0;
118 eventProperties.zTarget = 0;
119 eventProperties.aTarget = 0;
120 eventProperties.nPartTarget = 0;
121 eventProperties.nParticipantsTotal = 0;
122 eventProperties.nBinaryTotal = 0;
123 eventProperties.impactParameter = impact;
124 eventProperties.fractionalXSection = -99999;
125// eventProperties.referenceMultiplicity = getNParticlesAccepted();
126// eventProperties.particlesCounted = getNParticlesCounted();
127// eventProperties.particlesAccepted = getNParticlesAccepted();
128 incrementNEventsAccepted(0);
129}
