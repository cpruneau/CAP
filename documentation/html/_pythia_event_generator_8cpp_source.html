134void PythiaEventGenerator::execute()
135{
136//
137// if (reportStart(__FUNCTION__))
138// ;
139 incrementTaskExecuted();
140 Event & event = *getEventStream(0);
141 EventProperties & eventProperties = * event.getEventProperties();
142
143 Particle * interaction;
144 // resetParticleCounters();
145 if (standaloneMode)
146 {
147 // In this mode, we generate one PYTHIA (pp) collision per event. One interaction vertex is
148 // inserted in the event stream and PYTHIA is called to carry out the particle generation.
149 // factory and event resets done by the task iterator to avoid repetition.
150 event.reset();
151 particleFactory->reset();
152 // resetParticleCounters();
153 interaction = particleFactory->getNextObject();
154 interaction->reset();
155 interaction->setType( ParticleType::getInteractionType());
156 interaction->setXYZT(0.0, 0.0, 0.0, 0.0);
157 event.add(interaction);
158 event.setNucleusA(1.0,1.0);
159 event.setNucleusB(1.0,1.0);
160 generate(interaction);
161 eventProperties.zProjectile = 1; // atomic number projectile
162 eventProperties.aProjectile = 1; // mass number projectile
163 eventProperties.nPartProjectile = 1; // number of participants projectile
164 eventProperties.zTarget = 1; // atomic number target
165 eventProperties.aTarget = 1; // mass number target
166 eventProperties.nPartTarget = 1; // number of participants target
167 eventProperties.nParticipantsTotal = 2; // total number of participants
168 eventProperties.nBinaryTotal = 1; // total number of binary collisions
169 eventProperties.impactParameter = -99999; // nucleus-nucleus center distance in fm
170 eventProperties.fractionalXSection = -99999; // fraction cross section value
171// eventProperties.referenceMultiplicity = getNParticlesAccepted(); // nominal multiplicity in the reference range
172// eventProperties.particlesCounted = getNParticlesCounted();
173// eventProperties.particlesAccepted = getNParticlesAccepted();
174 }
175 else
176 {
177 // In this mode, we generate several PYTHIA collisions per event. Interaction vertices and their locations
178 // are assumed to be already loaded in the event stream and we produce as many PYTHIA events as there
179 // are nucleon-nucleon interactions in the stream. Objects nucleusA and nucleusB are assumed defined by
180 // an earlier task and the structure EventProperties is assumed filled by that earlier task.He we only fill the number of
181 // particle generated and accepted (in addition to storing the generated particles in the event)
182
183 // May skip this event if it does not satisfy the event cut.
184 if (!eventFilters[0]->accept(event)) return;
185 if (event.getParticleCount() < 1)
186 {
187 return;
188 }
189 vector<Particle*> interactions = event.getNucleonNucleonInteractions();
190
191 unsigned int n = interactions.size();
192// if (reportWarning("PythiaEventGenerator",getName(),"execute()"))
193// cout << "Size of interactions:" << n << endl;
194 for (unsigned int kInter=0; kInter<n; kInter++)
195 {
196 generate(interactions[kInter]);
197 }
198// eventProperties.referenceMultiplicity = getNParticlesAccepted(); // nominal multiplicity in the reference range
199// eventProperties.particlesCounted = getNParticlesCounted();
200// eventProperties.particlesAccepted = getNParticlesAccepted();
201 }
202 incrementNEventsAccepted(0);
203 if (reportDebug(__FUNCTION__)) eventProperties.printProperties(cout);
204}

221 if (nParticles>nMaxClonesArray)

227 if (dataOutputUsed)
