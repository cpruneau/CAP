113void ParticlePerformanceHistos::loadHistograms(TFile * inputFile)
114{
115 if ( reportStart("ParticlePerformanceHistos",getName(),"loadHistograms(TFile * inputFile)"))
116 { }
117 TString fct = "loadHistograms(TFile * inputFile)";
118 if (!ptrFileExist(fct,inputFile)) return;
119 TString bn = getHistoBaseName();
120
121 Configuration & configuration = getConfiguration();
122 fillEta = configuration.getValueBool("fillEta");
123 fillY = configuration.getValueBool("fillY");
124
125 h_n1_dPt = loadH1(inputFile, bn+TString("n1_dpt"));
126 h_n1_dPhi = loadH1(inputFile, bn+TString("n1_dPhi"));
127 h_n1_dPtVsPt = loadH2(inputFile, bn+TString("n1_dpt"));
128 h_n1_dPhiVsPt = loadH2(inputFile, bn+TString("n1_dpt"));
129
130 if (fillEta)
131 {
132 h_n1_dEta = loadH1(inputFile, bn+TString("n1_deta"));
133 h_n1_dEtaVsEta = loadH2(inputFile, bn+TString("n1_dEtaVsEta"));
134 }
135
136 if (fillY)
137 {
138 h_n1_dY = loadH1(inputFile, bn+TString("n1_dY"));
139 h_n1_dYVsY = loadH2(inputFile, bn+TString("n1_dYVsY"));
140 }
141
142 if ( reportEnd("ParticlePerformanceHistos",getName(),"loadHistograms(TFile * inputFile)"))
143 { }

149void ParticlePerformanceHistos::fill(Particle & recoParticle, double weight)
150{
151 Particle & genParticle = *recoParticle.getTruth();
152 TLorentzVector & recoMomentum = recoParticle.getMomentum();
153 TLorentzVector & genMomentum = genParticle.getMomentum();
154
155
156 double recoPt = recoMomentum.Pt();
157 double recoEta = recoMomentum.Eta();
158 double recoY = recoMomentum.Y();
159 double recoPhi = recoMomentum.Pt();
160
161 double genPt = genMomentum.Pt();
162 double genEta = genMomentum.Eta();
163 double genY = genMomentum.Y();
164 double genPhi = genMomentum.Pt();
165
166 double dPt = recoPt - genPt;
167 double dPhi = recoPhi - genPhi;
168 double dEta = recoEta - genEta;
169 double dY = recoY - genY;
170
171 h_n1_dPt->Fill(dPt,weight);
172 h_n1_dPhi->Fill(dPhi,weight);
173 h_n1_dPtVsPt->Fill(genPt,dPt,weight);
174 h_n1_dPhiVsPt->Fill(genPt,dPhi,weight);
175
176 if (fillEta)
177 {
178 h_n1_dEta->Fill(dEta,weight);
179 h_n1_dEtaVsEta->Fill(genEta,dEta,weight);
180 }
181
182 if (fillY)
183 {
184 h_n1_dY->Fill(dY,weight);
185 h_n1_dYVsY->Fill(genY,dY,weight);
186 }
187}
