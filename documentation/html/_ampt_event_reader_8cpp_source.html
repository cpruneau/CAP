100// Read an ampt event from file
101// Copy the event into Event for convenience...
103void AmptEventReader::execute()
104{
105 if (reportDebug()) cout << "AmptEventReader::execute() Started" << endl;
106 if (!fChain)
107 {
108 if (reportFatal()) cout << " AmptEventReader::execute() no TChain available" << endl;
109 postTaskFatal();
110 return;
111 }
112
113 Factory<Particle> * particleFactory = Particle::getFactory();
114
115 bool seekingEvent = true;
116 while (seekingEvent)
117 {
118 // load another event from the root file/TTree
119 Long64_t ientry = LoadTree(jentry++);
120
121 // returning a null point is an indication that
122 // there are no more events in the file or stack of files.
123 if (ientry < 0)
124 {
125 postTaskEod(); // end of data
126 return;
127 }
128 nb = fChain->GetEntry(jentry); nbytes += nb;
129 if (mult > arraySize)
130 {
131 if (reportError()) cout<< "AmptEventReader::execute() n particles is " << mult << " and exceeds capacity " << arraySize << endl;
132 postTaskError();
133 return;
134 }
135 event->eventNumber = eventNo;
136 event->impactParameter = impact;
137 event->nProjectile = Nproj;
138 event->nTarget = Ntarg;
139 event->nParticleTotal = Nparttotal;
140 event->multiplicity = mult;
141 if (eventFilter->accept(*event)) seekingEvent = false;
142 }
143 int thePid;
144 double charge, p_x, p_y, p_z, p_e, mass;
145 Particle aParticle;
146 Particle * particle;
147 int particleAccepted = 0;
148
149 //------------------- Randomizing the particle phi --------------Starts
150 double eventAngle= TMath::TwoPi() * gRandom->Rndm();
151 double cosPhi = cos(eventAngle);
152 double sinPhi = sin(eventAngle);
153
154 for (int iParticle; iParticle<mult; iParticle++)
155 {
156 thePid = pid[iParticle];
157 if ( thePid == 211 || thePid == 321 || thePid ==2212 )
158 charge = 1;
159 else if (thePid ==-211 || thePid ==-321 || thePid==-2212)
160 charge = -1;
161 else
162 charge = 0;
163 p_x = cosPhi*px[iParticle] - sinPhi*py[iParticle];
164 p_y = sinPhi*px[iParticle] + cosPhi*py[iParticle];
165 p_z = pz[iParticle];
166 mass = m[iParticle];
167 p_e =sqrt(p_x*p_x + p_y*p_y + p_z*p_z + mass*mass);
168 aParticle.setPidPxPyPzE(thePid, charge, p_x,p_y,p_z,p_e);
169 //if (reportDebug("AmptEventReader","execute()",getName())) aParticle.printProperties(cout);
170 if (particleFilter->accept(aParticle))
171 {
172 particle = particleFactory->getNextObject();
173 *particle = aParticle;
174 particleAccepted++;
175 }
176 }
177 event->nParticles = particleAccepted;
178 if (reportDebug("AmptEventReader","execute()",getName()))
179 {
180 event->printProperties(cout);
181 cout << "AmptEventReader::execute() event completed!" << endl;
182 }
183}

192Long64_t AmptEventReader::LoadTree(Long64_t entry)
193{
194 // Set the environment to read one entry
195 if (!fChain) return -5;

198 if (fChain->GetTreeNumber() != fCurrent) {

220 fChain->SetBranchAddress("py", py, &b_py);
221 fChain->SetBranchAddress("pz", pz, &b_pz);
222 fChain->SetBranchAddress("m", m, &b_m);
223 fChain->SetBranchAddress("Nx", Nx, &b_Nx);
