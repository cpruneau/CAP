35void AmptEventReader::execute()
36{
37 // if (reportDebug("AmptEventReader",getName(),"execute()"))
38 // ;
39 incrementTaskExecuted();
40 EventFilter & eventFilter = * eventFilters[0];
41 ParticleFilter & particleFilter = * particleFilters[0];
42 incrementTaskExecuted();
43 Event & event = * eventStreams[0];
44 event.reset();
45 particleFactory->reset();
46 // resetParticleCounters();
47 Particle * parentInteraction;
48 parentInteraction = particleFactory->getNextObject();
49 parentInteraction->reset();
50 parentInteraction->setType( ParticleType::getInteractionType());
51 parentInteraction->setXYZT(0.0, 0.0, 0.0, 0.0);
52 event.add(parentInteraction);
53 Long64_t ientry = LoadTree(entryIndex);
54 if (ientry < 0)
55 {
56 postTaskEod(); return;
57 }
58 nb = rootInputTreeChain()->GetEntry(entryIndex++);
59 nBytes += nb;
60 if (nParticles > arraySize)
61 {
62 if (reportError("AmptEventReader",getName(),"execute()"))
63 cout<< "nParticles: " << nParticles << " exceeds capacity " << arraySize << endl;
64 postTaskFatal();
65 exit(1);
66 }
67
68 double eventPhi;
69 double cosPhi;
70 double sinPhi;
71 if (useRandomizeEventPlane())
72 {
73 eventPhi = getRandomEventPlaneAngle();
74 cosPhi = cos(eventPhi);
75 sinPhi = sin(eventPhi);
76 }
77
78 TLorentzVector & sourcePosition = parentInteraction->getPosition();
79 double r_x = sourcePosition.X();
80 double r_y = sourcePosition.Y();
81 double r_z = sourcePosition.Z();
82 double r_t = sourcePosition.T();
83 ParticleType * type;
84 //if (reportDebug("AmptEventReader",getName(),"execute()")) cout << " Starting loop with nParticles: " << nParticles << endl;
85 for (int iParticle=0; iParticle<nParticles; iParticle++)
86 {
87 int pdgCode = pid[iParticle];
88 type = particleTypeCollection->findPdgCode(pdgCode);
89 if (type==nullptr)
90 {
91 if (reportWarning("AmptEventReader",getName(),"execute()")) cout << "Encountered unknown pdgCode: " << pdgCode << " Particle not added to event." << endl;
92 continue;
93 }
94 Particle * particle = particleFactory->getNextObject();
95 double p_x, p_y, p_z, p_e, mass;
96 if (useRandomizeEventPlane())
97 {
98 p_x = cosPhi*px[iParticle] - sinPhi*py[iParticle];
99 p_y = sinPhi*px[iParticle] + cosPhi*py[iParticle];
100 }
101 else
102 {
103 p_x = px[iParticle];
104 p_y = py[iParticle];
105 }
106 p_z = pz[iParticle];
107 mass = m[iParticle];
108 p_e =sqrt(p_x*p_x + p_y*p_y + p_z*p_z + mass*mass);
109 particle->set(type,p_x,p_y,p_z,p_e,r_x,r_y,r_z,r_t,true);
110 if (!particleFilter.accept(*particle)) continue;
111 event.add(particle);
112 incrementNParticlesAccepted();
113 }
114 event.setEventNumber(eventNo);
115 EventProperties & eventProperties = * event.getEventProperties();
116 eventProperties.zProjectile = 0;
117 eventProperties.aProjectile = 0;
118 eventProperties.nPartProjectile = Nproj;
119 eventProperties.zTarget = 0;
120 eventProperties.aTarget = 0;
121 eventProperties.nPartTarget = Ntarg;
122 eventProperties.nParticipantsTotal = Nproj+Ntarg;
123 eventProperties.nBinaryTotal = 0;
124 eventProperties.impactParameter = impact;
125 eventProperties.fractionalXSection = -99999;
126 // eventProperties.referenceMultiplicity = getNParticlesAccepted();
127 // eventProperties.particlesCounted = getNParticlesCounted();
128 // eventProperties.particlesAccepted = getNParticlesAccepted();
129 incrementNEventsAccepted(0);
130 // if (reportDebug("AmptEventReader",getName(),"execute()"))
131 // {
132 // eventProperties.printProperties(cout);
133 // cout << "AmptEventReader::execute() event completed!" << endl;
134 // }
135}
