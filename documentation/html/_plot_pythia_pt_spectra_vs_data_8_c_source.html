19int openRootInputFiles(TString & inputPathName,
20 vector<TString> & fileNames,
21 vector<TFile*> & inputFiles)
22{
23 cout << "<I> openRootInputFiles(...) Opening files..." << endl;
24 TString fileName;
25 for (unsigned int iFile=0;iFile<fileNames.size();iFile++)
26 {
27 fileName = inputPathName+fileNames[iFile];
28 TFile * file = new TFile(fileName,"READ");
29 if (file)
30 {
31 cout << "<I> openRootInputFiles() Successfully opened: " << fileName << endl;
32 inputFiles.push_back(file);
33 //file->ls();
34 }
35 else
36 {
37 cout << "<E> openRootInputFiles() Unable to open: " << fileName << endl;
38 return 1;
39 }
40 }
41 return 0;
42}
