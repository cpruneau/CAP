110void ParticleAnalyzer::execute()
111{
112 incrementEventProcessed();
113 bool accept;
114 unsigned int nEventFilters = eventFilters.size();
115 unsigned int nParticleFilters = particleFilters.size();
116 Event * event = eventStreams[0];
117 for (unsigned int iEventFilter=0; iEventFilter<nEventFilters; iEventFilter++ )
118 {
119 if (!eventFilters[iEventFilter]->accept(*event)) continue;
120 incrementEventAccepted(iEventFilter);
121 vector<double> nAccepted(nParticleFilters,0.0);
122 vector<double> totalEnergy(nParticleFilters,0.0);
123 unsigned int nParticles = event->getNParticles();
124 for (unsigned int iParticle=0; iParticle<nParticles; iParticle++)
125 {
126 Particle & particle = * event->getParticleAt(iParticle);
127 // particle.printProperties(cout);
128 for (unsigned int iParticleFilter=0; iParticleFilter<nParticleFilters; iParticleFilter++ )
129 {
130 ParticleHistos * histos = (ParticleHistos *) histograms[iParticleFilter+iEventFilter*nParticleFilters];
131 accept = particleFilters[iParticleFilter]->accept(particle);
132 if (accept)
133 {
134 nAccepted[iParticleFilter]++;
135 totalEnergy[iParticleFilter] += particle.getMomentum().E();
136 histos->fill(particle,1.0);
137 }
138 }
139 }
140 for (unsigned int iParticleFilter=0; iParticleFilter<nParticleFilters; iParticleFilter++ )
141 {
142 ParticleHistos * histos = (ParticleHistos *) histograms[iParticleFilter+iEventFilter*nParticleFilters];
143 histos->fillMultiplicity(nAccepted[iParticleFilter],totalEnergy[iParticleFilter],1.0);
144 }
145 }
