115 // Generate an event
117 Event * generate()
118 {
119 if (debugLevel>0) cout << "-DEBUG- UrqmdEventReader::generate() Started" << endl;
120
121 if (!fChain)
122 {
123 cout << "-FATAL- UrqmdEventReader::generate() no TChain available" << endl;
124 }
125
126 bool seekingEvent = true;
127 while (seekingEvent)
128 {
129 // load another event from the root file/TTree
130 Long64_t ientry = LoadTree(jentry);
131
132 // returning a null point is an indication that
133 // there are no more events in the file or stack of files.
134 if (ientry < 0) return 0;
135
136 nb = fChain->GetEntry(jentry++); nbytes += nb;
137 if (nb == 0) return 0;
138
139 if(anEvent->GetImpactParameter() > generatorConfiguration->impactParameterMinimum &&
140 anEvent->GetImpactParameter() <= generatorConfiguration->impactParameterMaximum) seekingEvent = false;
141 }
142
143 if (debugLevel>0) cout << "-DEBUG- UrqmdEventReader::generate() new event b: " << anEvent->GetImpactParameter() << ", ntracks: " << anEvent->GetNtrack() << endl;
144
146 double charge;
147
148 event->reset();
149 particleFactory->reset();
150 int particleAccepted = 0;
151 nleptons = 0L;
152
153 //------------------- Randomising the phi --------------Starts
154 float eventAngle=gRandom->Uniform(0,3.14);
155 //cout<< "------------------> "<<eventAngle <<endl;
156 //------------------- Randomising the phi --------------Ends
157
158
159 // load particles from TTree and copy those that are selected into event
160 Particle aParticle; /* we use a particle object to extract angular information for indexes extraction */
161 Particle *particle; /* this is the actual particle which will be stored */
162 for (int iPart1=0; iPart1 < anEvent->GetNtrack(); iPart1++)
163 {
164 UrQMDParticle *aUrQMDParticle = (UrQMDParticle *) anEvent->GetTracks()->At(iPart1);
165
166 charge = aUrQMDParticle->GetCharge();
167 if (charge==0) continue;
168
169 double p_x = aUrQMDParticle->GetPx();
170 double p_y = aUrQMDParticle->GetPy();
171 double p_z = aUrQMDParticle->GetPz();
172 double p_e = aUrQMDParticle->GetE();
173
174 aParticle.pid = aUrQMDParticle->GetItyp();
175 aParticle.charge = charge;
176 aParticle.setPxPyPzE(p_x,p_y,p_z,p_e);
177 aParticle.ixEtaPhi = -1;
178 aParticle.ixYPhi = -1;
179
180 if (aParticle.pt<generatorConfiguration->ptMinimum || aParticle.pt>generatorConfiguration->ptMaximum) continue;
181 if (aParticle.y<generatorConfiguration->yMinimum || aParticle.y>generatorConfiguration->yMaximum) continue;
182 if (aParticle.eta<generatorConfiguration->etaMinimum || aParticle.eta>generatorConfiguration->etaMaximum) continue;
183 if (!acceptPID(aParticle)) continue;
184
185 /* rotate according to the event plane angle */
186 double phi_new = (aParticle.phi-eventAngle);
187 double p_x_new= aParticle.pt*TMath::Cos(phi_new);
188 double p_y_new= aParticle.pt*TMath::Sin(phi_new);
189 aParticle.setPxPyPzE(p_x_new,p_y_new,p_z,p_e);
190
191 particle = particleFactory->getObject();
192 particle->Copy(aParticle);
193
194 //particle->printProperties(cout);
195 particleAccepted++;
196 }
197 event->nParticles = particleAccepted;
198
199 if (debugLevel>0) cout << "-DEBUG- UrqmdEventReader::generate() No of accepted Particles: "<<particleAccepted<<endl;
200 if (debugLevel>0) cout << "-DEBUG- UrqmdEventReader::generate() No of rejected leptons: " << nleptons << endl;
201 //event->printProperties(cout);
202 if (debugLevel>0) cout << "-DEBUG- UrqmdEventReader::generate() event completed!" << endl;
203 return event;
204 }
