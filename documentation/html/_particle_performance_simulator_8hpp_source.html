105 virtual double getEfficiency(double pt, double eta __attribute__((unused)), double phi __attribute__((unused)))
106 {
107 double eff;
108 double arg1 = pt-peakPt;
109 if (pt<peakPt)
110 {
111 double arg2 = arg1/peakRms;
112 eff = peakAmp*exp(-arg2*arg2/2.0);
113 }
114 else
115 {
116 eff = peakAmp + a1*arg1 + a2*arg1*arg1;
117 }
118 return (eff<1)? eff : 1.0 ;
119 }

167 virtual double getBias(double pt, double eta, double phi __attribute__((unused)))
168 {
169 double bias = 0.0;
170 switch (option)
171 {
172 case 0: bias = aInvBias/pt + a0Bias + a1Bias*pt + a2Bias*pt*pt; break;
173 case 1: bias = a0Bias + a1Bias*eta + a2Bias*eta*eta; break;
174 }
175 return bias;
176 }

184 case 1: rms = a0Rms + a1Rms*eta + a2Rms*eta*eta; break;
185 }
186 return rms;
