87 inputDataFile->GetObject(dataInputTreeName,inputTree);

91 cout << "No inputTree named: " << dataInputTreeName << " in file: " << inputFileName << endl;
92 postTaskFatal();
93 return;
94 }
95 Init(inputTree);
96 if (reportDebug("PythiaEventReader",getName(),"initialize()")) cout << "PythiaEventReader::initialize() Completed" << endl;

104void PythiaEventReader::execute()
105{
106 if (reportStart("PythiaEventReader",getName(),"execute()"))
107 ;
108
109 cout << "eventStreams.size() : " << eventStreams.size() << endl;
110
111 Event & event = * eventStreams[0];
112 event.reset();
113 particleFactory->reset();
114 int nparts;
115 bool seekingEvent = true;
116 //if (reportDebug("PythiaEventReader",getName(),"execute()")) cout << "Start seek loop" << endl;
117 while (seekingEvent)
118 {
119 //if (reportDebug("PythiaEventReader",getName(),"execute()")) cout << "jentry:" << jentry << endl;
120 // load another event from the root file/TTree
121 Long64_t ientry = LoadTree(jentry++);
122 //if (reportDebug("PythiaEventReader",getName(),"execute()")) cout << "ientry:" << ientry << endl;
123 // returning a null point is an indication that
124 // there are no more events in the file or stack of files.
125 if (ientry < 0)
126 {
127 postTaskEod(); // end of data
128 return;
129 }
130 nb = fChain->GetEntry(jentry); nbytes += nb;
131 nparts = particles_;
132 if (reportDebug("PythiaEventReader",getName(),"execute()")) cout << " nb:" << nb << " nparts:" << nparts << endl;
133 if (nparts>2) seekingEvent = false;
134 }
135
136 int thePid;
137 double charge, baryon, mass, p_x, p_y, p_z, p_e;
138 ParticleType * type;
139 Particle * particle;
140 int particleAccepted = 0;
141 int particleCounted = 0;
142
143 // load particles from TClone storage and copy into event.
144 //if (reportDebug()) cout << "PythiaEventGenerator::execute() starting copy loop into event..." << endl;
145
146 for (int iParticle = 0; iParticle < nparts; iParticle++)
147 {
148// if (reportDebug("PythiaEventReader",getName(),"execute()")) cout << "iParticle: " << iParticle << endl;
149
150 int ist = particles_fStatusCode[iParticle];
151 if (ist <= 0) continue;
152 int pdg = particles_fPdgCode[iParticle];
153 mass = TDatabasePDG::Instance()->GetParticle(pdg)->Mass();
154 if (mass<0.002) continue; // no photons, electrons...
155 charge = TDatabasePDG::Instance()->GetParticle(pdg)->Charge()/3.0;
156 double px = particles_fPx[iParticle];
157 double py = particles_fPy[iParticle];
158 double pz = particles_fPz[iParticle];
159 double e = particles_fE[iParticle];
160 double sourceX = 0.0;
161 double sourceY = 0.0;
162 double sourceZ = 0.0;
163 double sourceT = 0.0;
164 //aParticle.setPidPxPyPzE(pdg,p_x,p_y,p_z,p_e);
165
166 type = masterCollection->findPdgCode(pdg);
167 if (type==nullptr) continue;
168 particle = particleFactory->getNextObject();
169 particle->set(type,px,py,pz,e,sourceX,sourceY,sourceZ,sourceT,true);
170 incrementParticlesCounted(); // photons are NOT included in this tally
171 if (!particleFilter->accept(*particle)) continue;
172 event.add(particle);
173 incrementParticlesAccepted();
174 }
175 if (reportDebug("PythiaEventReader",getName(),"execute()"))
176 {
177 cout << endl;
178 cout << "PythiaEventGenerator::execute() No of accepted particles : "<< getNParticlesAccepted() << endl;
179 cout << "PythiaEventGenerator::execute() No of counted particles : " << getNParticlesCounted() << endl;
180 }
181}

190Long64_t PythiaEventReader::LoadTree(Long64_t entry)
191{
192 // Set the environment to read one entry
193 if (!fChain) return -5;
194 Long64_t centry = fChain->LoadTree(entry);
195 if (centry < 0) return centry;
196 if (fChain->GetTreeNumber() != fCurrent) {
197 fCurrent = fChain->GetTreeNumber();
198 Notify();
199 }
200 return centry;
201}
