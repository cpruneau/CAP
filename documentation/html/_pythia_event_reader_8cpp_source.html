40void PythiaEventReader::execute()
41{
42 if (reportStart(__FUNCTION__))
43 ;
44
45 //cout << "eventStreams.size() : " << eventStreams.size() << endl;
46 incrementTaskExecuted();
47 Event & event = * eventStreams[0];
48 event.reset();
49 particleFactory->reset();
50 bool seekingEvent = true;
51 //if (reportDebug("PythiaEventReader",getName(),"execute()")) cout << "Start seek loop" << endl;
52 while (seekingEvent)
53 {
54 //if (reportDebug("PythiaEventReader",getName(),"execute()")) cout << "jentry:" << entryIndex << endl;
55 // load another event from the root file/TTree
56 Long64_t ientry = LoadTree(entryIndex++);
57 //if (reportDebug("PythiaEventReader",getName(),"execute()")) cout << "ientry:" << ientry << endl;
58 // returning a null point is an indication that
59 // there are no more events in the file or stack of files.
60 if (ientry < 0)
61 {
62 postTaskEod(); // end of data
63 return;
64 }
65 nb = rootInputTreeChain()->GetEntry(entryIndex); nBytes += nb;
66 if (reportDebug(__FUNCTION__)) cout << " nb:" << nb << " nParticles:" << nParticles << endl;
67 if (nParticles>2) seekingEvent = false;
68 }
69
70 int thePid;
71 double charge, baryon, mass, p_x, p_y, p_z, p_e;
72 ParticleType * type;
73 Particle * particle;
74 int particleAccepted = 0;
75 int particleCounted = 0;
76
77 for (int iParticle = 0; iParticle < nParticles; iParticle++)
78 {
79 // if (reportDebug("PythiaEventReader",getName(),"execute()")) cout << "iParticle: " << iParticle << endl;
80
81 int ist = particles_fStatusCode[iParticle];
82 if (ist <= 0) continue;
83 int pdg = particles_fPdgCode[iParticle];
84 type = particleTypeCollection->findPdgCode(pdg);
85 if (type==nullptr) continue;
86 mass = type->getMass();
87 if (mass<0.002) continue; // no photons, electrons...
88 charge = type->getCharge();
89 double px = particles_fPx[iParticle];
90 double py = particles_fPy[iParticle];
91 double pz = particles_fPz[iParticle];
92 double e = particles_fE[iParticle];
93 double sourceX = 0.0;
94 double sourceY = 0.0;
95 double sourceZ = 0.0;
96 double sourceT = 0.0;
97 particle = particleFactory->getNextObject();
98 particle->set(type,px,py,pz,e,sourceX,sourceY,sourceZ,sourceT,true);
99 // incrementParticlesCounted(); // photons are NOT included in this tally
100 if (!particleFilters[0]->accept(*particle)) continue;
101 event.add(particle);
102 // // incrementParticlesAccepted();
103 }
104}
