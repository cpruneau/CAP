356vector<TString> Configuration::getSelectedValues(const TString keyBaseName, const TString defaultValue) const
357{
358 vector<TString> selectedValues;
359 for (std::map<TString,TString>::const_iterator it=stringMap.cbegin(); it!=stringMap.cend(); ++it)
