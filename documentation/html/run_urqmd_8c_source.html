132// Run an analysis with the Cluster Model
134void runUrqmd(const char *source, const char *outdir, const char *productiondir, Int_t centix, long nEventsRequested=5000000000)
135{
136 cout << "-INFO- runClusterModelAnalysis() Setup configuration" << endl;
137
138 // ---------------------------
139 // debugLevel Levels
140 // ---------------------------
141 bool debugLoop = true;
142 bool debugGenerator = true;
143 bool debugAnalyzer = false;
144
145 // ---------------------------
146 // Graphics parameters
147 // ---------------------------
148 setDefaultOptions(1);
149
150 // ---------------------------
151 // Histogram parameters
152 // ---------------------------
153
154 /* Cummulate errors by default */
155 TH1::SetDefaultSumw2(kTRUE);
156
157 AnalysisConfiguration * ac = new AnalysisConfiguration();
158 ac->outputPath = TString::Format("%s/%s/",outdir,productiondir).Data(); // "../OUTPUT/";
159 ac->baseName = TString::Format("Urqmd_%s",cent[centix]).Data();
160 ac->configFileName = TString::Format("Config_%s.txt",cent[centix]).Data();
161 ac->rootOutputFileName = TString::Format("Urqmd_Histograms_%s.root",cent[centix]).Data();
162
163 cout<<"================================================================"<<endl;
164 cout<<""<<endl;
165 cout<<" running for = "<<ac->baseName<<endl;
166 cout<<""<<endl;
167 cout<<"================================================================"<<endl;
168 ac->nBins_nCell = 100;
169 ac->min_nCell = 0.0;
170 ac->max_nCell = 100.0;
171 ac->nBins_nPartPerCell = 100;
172 ac->min_nPartPerCell = 0.0;
173 ac->max_nPartPerCell = 100.0;
174
175 ac->nBins_pt = 18;
176 ac->min_pt = 0.2;
177 ac->max_pt = 2.0;
178 ac->nBins_eta = 32;
179 ac->min_eta = -1.6;
180 ac->max_eta = 1.6;
181 ac->nBins_y = 40;
182 ac->min_y = -4.0;
183 ac->max_y = 4.0;
184 ac->nBins_phi = 72;
185 ac->min_phi = 0.0;
186 ac->max_phi = 2.0*3.1415927;
187
188 ac->nBins_DeltaPlong = 10;
189 ac->min_DeltaPlong = -1.0;
190 ac->max_DeltaPlong = 1.0;
191 ac->nBins_DeltaPside = 10;
192 ac->min_DeltaPside = -1.0;
193 ac->max_DeltaPside = 1.0;
194 ac->nBins_DeltaPout = 10;
195 ac->min_DeltaPout = -1.0;
196 ac->max_DeltaPout = 1.0;
197
198 ac->fillPairs = true;
199 ac->fill3D = false;
200 ac->fill6D = false;
201 ac->fillQ3D = false;
202 ac->fillY = false;
203
204 ac->scaleHistograms = false;
205 ac->externalScaling = 1.0;
206 ac->saveHistosToRootFile = true;
207 ac->forceRewrite = true;
208 ac->saveHistosToTextFile = false;
209 ac->calculateDerived = false;
210 ac->plotGraphs = false;
211
212 // use the following options to select which graphs are plotted on screen.
213 ac->plotSingles = true; // selects the single part histo to be printed
214 ac->plotPairs = true; // selects the part pair histo to be printed
215 ac->plotDerivedPairs = true; // selects the derived part pair histo to be printed
216 ac->plotCombined = true;
217 ac->plotEventHistos = true; // global event stuff
218
219 ac->printGraphs = false;
220 ac->plotAll = false;
221 ac->printAll = false;
222
223 ac->particleFilter1 = new ParticleFilter( 1,ac->min_pt+0.001,ac->max_pt,ac->min_eta,ac->max_eta,ac->min_y,ac->max_y); // +ve only
224 ac->particleFilter2 = new ParticleFilter(-1,ac->min_pt+0.001,ac->max_pt,ac->min_eta,ac->max_eta,ac->min_y,ac->max_y); // -ve only
225 ac->particlePairFilter12 = new ParticlePairFilter(0.0, 100.0,
226 0.0, 2.0*3.1415927,
227 0.0, 6.0);
228
229 ac->canvasConfiguration1D = new CanvasConfiguration(1);
230 ac->graphConfiguration1D = new GraphConfiguration(1,1);
231 ac->canvasConfiguration2D = new CanvasConfiguration(0);
232 ac->graphConfiguration2D = new GraphConfiguration(2);
233
234 // ==================================================================================
235 // Urqmd Model Selection Parameters
236 // ==================================================================================
237 GeneratorConfiguration * mc = new GeneratorConfiguration(5,"UrqmdModel");
238 // ---------------------------
239 // Cell Parameters
240 // ---------------------------
241 mc->impactParameterMinimum = lowb[centix];
242 mc->impactParameterMaximum = highb[centix];
243 mc->ptMinimum = ac->min_pt;
244 mc->ptMaximum = ac->max_pt;
245 mc->yMinimum = ac->min_y;
246 mc->yMaximum = ac->max_y;
247 mc->etaMinimum = ac->min_eta;
248 mc->etaMaximum = ac->max_eta;
249 mc->eventsSource = source;
250
251 MasterAnalysis * analysis = new MasterAnalysis(mc,ac, debugLoop, debugGenerator,debugAnalyzer);
252 analysis->run(nEventsRequested);
253
254 if (analysis->eventsAnalyzed != 0) {
255 gSystem->Exec(TString::Format("echo \"%s%s, %ld\" >> %s/fileslist_%s.txt",
256 ac->outputPath.Data(),
257 ac->rootOutputFileName.Data(),
258 analysis->eventsAnalyzed,
259 productiondir,
260 cent[centix]).Data());
261 }
262}
