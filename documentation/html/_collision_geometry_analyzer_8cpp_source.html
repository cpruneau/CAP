145void CollisionGeometryAnalyzer::execute()
146{
147 if (reportStart("CollisionGeometryAnalyzer",getName(),"execute()"))
148 ;
149 incrementEventProcessed();
150 unsigned int nEventFilters = eventFilters.size();
151 Event & event = *eventStreams[0];
152 for (unsigned int iEventFilter=0; iEventFilter<nEventFilters; iEventFilter++ )
153 {
154 if (eventFilters[iEventFilter]->accept(event))
155 {
156 incrementEventAccepted(iEventFilter); // count eventStreams used to fill histograms and for scaling at the end...
157 CollisionGeometryHistograms * histos = (CollisionGeometryHistograms *) histograms[iEventFilter];
158 histos->fill(event,1.0);
159 }
160 else
161 {
162 CollisionGeometryHistograms * histos = (CollisionGeometryHistograms *) histograms[iEventFilter];
163 histos->noFill(event,1.0);
164 }
165 }
166 if (reportEnd("CollisionGeometryAnalyzer",getName(),"execute()"))
167 ;
168}
