148void CollisionGeometryAnalyzer::execute()
149{
150
151 if (reportStart(__FUNCTION__))
152 ;
153 incrementTaskExecuted();
154 Event & event = *eventStreams[0];
155 for (int iEventFilter=0; iEventFilter<nEventFilters; iEventFilter++ )
156 {
157 if (eventFilters[iEventFilter]->accept(event))
158 {
159 incrementNEventsAccepted(iEventFilter); // count eventStreams used to fill histograms and for scaling at the end...
160 CollisionGeometryHistograms * histos = (CollisionGeometryHistograms *) histograms[iEventFilter];
161 histos->fill(event,1.0);
162 }
163 else
164 {
165 CollisionGeometryHistograms * histos = (CollisionGeometryHistograms *) histograms[iEventFilter];
166 histos->noFill(event,1.0);
167 }
168 }
169 if (reportEnd(__FUNCTION__))
170 ;
171}
