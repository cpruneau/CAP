42void ParticleTypeCollection::readFromFile(const TString & inputFileName)
43{
44 cout << "<I> ParticleTypeCollection::readFromFile() Reading particle list from file:" << inputFileName << endl;
45 ifstream inputFile(inputFileName.Data());
46 int pdgCode;
47 string name;
48 string title;
49 double mass, width;
50 int gSpin, gIsospin;
51 int baryon, strange, charm, bottom, charge;
52 int nDecayModes, decayNpart;
53 double decayBranchingRatio;
54 int decayPart[5] = {0, 0, 0, 0, 0};
55 ParticleType * particleType;
56 ParticleType * antiParticleType;
57 int dummy_int;
58 TString theName;
59 TString theAntiName;
60 TString theTitle;
61 TString theAntiTitle;
62 while (1)
63 {
64 //cout << "Reading in ParticleType resonance decay table... - 2 -" << endl;
65 inputFile >> pdgCode;
66 if (inputFile.eof())
67 {
68 cout << "Reach EOF" << endl;
69 break;
70 }
71 inputFile >> name;
72 inputFile >> mass;
73 inputFile >> width;
74 inputFile >> gSpin;
75 inputFile >> baryon;
76 inputFile >> strange;
77 inputFile >> charm;
78 inputFile >> bottom;
79 inputFile >> gIsospin;
80 inputFile >> charge;
81 inputFile >> nDecayModes;
82 inputFile >> title;
83 theName = name;
84 theTitle = title;
85 particleType= new ParticleType(pdgCode, theName, theTitle, mass, width, gSpin, baryon, strange,
86 charm, bottom, gIsospin, charge);
87 push_back(particleType);
88 if (particleType->isFermion())
89 {
90 theAntiName = "Anti-";
91 theAntiName += theName;
92 theAntiTitle = "#bar ";
93 theAntiTitle += theTitle;
94 antiParticleType = new ParticleType(-pdgCode, theAntiName, theAntiTitle, mass, width, gSpin,
95 -baryon, -strange, -charm, -bottom, gIsospin,
96 -charge);
97 push_back(antiParticleType);
98 }
99 // read decay information
100 for (int j = 0; j < nDecayModes; j++)
101 {
102 inputFile >> dummy_int;
103 inputFile >> decayNpart;
104 inputFile >> decayBranchingRatio;
105 inputFile >> decayPart[0];
106 inputFile >> decayPart[1];
107 inputFile >> decayPart[2];
108 inputFile >> decayPart[3];
109 inputFile >> decayPart[4];
110 decayNpart = abs(decayNpart);
111 std::vector<int> children;
112 std::vector<int> antiChildren;
113 for (int k=0; k<decayNpart; k++)
114 {
115 children.push_back(decayPart[k]);
116 }
117 if (!particleType->isFermion())
118 {
119 particleType->addDecayMode(decayBranchingRatio,children);
120 }
121 else
122 {
123 for (int k=0; k<decayNpart; k++)
124 {
125 antiChildren.push_back(findPdgCode(decayPart[k])->getAntiParticlePdgCode());
126 }
127 particleType->addDecayMode(decayBranchingRatio,children);
128 antiParticleType->addDecayMode(decayBranchingRatio,antiChildren);
129 }
130 }
131 }
132 //cout << "Reading in ParticleType resonance decay table... - 5- " << endl;
133 inputFile.close();
134 //particleList.erase(particleList.begin()); // delete gamma
135 cout << "<I> Total number of Hadrons: " << size() << endl;
136 resolveTypes();
137 sortByMass();
138
139 // setup the decay montecarlo probabilities....
140 for (unsigned int k=0; k<size(); k++)
141 {
142 objects[k]->setupDecayGenerator();
143 }
144
145 cout << "ParticleTypeCollection::readFromFile(const TString & inputFileName) Completed." << endl;
146}

207// ================================================================================================
208// Extract a subset of this collection based on the given ctriterion
209// option : types selected
210// 0 : stable particle types
211// 1 : weakly stable
212// 2 : unstable
213// 3 : mesons
214// 4 : baryons
215// 5 : leptons
216// 6 : fermions
217// 7 : bosons
218// ================================================================================================
219ParticleTypeCollection * ParticleTypeCollection::extractCollection(int option)
220{
221 ParticleTypeCollection * collection = new ParticleTypeCollection();
222 unsigned int n = size();
223 for (unsigned int iPart = 1; iPart < n; iPart++)
224 {
225 ParticleType * type = objects[iPart];
226 switch (option)
227 {
228 case 0: if (type->isStable() && !type->isPhoton()) collection->push_back(type); break;
229 case 1: if (type->isWeakStable() && !type->isPhoton()) collection->push_back(type); break;
230 case 2: if (!type->isStable()) collection->push_back(type); break;
231 case 3: if (!type->isWeakStable()) collection->push_back(type); break;
232 case 4: if (type->isMeson()) collection->push_back(type); break;
233 case 5: if (type->isBaryon()) collection->push_back(type); break;
234 case 6: if (type->isLepton()) collection->push_back(type); break;
235 case 7: if (type->isFermion()) collection->push_back(type); break;
236 case 8: if (type->isBoson()) collection->push_back(type); break;
237 }
238 }
239 return collection;
240}
