40bool EventFilter::accept(const Event & event)
41{
42 unsigned int nComponents = getNConditions();
43 if (nComponents<1) return true;
44 EventProperties * eventProperties = event.getEventProperties();
45 if (!eventProperties)
46 {
47 cout << " Event does NOT have properties" << endl;
48 return false;
49 }
50 double value;
51 for (unsigned int k = 0; k<getNConditions(); k++)
52 {
53 Condition & condition = *(conditions[k]);
54 unsigned index = condition.filterSubtype;
55 switch (condition.filterType)
56 {
57 case 0:
58 // model parameter
59 if (index>=eventProperties->modelParameters.size())
60 {
61 cout << "<E> EventFilter::accept(Event & event) index>=modelParameters.size()" << endl;
62 return false;
63 }
64 value = eventProperties->modelParameters[index]; break;
65 case 1:
66 // filtered n
67 if (index>=eventProperties->nFiltered.size())
68 {
69 cout << "<E> EventFilter::accept(Event & event) index>=eventProperties->nFiltered.size()" << endl;
70 return false;
71 }
72 value = eventProperties->nFiltered[index]; break;
73 case 2:
74 // filtered energy
75 if (index>=eventProperties->eFiltered.size())
76 {
77 cout << "<E> EventFilter::accept(Event & event) index>=eventProperties->eFiltered.size()" << endl;
78 return false;
79 }
80 value = eventProperties->eFiltered[index]; break;
81 case 3:
82 // filtered charge
83 if (index>=eventProperties->qFiltered.size())
84 {
85 cout << "<E> EventFilter::accept(Event & event) index>=eventProperties->qFiltered.size()" << endl;
86 return false;
87 }
88 value = eventProperties->qFiltered[index]; break;
89 case 4:
90 // filtered strangeness
91 if (index>=eventProperties->sFiltered.size())
92 {
93 cout << "<E> EventFilter::accept(Event & event) index>=eventProperties->sFiltered.size()" << endl;
94 return false;
95 }
96 value = eventProperties->sFiltered[index]; break;
97 case 5:
98 // filtered strangeness
99 if (index>=eventProperties->bFiltered.size())
100 {
101 cout << "<E> EventFilter::accept(Event & event) index>=eventProperties->bFiltered.size()" << endl;
102 return false;
103 }
104 value = eventProperties->bFiltered[index]; break;
105 }
106 if (!condition.accept(value)) return false;
107 }
108 return true;
109}
