64void ParticleDerivedHistos::createHistograms()
65{
66 if ( reportStart("ParticleDerivedHistos",getName(),"createHistograms()"))
67 { }
68 TString bn = getHistoBaseName();
69 const Configuration & configuration = getConfiguration();
70 fillEta = configuration.getValueBool("fillEta");
71 fillY = configuration.getValueBool("fillY");
72 fillP2 = configuration.getValueBool("fillP2");
73 nBins_pt = configuration.getValueInt("nBins_pt");
74 min_pt = configuration.getValueDouble("min_pt");
75 max_pt = configuration.getValueDouble("max_pt");
76 nBins_phi = configuration.getValueInt("nBins_phi");
77 min_phi = configuration.getValueDouble("min_phi");
78 max_phi = configuration.getValueDouble("max_phi");
79 nBins_eta = configuration.getValueInt("nBins_eta");
80 min_eta = configuration.getValueDouble("min_eta");
81 max_eta = configuration.getValueDouble("max_eta");
82 nBins_y = configuration.getValueInt("nBins_y");
83 min_y = configuration.getValueDouble("min_y");
84 max_y = configuration.getValueDouble("max_y");
85
86 h_n1_phi = createHistogram(bn+TString("n1_phi"), nBins_phi, min_phi, max_phi, "#varphi","#rho_{1}(#varphi)");
87
88 if (fillP2)
89 {
90 h_spt_phi = createHistogram(bn+TString("sumpt1_phi"), nBins_phi, min_phi, max_phi, "#varphi","#sum p_{T}");
91 h_pt_phi = createHistogram(bn+TString("pt_phi"), nBins_phi, min_phi, max_phi, "#varphi","<p_{T}>");
92 }
93
94 if (fillEta)
95 {
96 h_n1_eta = createHistogram(bn+TString("n1_eta"), nBins_eta, min_eta, max_eta, "#eta", "#rho_{1}(#eta)");
97 if (fillP2)
98 {
99 h_spt_eta = createHistogram(bn+TString("sumpt1_eta"), nBins_eta, min_eta, max_eta, "#eta", "#sum p_{T}");
100 h_pt_eta = createHistogram(bn+TString("pt_eta"), nBins_eta, min_eta, max_eta, "#eta", "<p_{T}>");
101 h_pt_phiEta = createHistogram(bn+TString("pt_phiEta"), nBins_eta, min_eta, max_eta, nBins_phi, min_phi, max_phi, "#eta", "#varphi","N");
102 }
103 }
104 if (fillY)
105 {
106 h_n1_y = createHistogram(bn+TString("n1_y"), nBins_y, min_y, max_y, "y", "#rho_{1}(y)");
107 if (fillP2)
108 {
109 h_spt_y = createHistogram(bn+TString("sumpt1_y"), nBins_eta, min_eta, max_eta, "y", "#sum p_{T}");
110 h_pt_y = createHistogram(bn+TString("pt_y"), nBins_y, min_y, max_y, "y", "<p_{T}>");
111 h_pt_phiY = createHistogram(bn+TString("pt_phiY"), nBins_y, min_y, max_y, nBins_phi, min_phi, max_phi, "y", "#varphi","N");
112 }
113 }
114 if ( reportEnd("ParticleDerivedHistos",getName(),"createHistograms()"))
115 { }
116}

119void ParticleDerivedHistos::loadHistograms(TFile * inputFile)
120{
121 if (reportStart("ParticleDerivedHistos",getName(),"loadHistograms(TFile * inputFile)"))
122 { }
123 TString fct = "loadHistograms(TFile * inputFile)";
124 if (!ptrFileExist(fct,inputFile)) return;
125 TString bn = getHistoBaseName();
126 h_n1_phi = loadH1(inputFile, bn+TString("n1_phi"));
127 h_spt_phi = loadH1(inputFile, bn+TString("spt_phi"));
128 h_pt_phi = loadH1(inputFile, bn+TString("pt_phi"));
129
130 if (fillEta)
131 {
132 h_n1_eta = loadH1(inputFile, bn+TString("n1_eta"));
133 if (fillP2)
134 {
135 h_spt_eta = loadH1(inputFile, bn+TString("spt_eta"));
136 h_pt_eta = loadH1(inputFile, bn+TString("pt_eta"));
137 h_pt_phiEta = loadH2(inputFile, bn+TString("pt_phiEta"));
138 }
139 }
140 if (fillY)
141 {
142 h_n1_y = loadH1(inputFile, bn+TString("n1_y"));
143 if (fillP2)
144 {
145 h_spt_y = loadH1(inputFile, bn+TString("spt_y"));
146 h_pt_y = loadH1(inputFile, bn+TString("pt_y"));
147 h_pt_phiY = loadH2(inputFile, bn+TString("pt_phiY"));
148 }
149 }
150
151 if ( reportEnd("ParticleDerivedHistos",getName(),"loadHistograms(TFile * inputFile)"))
152 { }
