84 cout << "-INFO- runClusterModelAnalysis() Setup configuration" << endl;

86 // ---------------------------

107 ac->baseName = TString::Format("Urqmd_%s",centname).Data();
108 ac->configFileName = TString::Format("Config_%d.txt",fileix).Data();
109 ac->rootOutputFileName = TString::Format("Urqmd_Histograms_%d.root",fileix).Data();
110
111 cout<<"================================================================"<<endl;
112 cout<<""<<endl;
113 cout<<" running for = "<<ac->baseName<<endl;
114 cout<<""<<endl;
115 cout<<"================================================================"<<endl;
116 ac->nBins_nCell = 100;
117 ac->min_nCell = 0.0;
118 ac->max_nCell = 100.0;
119 ac->nBins_nPartPerCell = 100;
120 ac->min_nPartPerCell = 0.0;

123 ac->nBins_pt = 18;
124 ac->min_pt = 0.2;
125 ac->max_pt = 2.0;
126 ac->nBins_eta = 32;

128 ac->max_eta = 1.6;
129 ac->nBins_y = 40;
130 ac->min_y = -4.0;
131 ac->max_y = 4.0;
132 ac->nBins_phi = 72;
133 ac->min_phi = 0.0;
134 ac->max_phi = 2.0*3.1415927;
135
136 ac->nBins_DeltaPlong = 10;
137 ac->min_DeltaPlong = -1.0;
138 ac->max_DeltaPlong = 1.0;
139 ac->nBins_DeltaPside = 10;

144 ac->max_DeltaPout = 1.0;
145
146 ac->fillPairs = true;
147 ac->fill3D = false;
148 ac->fill6D = false;
149 ac->fillQ3D = false;

151
152 ac->scaleHistograms = false;
153 ac->externalScaling = 1.0;
154 ac->saveHistosToRootFile = true;

156 ac->saveHistosToTextFile = false;
157 ac->calculateDerived = false;
158 ac->plotGraphs = false;
159
160 // use the following options to select which graphs are plotted on screen.
161 ac->plotSingles = true; // selects the single part histo to be printed
162 ac->plotPairs = true; // selects the part pair histo to be printed
163 ac->plotDerivedPairs = true; // selects the derived part pair histo to be printed
164 ac->plotCombined = true;

167 ac->printGraphs = false;

172 ac->particleFilter2 = new ParticleFilter(-1,ac->min_pt+0.001,ac->max_pt,ac->min_eta,ac->max_eta,ac->min_y,ac->max_y); // -ve only

197 mc->eventsSource = source;
198
199 MasterAnalysis * analysis = new MasterAnalysis(mc,ac, debugLoop, debugGenerator,debugAnalyzer);
200 analysis->run(nEventsRequested);
201
202 if (analysis->eventsAnalyzed != 0) {

205 ac->rootOutputFileName.Data(),
206 analysis->eventsAnalyzed,
207 productiondir,
208 centname));
209 }
210}
