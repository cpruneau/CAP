62void ParticleHistos::createHistograms()
63{
64 if ( reportStart("ParticleHistos",getName(),"createHistograms()"))
65 { }
66 TString bn = getHistoBaseName();
67 Configuration & configuration = getConfiguration();
68
69 nBins_n1 = configuration.getValueInt("nBins_n1");
70 min_n1 = configuration.getValueDouble("min_n1");
71 max_n1 = configuration.getValueDouble("max_n1");
72
73 nBins_pt = configuration.getValueInt("nBins_pt");
74 min_pt = configuration.getValueDouble("min_pt");
75 max_pt = configuration.getValueDouble("max_pt");
76 range_pt = max_pt - min_pt;
77
78 nBins_phi = configuration.getValueInt("nBins_phi");
79 min_phi = configuration.getValueDouble("min_phi");
80 max_phi = configuration.getValueDouble("max_phi");
81 range_phi = max_phi - min_phi;
82
83 nBins_eta = configuration.getValueInt("nBins_eta");
84 min_eta = configuration.getValueDouble("min_eta");
85 max_eta = configuration.getValueDouble("max_eta");
86 range_eta = max_eta - min_eta;
87
88 nBins_y = configuration.getValueInt("nBins_y");
89 min_y = configuration.getValueDouble("min_y");
90 max_y = configuration.getValueDouble("max_y");
91 range_y = max_y - min_y;
92
93 fillEta = configuration.getValueBool("fillEta");
94 fillY = configuration.getValueBool("fillY");
95 fillP2 = configuration.getValueBool("fillP2");
96
97 h_n1 = createHistogram(bn+TString("n1"), nBins_n1, min_n1, max_n1, "n_1","N");
98 h_n1_eTotal = createHistogram(bn+TString("n1_eTotal"), nBins_n1, min_n1, 10.0*max_n1, "n1_eTotal","N");
99 h_n1_pt = createHistogram(bn+TString("n1_pt"), nBins_pt, min_pt, max_pt, "p_{T}","N");
100 h_n1_ptXS = createHistogram(bn+TString("n1_ptXS"), nBins_pt, min_pt, max_pt, "p_{T}","1/p_{T} dN/p_{T}");
101
102 if (fillEta)
103 {
104 h_n1_phiEta = createHistogram(bn+TString("n1_phiEta"), nBins_eta, min_eta, max_eta, nBins_phi, min_phi, max_phi, "#eta", "#varphi","N");
105 if (fillP2)
106 {
107 h_spt_phiEta = createHistogram(bn+TString("spt_phiEta"), nBins_eta, min_eta, max_eta, nBins_phi, min_phi, max_phi, "#eta", "#varphi","N");
108 }
109 }
110 if (fillY)
111 {
112 h_n1_phiY = createHistogram(bn+TString("n1_phiY"), nBins_y, min_y, max_y, nBins_phi, min_phi, max_phi, "y", "#varphi","N");
113 if (fillP2)
114 {
115 h_spt_phiY = createHistogram(bn+TString("spt_phiY"), nBins_y, min_y, max_y, nBins_phi, min_phi, max_phi, "y", "#varphi","N");
116 }
117 }
118 if ( reportEnd("ParticleHistos",getName(),"createHistograms()"))
119 { }
120}

167void ParticleHistos::fill(vector<ParticleDigit*> & particles, double weight)
168{
169 double nSingles = 0;
170 double totalEnergy = 0;
171
172 for (unsigned int iPart=0; iPart<particles.size(); iPart++)
173 {
174 float e = particles[iPart]->e;
175 float pt = particles[iPart]->pt;
176 unsigned int iPt = particles[iPart]->iPt;
177 unsigned int iPhi = particles[iPart]->iPhi;
178 unsigned int iEta = particles[iPart]->iEta;
179 unsigned int iY = particles[iPart]->iY;
180
181 nSingles++;
182 totalEnergy += e;
183
184 int iG = h_n1_pt->GetBin(iPt);
185 h_n1_pt ->AddBinContent(iG,weight);
186 h_n1_ptXS->AddBinContent(iG,weight/pt);
187
188 if (fillEta)
189 {
190 iG = h_n1_phiEta->GetBin(iEta,iPhi);
191 if (iG<=0)
192 {
193 cout << "iG:" << iG << endl;
194 }
195 h_n1_phiEta->AddBinContent(iG,weight);
196 if (fillP2) h_spt_phiEta->AddBinContent(iG,weight*pt);
197 }
198
199 if (fillY)
200 {
201 iG = h_n1_phiY->GetBin(iY,iPhi);
202 if (iG<=0)
203 {
204 cout << "iG:" << iG << endl;
205 }
206 h_n1_phiY->AddBinContent(iG,weight);
207 if (fillP2) h_spt_phiY->AddBinContent(iG,weight*pt);
208 }
209 }
210 h_n1_pt->SetEntries(h_n1_pt->GetEntries()+nSingles);
211 h_n1_ptXS->SetEntries(h_n1_ptXS->GetEntries()+nSingles);
212 if (fillEta)
213 {
214 h_n1_phiEta->SetEntries(h_n1_phiEta->GetEntries()+nSingles);
215 if (fillP2) h_spt_phiEta->SetEntries(h_spt_phiEta->GetEntries()+nSingles);
216 }
217 if (fillY)
218 {
219 h_n1_phiY->SetEntries(h_n1_phiY->GetEntries()+nSingles);
220 if (fillP2) h_spt_phiY->SetEntries(h_spt_phiY->GetEntries()+nSingles);
221 }
222 h_n1->Fill(nSingles, weight);
223 h_n1_eTotal->Fill(totalEnergy, weight);
224}

229void ParticleHistos::fill(Particle & particle, double weight)
230{
231 TLorentzVector & momentum = particle.getMomentum();
232 float pt = momentum.Pt();
233 float eta = momentum.Eta();
234 float phi = momentum.Phi();
235 if (phi<0) phi += TMath::TwoPi();
236 h_n1_pt ->Fill(pt,weight);
237 h_n1_ptXS->Fill(pt,weight/pt);
238 if (fillEta)
239 {
240 h_n1_phiEta->Fill(eta,phi,weight);
241 if (fillP2) h_spt_phiEta->Fill(eta,phi,weight*pt);
242 }
243 if (fillY)
244 {
245 float rapidity = momentum.Rapidity();
246 h_n1_phiY->Fill(rapidity,phi,weight);
247 if (fillP2) h_spt_phiY->Fill(rapidity,phi,weight*pt);
248 }
249}

258 h_n1_eTotal->Fill(totalEnergy, weight);
