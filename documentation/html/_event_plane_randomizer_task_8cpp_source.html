41void EventPlaneRandomizerTask::execute()
42{
43 double eventAngle= TMath::TwoPi() * gRandom->Rndm();
44 Event * event = eventStreams[0];
45 unsigned int nParticles = event->getNParticles();
46 for (unsigned int iParticle = 0; iParticle < nParticles; iParticle++)
47 {
48 Particle * particle = event->getParticleAt(iParticle);
49 particle->getPosition().RotateZ(eventAngle);
50 particle->getMomentum().RotateZ(eventAngle);
51 }
52 incrementEventProcessed();
53}
