33void UrqmdEventReader::execute()
34{
35 // if (reportDebug("UrqmdEventReader",getName(),"execute()"))
36 // ;
37 incrementTaskExecuted();
38 EventFilter & eventFilter = * eventFilters[0];
39 ParticleFilter & particleFilter = * particleFilters[0];
40 incrementTaskExecuted();
41 Event & event = * eventStreams[0];
42 event.reset();
43 particleFactory->reset();
44 // resetParticleCounters();
45 Particle * parentInteraction;
46 parentInteraction = particleFactory->getNextObject();
47 parentInteraction->reset();
48 parentInteraction->setType( ParticleType::getInteractionType());
49 parentInteraction->setXYZT(0.0, 0.0, 0.0, 0.0);
50 event.add(parentInteraction);
51 Long64_t ientry = LoadTree(entryIndex);
52 if (ientry < 0)
53 {
54 postTaskEod(); return;
55 }
56 nb = rootInputTreeChain()->GetEntry(entryIndex++);
57 nBytes += nb;
58 int nParticles = fTracks_;
59
60 if (nParticles > arraySize)
61 {
62 if (reportError("AmptEventReader",getName(),"execute()"))
63 cout<< "nParticles: " << nParticles << " exceeds capacity " << arraySize << endl;
64 postTaskFatal();
65 exit(1);
66 }
67
68 double eventPhi;
69 double cosPhi;
70 double sinPhi;
71 if (useRandomizeEventPlane())
72 {
73 eventPhi = getRandomEventPlaneAngle();
74 cosPhi = cos(eventPhi);
75 sinPhi = sin(eventPhi);
76 }
77
78 TLorentzVector & sourcePosition = parentInteraction->getPosition();
79 double r_x;// = sourcePosition.X();
80 double r_y;// = sourcePosition.Y();
81 double r_z;// = sourcePosition.Z();
82 double r_t;// = sourcePosition.T();
83 ParticleType * type;
84
85 for (int iParticle=0; iParticle<nParticles; iParticle++)
86 {
87 int pdgCode = pid[iParticle];
88 type = particleTypeCollection->findPdgCode(pdgCode);
89 if (type==nullptr)
90 {
91 if (reportWarning("AmptEventReader",getName(),"execute()")) cout << "Encountered unknown pdgCode: " << pdgCode << " Particle not added to event." << endl;
92 continue;
93 }
94
95 Particle * particle = particleFactory->getNextObject();
96 double p_x, p_y, p_z, p_e;
97 double mass = type->getMass();
98 if (useRandomizeEventPlane())
99 {
100 p_x = cosPhi*px[iParticle] - sinPhi*py[iParticle];
101 p_y = sinPhi*px[iParticle] + cosPhi*py[iParticle];
102 }
103 else
104 {
105 p_x = px[iParticle];
106 p_y = py[iParticle];
107 }
108 p_z = pz[iParticle];
109 p_e =sqrt(p_x*p_x + p_y*p_y + p_z*p_z + mass*mass);
110 particle->set(type,p_x,p_y,p_z,p_e,r_x,r_y,r_z,r_t,true);
111 incrementParticlesCounted(); // photons are NOT included in this tally
112 if (!particleFilter.accept(*particle)) continue;
113 event.add(particle);
114 // // incrementParticlesAccepted();
115 }
116 event.setEventNumber(events);
117 EventProperties & eventProperties = * event.getEventProperties();
118 eventProperties.zProjectile = 0;
119 eventProperties.aProjectile = 0;
120 eventProperties.nPartProjectile = 0;
121 eventProperties.zTarget = 0;
122 eventProperties.aTarget = 0;
123 eventProperties.nPartTarget = 0;
124 eventProperties.nParticipantsTotal = 0;
125 eventProperties.nBinaryTotal = 0;
126 eventProperties.impactParameter = impact;
127 eventProperties.fractionalXSection = -99999;
128// eventProperties.referenceMultiplicity = getNParticlesAccepted();
129// eventProperties.particlesCounted = getNParticlesCounted();
130// eventProperties.particlesAccepted = getNParticlesAccepted();
131 incrementNEventsAccepted(0);
132}

159Event * generate()
160{
161 if (debugLevel>0) cout << "-DEBUG- UrqmdEventReader::generate() Started" << endl;
162
163 if (!fChain)
164 {
165 cout << "-FATAL- UrqmdEventReader::generate() no TChain available" << endl;
166 }
167
168 bool seekingEvent = true;
169 while (seekingEvent)
170 {
171 // load another event from the root file/TTree
172 Long64_t ientry = LoadTree(jentry);
173
174 // returning a null point is an indication that
175 // there are no more events in the file or stack of files.
176 if (ientry < 0) return 0;
177
178 nb = fChain->GetEntry(jentry++); nbytes += nb;
179 if (nb == 0) return 0;
180
181 if(anEvent->GetImpactParameter() > generatorConfiguration->impactParameterMinimum &&
182 anEvent->GetImpactParameter() <= generatorConfiguration->impactParameterMaximum) seekingEvent = false;
183 }
184
185 if (debugLevel>0) cout << "-DEBUG- UrqmdEventReader::generate() new event b: " << anEvent->GetImpactParameter() << ", ntracks: " << anEvent->GetNtrack() << endl;
186
188 double charge;
189
190 event->reset();
191 particleFactory->reset();
192 int particleAccepted = 0;
193 nleptons = 0L;
194
195 //------------------- Randomising the phi --------------Starts
196 float eventAngle=gRandom->Uniform(0,3.14);
197 //cout<< "------------------> "<<eventAngle <<endl;
198 //------------------- Randomising the phi --------------Ends
199
200
201 // load particles from TTree and copy those that are selected into event
202 Particle aParticle; /* we use a particle object to extract angular information for indexes extraction */
203 Particle *particle; /* this is the actual particle which will be stored */
204 for (int iPart1=0; iPart1 < anEvent->GetNtrack(); iPart1++)
205 {
206 UrQMDParticle *aUrQMDParticle = (UrQMDParticle *) anEvent->GetTracks()->At(iPart1);
207
208 charge = aUrQMDParticle->GetCharge();
209 if (charge==0) continue;
210
211 double p_x = aUrQMDParticle->GetPx();
212 double p_y = aUrQMDParticle->GetPy();
213 double p_z = aUrQMDParticle->GetPz();
214 double p_e = aUrQMDParticle->GetE();
215
216 aParticle.pid = aUrQMDParticle->GetItyp();
217 aParticle.charge = charge;
218 aParticle.setPxPyPzE(p_x,p_y,p_z,p_e);
219 aParticle.ixEtaPhi = -1;
220 aParticle.ixYPhi = -1;
221
222 if (aParticle.pt<generatorConfiguration->ptMinimum || aParticle.pt>generatorConfiguration->ptMaximum) continue;
223 if (aParticle.y<generatorConfiguration->yMinimum || aParticle.y>generatorConfiguration->yMaximum) continue;
224 if (aParticle.eta<generatorConfiguration->etaMinimum || aParticle.eta>generatorConfiguration->etaMaximum) continue;
225 if (!acceptPID(aParticle)) continue;
226
227 /* rotate according to the event plane angle */
228 double phi_new = (aParticle.phi-eventAngle);
229 double p_x_new= aParticle.pt*TMath::Cos(phi_new);
230 double p_y_new= aParticle.pt*TMath::Sin(phi_new);
231 aParticle.setPxPyPzE(p_x_new,p_y_new,p_z,p_e);
232
233 particle = particleFactory->getObject();
234 particle->Copy(aParticle);
235
236 //particle->printProperties(cout);
237 particleAccepted++;
238 }
239 event->nParticles = particleAccepted;
240
241 if (debugLevel>0) cout << "-DEBUG- UrqmdEventReader::generate() No of accepted Particles: "<<particleAccepted<<endl;
242 if (debugLevel>0) cout << "-DEBUG- UrqmdEventReader::generate() No of rejected leptons: " << nleptons << endl;
243 //event->printProperties(cout);
244 if (debugLevel>0) cout << "-DEBUG- UrqmdEventReader::generate() event completed!" << endl;
245 return event;
246}
