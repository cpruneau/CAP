204void ParticlePairAnalyzer::execute()
205{
206 incrementEventProcessed();
207 ParticlePairAnalyzerConfiguration & ac = * (ParticlePairAnalyzerConfiguration *) getConfiguration();
208 Event & event = *eventStreams[0];
209 unsigned int nEventFilters = eventFilters.size();
210 unsigned int nParticleFilters = particleFilters.size();
211 unsigned int nParticles = event.getNParticles();
212
213 /* before filtering let's build the particle indexes to speed up the process */
214 for (unsigned int iParticle = 0; iParticle < nParticles; iParticle++)
215 {
216 Particle & particle = *event.getParticleAt(iParticle);
217 particle.ixEtaPhi = ac.getIxEtaPhi(particle.getMomentum().Eta(),particle.getMomentum().Phi());
218 particle.ixYPhi = ac.getIxYPhi(particle.getMomentum().Rapidity(),particle.getMomentum().Phi());
219 }
220
221 unsigned int index;
222 for (unsigned int iEventFilter=0; iEventFilter<nEventFilters; iEventFilter++ )
223 {
224 if (!eventFilters[iEventFilter]->accept(event)) continue;
225 incrementEventAccepted(iEventFilter); // count eventStreams used to fill histograms and for scaling at the end...
226 vector<double> nAccepted(nParticleFilters,0.0);
227 vector<double> totalEnergy(nParticleFilters,0.0);
228
229 unsigned int baseSingle = iEventFilter*nHistos;
230 unsigned int basePair = iEventFilter*nHistos+nParticleFilters;
231
232 // Fill singles
233
234 //cout << "ParticlePairAnalyzer::execute() ----------------------------------------- " << endl;
235 //cout << "ParticlePairAnalyzer::execute() nParticleFilters:" << nParticleFilters << endl;
236 for (unsigned int iParticle1=0; iParticle1<nParticles; iParticle1++)
237 {
238 Particle & particle1 = * event.getParticleAt(iParticle1);
239 //cout << "ParticlePairAnalyzer::execute() iParticle1:" << iParticle1 << endl;
240 for (unsigned int iParticleFilter1=0; iParticleFilter1<nParticleFilters; iParticleFilter1++)
241 {
242 //cout << "ParticlePairAnalyzer::execute() iParticleFilter1:" << iParticleFilter1 << endl;
243
244 if (particleFilters[iParticleFilter1]->accept(particle1))
245 {
246 //cout << "ParticlePairAnalyzer::execute() accepted:" << endl;
247 nAccepted[iParticleFilter1]++;
248 totalEnergy[iParticleFilter1] += particle1.getMomentum().E();
249 index = baseSingle+iParticleFilter1;
250 ParticleHistos * histos = (ParticleHistos *) histograms[index];
251 histos->fill(particle1, 1.0);
252 }
253 }
254 }
255
256 for (unsigned int iParticleFilter=0; iParticleFilter<nParticleFilters; iParticleFilter++ )
257 {
258 index = baseSingle+iParticleFilter;
259 ParticleHistos * histos = (ParticleHistos *) histograms[index];
260 histos->fillMultiplicity(nAccepted[iParticleFilter],totalEnergy[iParticleFilter],1.0);
261 }
262
263 //ParticleHistos * histos = (ParticleHistos *) histograms[iEventFilter*nHistos+iParticleFilter1];
264 //histos->fill(nAccepted,totalEnergy, 1.0);
265
266 // Fill pairs
267 for (unsigned int iParticle1=0; iParticle1<nParticles; iParticle1++)
268 {
269 Particle & particle1 = * event.getParticleAt(iParticle1);
270 for (unsigned int iParticle2=0; iParticle2<nParticles; iParticle2++)
271 {
272 if (iParticle1==iParticle2) continue;
273 Particle & particle2 = * event.getParticleAt(iParticle2);
274 for (unsigned int iParticleFilter1=0; iParticleFilter1<nParticleFilters; iParticleFilter1++)
275 {
276 if (!particleFilters[iParticleFilter1]->accept(particle1)) continue;
277 for (unsigned int iParticleFilter2=0; iParticleFilter2<nParticleFilters; iParticleFilter2++)
278 {
279 if (!particleFilters[iParticleFilter2]->accept(particle2)) continue;
280 index = basePair+iParticleFilter1*nParticleFilters+iParticleFilter2;
281 ParticlePairHistos * histos = (ParticlePairHistos *) histograms[index];
282 histos->fill(particle1,particle2, 1.0);
283 } // iParticleFilter2
284 } // iParticleFilter1
285 } //iParticle2
286 } // iParticle1
287 } // eventFilters
288}

294void ParticlePairAnalyzer::createDerivedHistograms()
295{
296 unsigned int nEventFilters = eventFilters.size();
297 unsigned int nParticleFilters = particleFilters.size();
298 if (reportInfo("ParticlePairAnalyzer",getName(),"createDerivedHistograms()"))
299 {
300 cout << " nEventFilters:" << nEventFilters << endl;
301 cout << " nParticleFilters:" << nParticleFilters << endl;
302 cout << " nHistos:" << nHistos << endl;
303 cout << " nDerivedHistos:" << nDerivedHistos << endl;
304 cout << " nCombinedHistos:" << nCombinedHistos << endl;
305 }
306 ParticlePairAnalyzerConfiguration & ac = *(ParticlePairAnalyzerConfiguration *) getConfiguration();
307 for (unsigned int iEventFilter=0; iEventFilter<nEventFilters; iEventFilter++ )
308 {
309 if (reportInfo("ParticlePairAnalyzer",getName(),"createDerivedHistograms()"))
310 {
311 cout << " iEventFilter:" << iEventFilter << endl;
312 cout << " Do Singles:" << endl;
313 }
314 unsigned int baseSingle = iEventFilter*nHistos;
315 unsigned int basePair = iEventFilter*nHistos+nParticleFilters;
316 unsigned int baseDerivedPair = iEventFilter*nDerivedHistos;
317 unsigned int baseCombinedPair = iEventFilter*nCombinedHistos;
318
319 // singles
320 for (unsigned int iParticleFilter1=0; iParticleFilter1<nParticleFilters; iParticleFilter1++)
321 {
322 ParticleHistos * histos = (ParticleHistos *) histograms[baseSingle+iParticleFilter1];
323 if (reportInfo("ParticlePairAnalyzer",getName(),"createDerivedHistograms()"))
324 {
325 cout << " iParticleFilter1:" << iParticleFilter1 << endl;
326 cout << " completeFill():" << endl;
327 }
328 histos->completeFill();
329 if (reportInfo("ParticlePairAnalyzer",getName(),"createDerivedHistograms()"))
330 {
331 cout << " calculateAverages():" << endl;
332 }
333 histos->calculateAverages();
334
335 }
336
337 // pairs
338 if (reportInfo("ParticlePairAnalyzer",getName(),"createDerivedHistograms()"))
339 {
340 cout << " Do Pairs:" << endl;
341 }
342 for (unsigned int iParticleFilter1=0; iParticleFilter1<nParticleFilters; iParticleFilter1++)
343 {
344 ParticleHistos * histos1 = (ParticleHistos *) histograms[baseSingle+iParticleFilter1];
345 for (unsigned int iParticleFilter2=0; iParticleFilter2<nParticleFilters; iParticleFilter2++)
346 {
347 if (reportInfo("ParticlePairAnalyzer",getName(),"createDerivedHistograms()"))
348 {
349 cout << "iParticleFilter1:" << iParticleFilter1 << "iParticleFilter2:" << iParticleFilter2 << endl;
350 }
351 ParticleHistos * histos2 = (ParticleHistos *) histograms[baseSingle+iParticleFilter2];
352 ParticlePairHistos * pairHistos = (ParticlePairHistos *) histograms[basePair+iParticleFilter1*nParticleFilters+iParticleFilter2];
353 pairHistos->completeFill();
354 ParticlePairDerivedHistos * derivedPairHistos = (ParticlePairDerivedHistos *) derivedHistograms[baseDerivedPair+iParticleFilter1*nParticleFilters+iParticleFilter2];
355 derivedPairHistos->calculatePairDerivedHistograms(*histos1,*histos2,*pairHistos,ac.binCorrPP);
356 }
357 }
358 if (!ac.createCombinedHistograms) continue;
359
360
361 // ======================================================================
362 // CI and CD combined correlators
363 // In the following f# stands for a particle filter index, e.g., f1, f2
364 // Two possible scenarios:
365 // (1) a la nu-dyn: two species without respect to charge:
366 // CI = (f1f2) + (f2f1) + (f1f1) + (f2f2)
367 // CD = (f1f2) + (f2f1) - (f1f1) - (f2f2)
368 // (2) for charge general balance functions
369 // f1: species 1, +ve
370 // f2: species 2, +ve
371 // f3: species 1, -ve
372 // f4: species 2, -ve
373 // CI = (f1f4) + (f3f2) + (f1f2) + (f3f4)
374 // CD = (f1f4) + (f3f2) - (f1f2) - (f3f4)
375 //
376 // Internally, there is no way to distinguish cases (1) and (2)
377 // given there can be an arbitrary number of filters on input, so we need
378 // to supply on input the relevant combinations of filters
379 // ======================================================================
380 unsigned int nCombinations = ac.combinations.size();
381
382 if (reportInfo("ParticlePairAnalyzer",getName(),"createDerivedHistograms()"))
383 {
384 cout << ">>>>>>>>>>> Computing nCombinations:" << nCombinations << endl;
385 }
386 for (unsigned int k=0; k<nCombinations; k++)
387 {
388 unsigned int iParticleFilter1;
389 unsigned int iParticleFilter2;
390 unsigned int iParticleFilter3;
391 unsigned int iParticleFilter4;
392
393 unsigned int index1 = 0;
394 unsigned int index2 = 0;
395 unsigned int index3 = 0;
396 unsigned int index4 = 0;
397 unsigned int indexCI = 0;
398 unsigned int indexCD = 0;
399 if (reportInfo("ParticlePairAnalyzer",getName(),"createDerivedHistograms()"))
400 {
401 cout << "Computing combination k:" << k << endl;
402 }
403 vector<unsigned int> combination = ac.combinations[k]; // f1, f2 or f1,f2, f3, f4
404 if (reportInfo("ParticlePairAnalyzer",getName(),"createDerivedHistograms()"))
405 {
406 cout << "combination.size():" << combination.size() << endl;
407 }
408 if (combination.size()==2)
409 {
410 iParticleFilter1 = combination[0];
411 iParticleFilter2 = combination[1];
412 cout << "iParticleFilter1:" << iParticleFilter1 << " iParticleFilter2:" << iParticleFilter2 << endl;
413 index1 = baseDerivedPair+iParticleFilter1*nParticleFilters+iParticleFilter2;
414 index2 = baseDerivedPair+iParticleFilter2*nParticleFilters+iParticleFilter1;
415 index3 = baseDerivedPair+iParticleFilter1*nParticleFilters+iParticleFilter1;
416 index4 = baseDerivedPair+iParticleFilter2*nParticleFilters+iParticleFilter2;
417
418 }
419 else if (combination.size()==4)
420 {
421 iParticleFilter1 = combination[0];
422 iParticleFilter2 = combination[1];
423 iParticleFilter3 = combination[2];
424 iParticleFilter4 = combination[3];
425 cout << "iParticleFilter1:" << iParticleFilter1 << " iParticleFilter2:" << iParticleFilter2 << endl;
426 cout << "iParticleFilter3:" << iParticleFilter3 << " iParticleFilter4:" << iParticleFilter4 << endl;
427 index1 = baseDerivedPair+iParticleFilter1*nParticleFilters+iParticleFilter4;
428 index2 = baseDerivedPair+iParticleFilter3*nParticleFilters+iParticleFilter2;
429 index3 = baseDerivedPair+iParticleFilter1*nParticleFilters+iParticleFilter2;
430 index4 = baseDerivedPair+iParticleFilter3*nParticleFilters+iParticleFilter4;
431 }
432 else
433 {
434 if (reportError("ParticlePairAnalyzer",getName(),"createDerivedHistograms()"))
435 {
436 cout << "Invalid number of combinatinos encounted. Abort." << endl;
437 }
438 exit(0);
439 }
440 indexCI = baseCombinedPair + 2*k;
441 indexCD = baseCombinedPair + 2*k + 1;
442 ParticlePairDerivedHistos * derivedHistos1 = (ParticlePairDerivedHistos *) derivedHistograms[index1];
443 ParticlePairDerivedHistos * derivedHistos2 = (ParticlePairDerivedHistos *) derivedHistograms[index2];
444 ParticlePairDerivedHistos * derivedHistos3 = (ParticlePairDerivedHistos *) derivedHistograms[index3];
445 ParticlePairDerivedHistos * derivedHistos4 = (ParticlePairDerivedHistos *) derivedHistograms[index4];
446 ParticlePairCombinedHistos * combinedHistosCI = (ParticlePairCombinedHistos *) combinedHistograms[indexCI];
447 ParticlePairCombinedHistos * combinedHistosCD = (ParticlePairCombinedHistos *) combinedHistograms[indexCD];
448 combinedHistosCI->calculate(derivedHistos1,derivedHistos2,derivedHistos3, derivedHistos4, 0.25, 0.25, 0.25, 0.25);
449 combinedHistosCD->calculate(derivedHistos1,derivedHistos2,derivedHistos3, derivedHistos4, 0.25, 0.25, -0.25,-0.25);
450 }
451 }
452 if (reportEnd("ParticlePairAnalyzer",getName(),"createDerivedHistograms()"))
453 ;
454}
