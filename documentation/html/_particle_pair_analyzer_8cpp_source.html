229void ParticlePairAnalyzer::execute()
230{
231//
232// if (reportStart(__FUNCTION__))
233// ;
234 incrementTaskExecuted();
235 Event & event = *eventStreams[0];
236
237 //Is this event accepted by this task's event filters?
238 bool analyzeThisEvent = false;
239 unsigned int nEventFilters = eventFilters.size();
240 vector<unsigned int> eventFilterPassed;
241 eventFilterPassed.clear();
242 resetNParticlesAcceptedEvent();
243 for (unsigned int iEventFilter=0; iEventFilter<nEventFilters; iEventFilter++ )
244 {
245 if (!eventFilters[iEventFilter]->accept(event)) continue;
246 incrementNEventsAccepted(iEventFilter);
247 eventFilterPassed.push_back(iEventFilter);
248 analyzeThisEvent = true;
249 }
250 //if (reportInfo("ParticlePairAnalyzer",getName(),"createHistograms()")) cout << " -- 2 --" << endl;
251
252 if (!analyzeThisEvent) return;
253
254 //if (reportInfo("ParticlePairAnalyzer",getName(),"createHistograms()")) cout << " -- 2b --" << endl;
255
256 // The event is accepted at least one event filter
257 // but it may not have particle pairs. If so skip out.
258 // Doing the checks in this order guarantees the accepted
259 // event count is correct for normalization purposes.
260 unsigned int nParticles = event.getNParticles();
261 if (nParticles<2) return;
262
263 //if (reportInfo("ParticlePairAnalyzer",getName(),"createHistograms()")) cout << " -- 3 --" << endl;
264
265 Factory<ParticleDigit> * factory = ParticleDigit::getFactory();
266 factory->reset();
267
268 //if (reportInfo("ParticlePairAnalyzer",getName(),"createHistograms()")) cout << " -- 4 --" << endl;
269
270 // produce sublists with ParticleDigits so we do not have to digitize too many
271 // times...
272 // The histo instance fetched here is used for digitization only. So
273 // we use instance [0];
274 ParticlePairHistos * histos = (ParticlePairHistos *) basePairHistograms[0];
275
276// cout << "Particle Pair Analyzer Test :" << endl;
277// cout << histos->getPtBinFor(0.5) << endl;
278// exit(1);
279
280
281 unsigned int nParticleFilters = particleFilters.size();
282 for (unsigned int iParticleFilter=0; iParticleFilter<nParticleFilters; iParticleFilter++ ) filteredParticles[iParticleFilter].clear();
283
284 //if (reportInfo("ParticlePairAnalyzer",getName(),"createHistograms()")) cout << " -- 5 --" << endl;
285
286 for (unsigned int iParticle=0; iParticle<nParticles; iParticle++)
287 {
288 Particle & particle = * event.getParticleAt(iParticle);
289 // TLorentzVector & what = particle.getMomentum();
290 // double rap = what.Rapidity();
291 // if (rap<-1.0)
292 // {
293 // cout << " rap: " << rap << endl;
294 // }
295
296 // int pdg = particle.getType().getPdgCode();
297 // if (pdg==3122|| pdg==-3122)
298 // {
299 // cout << "pdg:" << pdg << endl;
300 // cout << "Found L" << endl;
301 // exit(0);
302 // }
303 float pt, e;
304 int iPt, iPhi, iEta, iY;
305 ParticleDigit * pd;
306 bool digitized = false;
307 for (unsigned int iParticleFilter=0; iParticleFilter<nParticleFilters; iParticleFilter++ )
308 {
309 if (particleFilters[iParticleFilter]->accept(particle))
310 {
311
312 if (!digitized)
313 {
314 TLorentzVector & momentum = particle.getMomentum();
315 pt = momentum.Pt();
316 e = momentum.E();
317 iPt = histos->getPtBinFor(pt);
318 iPhi = histos->getPhiBinFor(momentum.Phi());
319 iEta = fillEta ? histos->getEtaBinFor(momentum.Eta()) : -1;
320 iY = fillY ? histos->getYBinFor(momentum.Rapidity()) : -1;
321
322 // cout << " pt: " << pt << " e:" << e << " iPt: " << iPt << " iPhi:" << iPhi << " iEta: " << iEta << " iY: " << iY << endl;
323
324 pd = factory->getNextObject();
325 pd->iY = iY;
326 pd->iEta = iEta;
327 pd->iPt = iPt;
328 pd->iPhi = iPhi;
329 pd->pt = pt;
330 pd->e = e;
331 digitized = true; // so no need to digitize this particle again...
332 }
333 if (digitized && iPt>=0 && iPhi>=0 && (iEta>=0 || iY>=0) )
334 {
335 filteredParticles[iParticleFilter].push_back(pd);
336 }
337 //if (reportInfo("ParticlePairAnalyzer",getName(),"createHistograms()")) cout << " -- 7 --" << endl;
338 } // particle accepted by filter
339 } //particle loop
340 } // particle filter loop
341 //if (reportInfo("ParticlePairAnalyzer",getName(),"createHistograms()")) cout << " -- 8 --" << endl;
342 // use the filtered particles to fill the histos for the accepted event filters
343 for (unsigned int jEventFilter=0; jEventFilter<eventFilterPassed.size(); jEventFilter++ )
344 {
345 //if (reportInfo("ParticlePairAnalyzer",getName(),"createHistograms()")) cout << " -- 9 --" << endl;
346 unsigned int iEventFilter = eventFilterPassed[jEventFilter];
347 unsigned int baseSingle = iEventFilter*nParticleFilters;
348 unsigned int basePair = iEventFilter*nParticleFilters*nParticleFilters;
349 unsigned int index;
350 for (unsigned int iParticleFilter1=0; iParticleFilter1<nParticleFilters; iParticleFilter1++ )
351 {
352 incrementNParticlesAccepted(jEventFilter,iParticleFilter1);
353 //if (reportInfo("ParticlePairAnalyzer",getName(),"createHistograms()")) cout << " -- 10 --" << endl;
354 index = baseSingle + iParticleFilter1;
355 ParticleHistos * histos = (ParticleHistos *) baseSingleHistograms[index];
356 histos->fill(filteredParticles[iParticleFilter1],1.0);
357 for (unsigned int iParticleFilter2=0; iParticleFilter2<nParticleFilters; iParticleFilter2++ )
358 {
359 //if (reportInfo("ParticlePairAnalyzer",getName(),"createHistograms()")) cout << " -- 11 --" << endl;
360 index = basePair + iParticleFilter1*nParticleFilters + iParticleFilter2;
361 ParticlePairHistos * histos = (ParticlePairHistos *) basePairHistograms[index];
362 histos->fill(filteredParticles[iParticleFilter1],filteredParticles[iParticleFilter2],iParticleFilter1==iParticleFilter2,1.0);
363 //if (reportInfo("ParticlePairAnalyzer",getName(),"createHistograms()")) cout << " -- 13 --" << endl;
364 }
365 }
366 }
367// if (reportEnd(__FUNCTION__))
368// ;
369}

372void ParticlePairAnalyzer::scaleHistograms()
373{
374
375 if (reportStart(__FUNCTION__))
376 ;
377 double scalingFactor;
378 if (reportInfo(__FUNCTION__)) printEventStatistics();
379 int index = 0;
380 for (int iEventFilter=0; iEventFilter<nEventFilters; iEventFilter++ )
381 {
382 long nAccepted = getNEventsAcceptedReset();
383 if (nAccepted>1)
384 {
385
386 scalingFactor = 1.0/double(nAccepted);
387 for (int iParticleFilter=0; iParticleFilter<nParticleFilters; iParticleFilter++ )
388 {
389 index = iEventFilter*nParticleFilters + iParticleFilter;
390 baseSingleHistograms[index]->scale(scalingFactor);
391 }
392 for (int iParticleFilter1=0; iParticleFilter1<nParticleFilters; iParticleFilter1++ )
393 {
394 for (int iParticleFilter2=0; iParticleFilter2<nParticleFilters; iParticleFilter2++ )
395 {
396 index = iEventFilter*nParticleFilters*nParticleFilters + iParticleFilter1*nParticleFilters + iParticleFilter2;
397 basePairHistograms[index]->scale(scalingFactor);
398 }
399 }
400 }
401 else
402 {
403 if (reportWarning(__FUNCTION__))
404 {
405 cout << endl;
406 cout << " iEventFilter: " << iEventFilter<< endl;
407 cout << " nEventsAccepted[iEventFilter]: " << nAccepted<< endl;
408 cout << " no scaling performed: " << endl;
409 }
410 }
411 }
412 if (reportEnd(__FUNCTION__))
413 ;
