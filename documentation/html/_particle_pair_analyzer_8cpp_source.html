249void ParticlePairAnalyzer::execute()
250{
251 //if (reportInfo("ParticlePairAnalyzer",getName(),"createHistograms()")) cout << " -- 1 --" << endl;
252
253 incrementEventProcessed();
254 Event & event = *eventStreams[0];
255
256 //Is this event accepted by this task's event filters?
257 bool analyzeThisEvent = false;
258 unsigned int nEventFilters = eventFilters.size();
259 vector<unsigned int> eventFilterPassed;
260 eventFilterPassed.clear();
261 for (unsigned int iEventFilter=0; iEventFilter<nEventFilters; iEventFilter++ )
262 {
263 if (!eventFilters[iEventFilter]->accept(event)) continue;
264 incrementEventAccepted(iEventFilter);
265 eventFilterPassed.push_back(iEventFilter);
266 analyzeThisEvent = true;
267 }
268 //if (reportInfo("ParticlePairAnalyzer",getName(),"createHistograms()")) cout << " -- 2 --" << endl;
269
270 if (!analyzeThisEvent) return;
271
272 //if (reportInfo("ParticlePairAnalyzer",getName(),"createHistograms()")) cout << " -- 2b --" << endl;
273
274 // The event is accepted at least one event filter
275 // but it may not have particle pairs. If so skip out.
276 // Doing the checks in this order guarantees the accepted
277 // event count is correct for normalization purposes.
278 unsigned int nParticles = event.getNParticles();
279 if (nParticles<2) return;
280
281 //if (reportInfo("ParticlePairAnalyzer",getName(),"createHistograms()")) cout << " -- 3 --" << endl;
282
283 Factory<ParticleDigit> * factory = ParticleDigit::getFactory();
284 factory->reset();
285
286 //if (reportInfo("ParticlePairAnalyzer",getName(),"createHistograms()")) cout << " -- 4 --" << endl;
287
288 // produce sublists with ParticleDigits so we do not have to digitize too many
289 // times...
290 // The histo instance fetched here is used for digitization only. So
291 // we use instance [0];
292 ParticlePairHistos * histos = (ParticlePairHistos *) basePairHistograms[0];
293
294// cout << "Particle Pair Analyzer Test :" << endl;
295// cout << histos->getPtBinFor(0.5) << endl;
296// exit(1);
297
298
299 unsigned int nParticleFilters = particleFilters.size();
300 for (unsigned int iParticleFilter=0; iParticleFilter<nParticleFilters; iParticleFilter++ ) filteredParticles[iParticleFilter].clear();
301
302 //if (reportInfo("ParticlePairAnalyzer",getName(),"createHistograms()")) cout << " -- 5 --" << endl;
303
304 for (unsigned int iParticle=0; iParticle<nParticles; iParticle++)
305 {
306 Particle & particle = * event.getParticleAt(iParticle);
307
308 // TLorentzVector & what = particle.getMomentum();
309 // double rap = what.Rapidity();
310 // if (rap<-1.0)
311 // {
312 // cout << " rap: " << rap << endl;
313 // }
314
315 // int pdg = particle.getType().getPdgCode();
316 // if (pdg==3122|| pdg==-3122)
317 // {
318 // cout << "pdg:" << pdg << endl;
319 // cout << "Found L" << endl;
320 // exit(0);
321 // }
322 float pt, e;
323 int iPt, iPhi, iEta, iY;
324 ParticleDigit * pd;
325 bool digitized = false;
326 for (unsigned int iParticleFilter=0; iParticleFilter<nParticleFilters; iParticleFilter++ )
327 {
328 if (particleFilters[iParticleFilter]->accept(particle))
329 {
330 //if (reportInfo("ParticlePairAnalyzer",getName(),"createHistograms()")) cout << " -- 6 --" << endl;
331
332 if (!digitized)
333 {
334 TLorentzVector & momentum = particle.getMomentum();
335 pt = momentum.Pt();
336 e = momentum.E();
337 iPt = histos->getPtBinFor(pt);
338 iPhi = histos->getPhiBinFor(momentum.Phi());
339 iEta = fillEta ? histos->getEtaBinFor(momentum.Eta()) : -1;
340 iY = fillY ? histos->getYBinFor(momentum.Rapidity()) : -1;
341
342 // cout << " pt: " << pt << " e:" << e << " iPt: " << iPt << " iPhi:" << iPhi << " iEta: " << iEta << " iY: " << iY << endl;
343
344 pd = factory->getNextObject();
345 pd->iY = iY;
346 pd->iEta = iEta;
347 pd->iPt = iPt;
348 pd->iPhi = iPhi;
349 pd->pt = pt;
350 pd->e = e;
351 digitized = true; // so no need to digitize this particle again...
352 }
353 if (digitized && iPt>=0 && iPhi>=0 && (iEta>=0 || iY>=0) )
354 {
355 filteredParticles[iParticleFilter].push_back(pd);
356 }
357 //if (reportInfo("ParticlePairAnalyzer",getName(),"createHistograms()")) cout << " -- 7 --" << endl;
358 } // particle accepted by filter
359 } //particle loop
360 } // particle filter loop
361 //if (reportInfo("ParticlePairAnalyzer",getName(),"createHistograms()")) cout << " -- 8 --" << endl;
362 // use the filtered particles to fill the histos for the accepted event filters
363 for (unsigned int jEventFilter=0; jEventFilter<eventFilterPassed.size(); jEventFilter++ )
364 {
365 //if (reportInfo("ParticlePairAnalyzer",getName(),"createHistograms()")) cout << " -- 9 --" << endl;
366 unsigned int iEventFilter = eventFilterPassed[jEventFilter];
367 unsigned int baseSingle = iEventFilter*nParticleFilters;
368 unsigned int basePair = iEventFilter*nParticleFilters*nParticleFilters;
369 unsigned int index;
370 for (unsigned int iParticleFilter1=0; iParticleFilter1<nParticleFilters; iParticleFilter1++ )
371 {
372 //if (reportInfo("ParticlePairAnalyzer",getName(),"createHistograms()")) cout << " -- 10 --" << endl;
373 index = baseSingle + iParticleFilter1;
374 ParticleHistos * histos = (ParticleHistos *) baseSingleHistograms[index];
375 histos->fill(filteredParticles[iParticleFilter1],1.0);
376 for (unsigned int iParticleFilter2=0; iParticleFilter2<nParticleFilters; iParticleFilter2++ )
377 {
378 //if (reportInfo("ParticlePairAnalyzer",getName(),"createHistograms()")) cout << " -- 11 --" << endl;
379 index = basePair + iParticleFilter1*nParticleFilters + iParticleFilter2;
380 ParticlePairHistos * histos = (ParticlePairHistos *) basePairHistograms[index];
381 histos->fill(filteredParticles[iParticleFilter1],filteredParticles[iParticleFilter2],iParticleFilter1==iParticleFilter2,1.0);
382 //if (reportInfo("ParticlePairAnalyzer",getName(),"createHistograms()")) cout << " -- 13 --" << endl;
383 }
384 }
385 }
386 //if (reportInfo("ParticlePairAnalyzer",getName(),"createHistograms()")) cout << " -- 14 --" << endl;
387}

390void ParticlePairAnalyzer::scaleHistograms()
391{
392 TString fct = "scaleHistograms()";
393 double scalingFactor;
394 unsigned int nEventFilters = eventFilters.size();
395 unsigned int nParticleFilters = particleFilters.size();
396 if (reportInfo(fct))
397 {
398 cout << endl;
399 cout << " Accepted number of events: " << nEventAccepted << endl;
400 cout << " nEventFilters: " << nEventFilters << endl;
401 cout << " nParticleFilters: " << nParticleFilters << endl;
402 cout << " nBaseSingleHistograms: " << getNBaseSingleHistograms() << endl;
403 cout << " nBasePairHistograms: " << getNBasePairHistograms() << endl;
404 cout << "-------------- Accumulated statistics: ----------------------------------" << endl;
405 }
406
407 unsigned int index = 0;
408 for (unsigned int iEventFilter=0; iEventFilter<nEventFilters; iEventFilter++ )
409 {
410 if (nFilteredEventsAccepted[iEventFilter]>1)
411 {
412 scalingFactor = 1.0/double(nFilteredEventsAccepted[iEventFilter]);
413 if (reportInfo(fct))
414 {
415 cout << endl;
416 cout << " iEventFilter: " << iEventFilter<< endl;
417 cout << " nFilteredEventsAccepted[iEventFilter]: " << nFilteredEventsAccepted[iEventFilter]<< endl;
418 cout << " scalingFactor: " << scalingFactor << endl;
419 }
420 for (unsigned int iParticleFilter=0; iParticleFilter<nParticleFilters; iParticleFilter++ )
421 {
422 index = iEventFilter*nParticleFilters + iParticleFilter;
423 baseSingleHistograms[index]->scale(scalingFactor);
424 }
425 for (unsigned int iParticleFilter1=0; iParticleFilter1<nParticleFilters; iParticleFilter1++ )
426 {
427 for (unsigned int iParticleFilter2=0; iParticleFilter2<nParticleFilters; iParticleFilter2++ )
428 {
429 index = iEventFilter*nParticleFilters*nParticleFilters + iParticleFilter1*nParticleFilters + iParticleFilter2;
430 basePairHistograms[index]->scale(scalingFactor);
431 }
432 }
433 }
434 else
435 {
436 if (reportWarning(fct))
437 {
438 cout << endl;
439 cout << " iEventFilter: " << iEventFilter<< endl;
440 cout << " nFilteredEventsAccepted[iEventFilter]: " << nFilteredEventsAccepted[iEventFilter]<< endl;
441 cout << " no scaling performed: " << endl;
442 }
443 }
444 }
