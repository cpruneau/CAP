47void ClosureTest::execute()
48{
49
50 if (reportStart(__FUNCTION__))
51 ;
52 incrementTaskExecuted();
53 bool forceHistogramsRewrite = configuration.getValueBool("forceHistogramsRewrite");
54 int selectedMethod = configuration.getValueInt("selectedMethod");
55 TString histoInputPath = configuration.getValueString("histoInputPath");
56 TString histoGeneratorFileName = configuration.getValueString("histoGeneratorFileName");
57 TString histoDetectorFileName = configuration.getValueString("histoDetectorFileName");
58 TString histoOutputPath = configuration.getValueString("histoOutputPath");
59 TString histoClosureFileName = configuration.getValueString("histoClosureFileName");
60
61 if (reportInfo(__FUNCTION__))
62 {
63 cout
64 << endl
65 << " Starting closure test calculation for :" << endl
66 << " histoInputPath: " << histoInputPath << endl
67 << " histoGeneratorFileName: " << histoGeneratorFileName << endl
68 << " histoDetectorFileName: " << histoDetectorFileName << endl
69 << " histoOutputPath: " << histoOutputPath << endl
70 << " histoClosureFileName: " << histoClosureFileName << endl;
71 switch (selectedMethod)
72 {
73 case 0: cout << " selectedMethod: Difference" << endl; break;
74 case 1: cout << " selectedMethod: Ratio" << endl; break;
75 }
76 }
77
78 TFile * generatorFile = openRootFile(histoInputPath, histoGeneratorFileName, "READ");
79 TFile * detectorFile = openRootFile(histoInputPath, histoDetectorFileName, "READ");
80 TFile * closureFile;
81 if (forceHistogramsRewrite)
82 closureFile = openRootFile(histoOutputPath, histoClosureFileName,"RECREATE");
83 else
84 closureFile = openRootFile(histoOutputPath, histoClosureFileName,"NEW");
85
86 if (!generatorFile || !detectorFile || !closureFile || !isTaskOk()) return;
87
88 HistogramCollection * generatorCollection = new HistogramCollection("GeneratorLevel",getReportLevel());
89 HistogramCollection * detectorCollection = new HistogramCollection("DetectorLevel", getReportLevel());
90 HistogramCollection * closureCollection = new HistogramCollection("Closure", getReportLevel());
91 generatorCollection->loadCollection(generatorFile);
92 detectorCollection->loadCollection(detectorFile);
93 generatorCollection->setOwnership(false);
94 detectorCollection->setOwnership(false);
95 closureCollection->setOwnership(false);
96 switch (selectedMethod)
97 {
98 case 0: closureCollection->differenceCollection(*detectorCollection,*generatorCollection,true); break;
99 case 1: closureCollection->ratioCollection(*detectorCollection,*generatorCollection,true); break;
100 }
101
102 //if (!isTaskOk()) return;
103 closureCollection->saveHistograms(closureFile);
104 generatorFile->Close();
105 detectorFile->Close();
106 closureFile->Close();
107 delete generatorCollection;
108 delete detectorCollection;
109 delete closureCollection;
110 if (reportInfo ()) cout << "Closure Test Completed." << endl;
111}
