62void ClosureTest::execute()
63{
64 if (reportInfo("ClosureTest",getName(),"execute()"))
65 {
66 cout
67 << " Starting closure test:" << endl
68 << " inputPath:" << inputPath << endl
69 << " outputPath:" << outputPath << endl
70 << " generatorLevelFileName:" << generatorLevelFileName << endl
71 << " detectorLevelFileName:" << detectorLevelFileName << endl;
72 }
73
74
75 TFile * generatorFile = openRootFile(inputPath, generatorLevelFileName, "READ");
76 TFile * detectorFile = openRootFile(inputPath, detectorLevelFileName, "READ");
77 TFile * closureFile;
78
79 if (!generatorFile)
80 {
81 if (reportError("ClosureTest",getName(),"execute()"))
82 {
83 cout << "Generator File Name: " << generatorLevelFileName << " NOT FOUND" << endl;
84 }
85 setTaskError();
86 return;
87 }
88 if (!detectorFile)
89 {
90 if (reportError("ClosureTest",getName(),"execute()"))
91 {
92 cout << "Detector File Name: " << detectorFile << " NOT FOUND" << endl;
93 }
94 setTaskError();
95 return;
96 }
97
98 HistogramCollection * generatorCollection = new HistogramCollection("GeneratorLevel",getReportLevel());
99 HistogramCollection * detectorCollection = new HistogramCollection("DetectorLevel", getReportLevel());
100 HistogramCollection * closureCollection = new HistogramCollection("Closure", getReportLevel());
101
102 generatorCollection->loadCollection(generatorFile);
103 detectorCollection ->loadCollection(detectorFile);
104
105 closureCollection->ratio(*detectorCollection, *generatorCollection);
106 TFile * outputFile = openRootFile(outputPath, closureFileName, "RECREATE");
107 if (!isTaskOk()) return;
108 closureCollection->saveHistograms(outputFile);
109 delete generatorCollection;
110 delete detectorCollection;
111 delete closureCollection;
112 if (reportInfo ("ClosureTest",getName(),"execute()")) cout << "Closure Test Completed." << endl;
113}
