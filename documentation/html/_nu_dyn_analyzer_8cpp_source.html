134void NuDynAnalyzer::execute()
135{
136 incrementTaskExecuted();
137 unsigned int nEventFilters = eventFilters.size();
138 unsigned int nParticleFilters = particleFilters.size();
139 Event * event = eventStreams[0];
140 resetNParticlesAcceptedEvent();
141 for (unsigned int iEventFilter=0; iEventFilter<nEventFilters; iEventFilter++ )
142 {
143 if (!eventFilters[iEventFilter]->accept(*event)) continue;
144 incrementNEventsAccepted(iEventFilter); // count eventStreams used to fill histograms and for scaling at the end...
145
146 vector<double> nAccepted(nParticleFilters,0.0);
147 for (unsigned long iParticle=0; iParticle<event->getNParticles(); iParticle++)
148 {
149 Particle & particle = * event->getParticleAt(iParticle);
150 for (unsigned int iParticleFilter=0; iParticleFilter<nParticleFilters; iParticleFilter++)
151 {
152 if (particleFilters[iParticleFilter]->accept(particle))
153 {
154 incrementNParticlesAccepted(iEventFilter,iParticleFilter);
155 nAccepted[iParticleFilter]++;
156 }
157 }
158 }
159 EventProperties ep = * event->getEventProperties();
160 NuDynHistos * nuDynHistos = (NuDynHistos *) histograms[iEventFilter];
161 switch ( multiplicityType )
162 {
163 case 0: nuDynHistos->fill(ep.fractionalXSection, nAccepted,1.0); break;
164 case 1: nuDynHistos->fill(ep.referenceMultiplicity, nAccepted,1.0); break;
165 case 2: nuDynHistos->fill(ep.referenceMultiplicity, nAccepted,1.0); break;
166 }
167 }
