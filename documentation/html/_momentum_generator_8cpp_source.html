9double probDensity(double p, double mass, double temperature, double mu, double stat)
10{
11 double e = sqrt(p*p + mass*mass);
12 double beta = 1.0/temperature;
13 double arg = beta*(e-mu);
14 double density = p*p/(exp(arg)+stat);
15 return density;
16}

19MomentumGenerator::MomentumGenerator(int _generatorType, ParticleType * _particleType,
20 vector<double> _parameters, TRandom * _selectedRandom)
21:
22generatorType(_generatorType),
23particleType(_particleType),
24mass(0),
25stat(0),
26parameters(_parameters),
27histograms(),
28random(_selectedRandom)
29{
30 mass = particleType->getMass();
31 stat = 0; //particleType->getStat();
32
33 bool invalidSetup = false;
34 switch (generatorType)
35 {
36 case 0:
37 // generate isotropic distribution
38 // momentum is Gaussian distributed
39 // parameters[0] : gaussian width
40 if (parameters.size()!=1) invalidSetup = true;
41 break;
42
43 case 1:
44 // generate isotropic distribution
45 // momentum is exponential distributed
46 // parameters[0] : average momentum
47 if (parameters.size()!=1) invalidSetup = true;
48 break;
49
50 case 2:
51 // generate isotropic distribution
52 // momentum is uniform distributed in range min -- max
53 // parameters[0] : minimum momentum
54 // parameters[1] : maximum momentum
55 if (parameters.size()!=2) invalidSetup = true;
56 break;
57
58 case 3:
59 // generate isotropic distribution w/ uniform density
60 // parameters[0] : minimum momentum
61 // parameters[1] : maximum momentum
62 if (parameters.size()!=2) invalidSetup = true;
63 break;
64
65 case 4:
66 // generate isotropic Maxwell Boltzmann distribution
67 // parameters[0] : temperature
68 if (parameters.size()!=1) invalidSetup = true;
69 break;
70
71 case 5:
72 // generate cylindrical distribution w/ pt gaussian distribution
73 // parameters[0] : minimum rapidity
74 // parameters[1] : width of rapidity window
75 // parameters[2] : width of Gauss pt
76 if (parameters.size()!=3) invalidSetup = true;
77 break;
78
79 case 6:
80 // generate cylindrical distribution w/ pt exponential distribution
81 // parameters[0] : minimum rapidity
82 // parameters[1] : width of rapidity window
83 // parameters[2] : exponential slope
84 if (parameters.size()!=3) invalidSetup = true;
85 break;
86
87 case 7:
88 // generate cylindrical distribution
89 // parameters[0] : minimum rapidity
90 // parameters[1] : width of rapidity window
91 // parameters[2] : temperature (GeV)
92 // parameters[3] : n bins in pT
93 // parameters[4] : miniumum pT
94 // parameters[5] : maxiumum pT
95 if (parameters.size()!=6) invalidSetup = true;
96 double temp = parameters[2];
97 double zero = 0.0;
98 TString hName = particleType->getName();
99 hName += int(1000*temp);
100 hName += "MeV";
101 double mu = 0.0;
102 int nBinsP = int(parameters[3]);
103 double minP = parameters[4];
104 double maxP = parameters[5];
105 double dp = (maxP-minP)/double(nBinsP);
106 TH1 * h = new TH1D(hName,hName, nBinsP, minP, maxP);
107 double p = -dp;
108 double v;
109 for (int j=0; j<nBinsP; j++)
110 {
111 p += dp;
112 v = probDensity(p,mass,temp,mu,stat);
113 h->SetBinContent(j,v);
114 h->SetBinError(j,zero);
115 }
116 histograms.push_back(h);
117 break;
118 }
119 if (invalidSetup)
120 {
121 cout << "<F> MomentumGenerator(...)" << endl;
122 cout << " generatorType: " << generatorType << endl;
123 cout << " particleType: " << particleType->getName() << endl;
124 cout << " parameters.size(): " << parameters.size() << endl;
125 cout << " ABORT!!!! " << endl;
126 exit(1);
127 }
128}

131void MomentumGenerator::generate(TLorentzVector & momentum)
132{
133 double px, py, pz, pt, mt, p, e, ptSq;
134 double phi, cosTh, y;
135 switch (generatorType)
136 {
137 case IsotropicGaussP:
138 // generate isotropic distribution
139 // p is Gaussian distributed
140 phi = TMath::TwoPi()*random->Rndm( );
141 cosTh = -1.0 + 2.0*random->Rndm( );
142 p = fabs(random->Gaus(0.0,parameters[0]));
143 e = sqrt(mass*mass + p*p);
144 pt = p*sqrt(1.0-cosTh*cosTh);
145 px = pt*cos(phi);
146 py = pt*sin(phi);
147 pz = p*cosTh;
148 momentum.SetPxPyPzE (px,py,pz,e);
149 break;
150
151 case IsotropicExpP:
152 // generate isotropic distribution
153 // p is exponential distributed
154 // exp( -t/tau )
155 phi = TMath::TwoPi()*random->Rndm( );
156 cosTh = -1.0 + 2.0*random->Rndm( );
157 p = random->Exp(parameters[0]);
158 e = sqrt(mass*mass + p*p);
159 pt = p*sqrt(1.0-cosTh*cosTh);
160 px = pt*cos(phi);
161 py = pt*sin(phi);
162 pz = p*cosTh;
163 momentum.SetPxPyPzE (px,py,pz,e);
164 break;
165
166 case IsotropicUniformP:
167 // generate isotropic distribution
168 // p is uniform distributed
169 phi = TMath::TwoPi()*random->Rndm( );
170 cosTh = -1.0 + 2.0*random->Rndm( );
171 p = parameters[0] + random->Rndm()*parameters[1];
172 e = sqrt(mass*mass + p*p);
173 pt = p*sqrt(1.0-cosTh*cosTh);
174 px = pt*cos(phi);
175 py = pt*sin(phi);
176 pz = p*cosTh;
177 momentum.SetPxPyPzE (px,py,pz,e);
178 break;
179
180 case IsotropicUniformDensity:
181 // generate isotropic distribution
182 // uniform density
183 phi = TMath::TwoPi()*random->Rndm( );
184 cosTh = -1.0 + 2.0*random->Rndm( );
185 p = parameters[0] + random->Rndm()*parameters[1];
186 e = sqrt(mass*mass + p*p);
187 pt = p*sqrt(1.0-cosTh*cosTh);
188 px = pt*cos(phi);
189 py = pt*sin(phi);
190 pz = p*cosTh;
191 momentum.SetPxPyPzE (px,py,pz,e);
192 break;
193
194 case IsotropicMaxwellP:
195 // generate isotropic distribution
196 // Maxwell Boltzmann
197 px = random->Gaus(0.0,parameters[0] );
198 py = random->Gaus(0.0,parameters[0] );
199 pz = random->Gaus(0.0,parameters[0] );
200 e = sqrt(mass*mass + px*px + py*py + pz*pz);
201 momentum.SetPxPyPzE (px,py,pz,e);
202 break;
203
204// case IsotropicHistoP:
205// // generate isotropic distribution
206// // p determined by histogram
207// phi = TMath::TwoPi()*random->Rndm( );
208// cosTh = -1.0 + 2.0*random->Rndm( );
209// p = histogramP->GetRandom();

213// py = pt*sin(phi);
214// pz = p*cosTh;
215// momentum.SetPxPyPzE (px,py,pz,e);
216// break;
217
218 case CylindricalGaussPtUniformY:
219 // generate cylindrical distribution
220 // parameters[0] : minimum rapidity
221 // parameters[1] : width of rapidity window
222 // parameters[2] : width of Gauss pt
223 phi = TMath::TwoPi()*random->Rndm( );
224 px = random->Gaus(0.0,parameters[2] );
225 py = random->Gaus(0.0,parameters[2] );
226 ptSq = px*px + py*py;
227 pt = sqrt(ptSq);
228 mt = sqrt(mass*mass + ptSq);
229 y = parameters[1]+parameters[2]*random->Rndm( );
230 e = mt*cosh(y);
231 pz = mt*sinh(y);
232 px = pt*cos(phi);
233 py = pt*sin(phi);
234 momentum.SetPxPyPzE (px,py,pz,e);
235 break;
236
237 case CylindricalExpPtUniformY:
238 // generate cylindrical distribution
239 // parameters[0] : minimum rapidity
240 // parameters[1] : width of rapidity window
241 // parameters[2] : exponential slope
242 phi = TMath::TwoPi()*random->Rndm( );
243 pt = random->Exp(parameters[2]);
244 mt = sqrt(mass*mass + pt*pt);
245 y = parameters[0]+parameters[1]*random->Rndm( );
246 e = mt*cosh(y);
247 pz = mt*sinh(y);
248 px = pt*cos(phi);
249 py = pt*sin(phi);
250 momentum.SetPxPyPzE (px,py,pz,e);
251 break;
252
253 case CylindricalMaxwellPtUniformY:
254 // generate cylindrical distribution
255 // parameters[0] : minimum rapidity
256 // parameters[1] : width of rapidity window
257 // parameters[2] : Gauss Width
258 pt = histograms[0]->GetRandom();
259 phi = TMath::TwoPi()*random->Rndm( );
260 px = pt*cos(phi);
261 py = pt*sin(phi);
262 mt = sqrt(mass*mass+pt*pt);
263 y = parameters[0]+parameters[1]*random->Rndm( );
264 e = mt*cosh(y);
265 pz = mt*sinh(y);
266 momentum.SetPxPyPzE (px,py,pz,e);
267 break;
268 }
