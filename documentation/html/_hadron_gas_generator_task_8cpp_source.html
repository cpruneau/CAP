126 if (reportDebug("HadronGasGeneratorTask",getName(),"initialize()"))

128 gas->printProperties(std::cout);

131
132 if (reportDebug("HadronGasGeneratorTask",getName(),"initialize()"))
133 {
134 cout << "Setting up momentum generation parameters." << endl;
135 }
136 int generatorType = config.getValueInt("generatorType");
137 vector<double> parameters;
138 parameters.clear();
139 switch (generatorType)
140 {
141 case 0:
142 case 1:
143 case 2:
144 parameters.push_back(temperature);
145 break;
146
147 case 3:
148 case 4:
149 parameters.push_back(0.001);
150 parameters.push_back(10.00);
151 break;
152
153 case 5:
154 case 6:
155 parameters.push_back(minY);
156 parameters.push_back(rangeY);
157 parameters.push_back(temperature);
158 break;
159

161 parameters.push_back(minY);
162 parameters.push_back(rangeY);

176 cout << " parameter index: " << k << " value: " << parameters[k] << endl;
177 }
178 }
179
180 unsigned int nSpecies = particleTypes->size();

182 for (unsigned int k=0; k<nSpecies; k++)

184 ParticleType * particleType = particleTypes->getParticleType(k);
185
186 MomentumGenerator * gen = new MomentumGenerator(generatorType, particleType, parameters);
187 momentumGenerators.push_back(gen);

193 cout << " Filling abundance histograms." << endl;
194 cout << " hadronGases.size:" << hadronGases.size() << endl;

198 int nSpeciesConsidered = d.size();
199 TString histoName;
200 histoName = getName();

207 for (unsigned int k=1; k<=d.size();k++)
208 {
209 relativeAbundancesGas->SetBinContent(k, d[k-1]);
210 relativeAbundancesGas->SetBinError(k, 0.0);
211 }
212 if (reportEnd(__FUNCTION__))

216void HadronGasGeneratorTask::execute()
217{
218
219 if (reportStart(__FUNCTION__))
220 ;
221 incrementTaskExecuted();
222 Event & event = *eventStreams[0];
223 Particle * interaction;
225
226 if (standaloneMode)
227 {
228 // In this mode, we generate one HG event per call. One interaction vertex is
229 // inserted in the event stream and generate is called to carry out the particle generation.
230 particleFactory->reset();
231 event.reset();
232 interaction = particleFactory->getNextObject();
233 interaction->reset();
234 interaction->setType( ParticleType::getInteractionType());
235 interaction->setXYZT(0.0, 0.0, 0.0, 0.0);
236 //interaction->setRThetaPhiT(3.0, TMath::PiOver2(), TMath::TwoPi()*taskRandomGenerator->Rndm(),0.0);
237 event.add(interaction);
238 event.setNucleusA(1.0,1.0);
239 event.setNucleusB(1.0,1.0);
240 generate(interaction);
241 EventProperties & ep = * eventStreams[0]->getEventProperties();
242 ep.zProjectile = 1; // atomic number projectile
243 ep.aProjectile = 1; // mass number projectile
244 ep.nPartProjectile = 1; // number of participants projectile
245 ep.zTarget = 1; // atomic number target
246 ep.aTarget = 1; // mass number target
247 ep.nPartTarget = 1; // number of participants target
248 ep.nParticipantsTotal= 2; // total number of participants
249 ep.nBinaryTotal = 1; // total number of binary collisions
250 ep.impactParameter = -99999; // nucleus-nucleus center distance in fm
251 ep.fractionalXSection= -99999; // fraction cross section value
252// ep.referenceMultiplicity = eventStreams[0]->getNParticles();// nominal multiplicity in the reference range
253// ep.particlesCounted = getNParticlesCounted();
254// ep.particlesAccepted = getNParticlesAccepted();
255 }
256 else
257 {
258 // In this mode, we generate several HG cells per event. Interaction vertices and their locations
259 // are assumed to be already loaded in the event stream and we produce as many HG cells as there
260 // are nucleon-nucleon interactions in the stream. Objects nucleusA and nucleusB are assumed defined by
261 // an earlier task and the structure EventProperties is assumed filled by that earlier task. We only fill the number of
262 // particle generated and accepted (in addition to storing the generated particles in the event)
263
264 // May skip this event if it does not satisfy the event cut.
265 if (!eventFilters[0]->accept(event)) return;
266 if (event.getParticleCount() < 1)
267 {
268 return;
269 }
270 vector<Particle*> interactions = event.getNucleonNucleonInteractions();
271
272 unsigned int n = interactions.size();
273// if (reportWarning("PythiaEventGenerator",getName(),"execute()"))
274// cout << "Size of interactions:" << n << endl;
275 for (unsigned int kInter=0; kInter<n; kInter++)
276 {
277 generate(interactions[kInter]);
278 }

286void HadronGasGeneratorTask::generate(Particle * parent)
287{
288
289 if (reportStart(__FUNCTION__))
290 ;
291 Event & event = * eventStreams[0];
292 int multiplicity = minTotalMult + double(rangeTotalMult)*gRandom->Rndm();
293 //ParticleFilter & particleFilter = * particleFilters[0];
294 Particle * particle;
295 TLorentzVector parentPosition = parent->getPosition();
296 for (int iPart=0; iPart< multiplicity; iPart++)
297 {
298 // implement basic no correlation scenario.
299 //ParticleType * type = hadronGases[0]->generateRandomHadron();
300 int index = hadronGases[0]->generateRandomHadronByIndex();
301 double rindex = index + 0.01;
302 relativeAbundances->Fill(rindex);
303 if (index<0)
304 {
305 if (reportFatal())
306 {
307 cout << "hadron index=" << index << endl;
308 }
309 exit(1);
310 }
311 ParticleType * type = particleTypes->getParticleType(index);
312 if (type==nullptr)
313 {
314 if (reportError(__FUNCTION__)) cout << "HadronGas::generateRandomHadron() return null pointer. Post task error." << endl;
315 postTaskError();
316 exit(1);
317 return;
318 }
319 TLorentzVector momentum;
320 momentumGenerators[index]->generate(momentum);
321 particle = particleFactory->getNextObject();
322 particle->setParent(parent);
323 particle->set(type, momentum, parentPosition,true);
324 //particle->printProperties(cout);
325 //incrementParticlesCounted(); // photons are NOT included in this tally
326 //if (!particleFilter.accept(*particle)) continue; // no filter for now...
327 event.add(particle);
