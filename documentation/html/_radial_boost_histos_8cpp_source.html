31void RadialBoostHistos::createHistograms()
32{
33 Configuration & configuration = getConfiguration();
34 int nBins_phi = configuration.getValueInt("nBins_phi");
35 double min_phi = configuration.getValueDouble("min_phi");
36 double max_phi = configuration.getValueDouble("max_phi");
37 int nBins_r = configuration.getValueInt("nBins_r");
38 double min_r = configuration.getValueDouble("min_r");
39 double max_r = configuration.getValueDouble("max_r");
40 int nBins_beta = configuration.getValueInt("nBins_beta");
41 double min_beta = configuration.getValueDouble("min_beta");
42 double max_beta = configuration.getValueDouble("max_beta");
43
44 TString bn = getHistoBaseName();
45 h_phi = createHistogram(bn+TString("ns_phi"), nBins_phi, min_phi, max_phi, "#phi", "N_{s}");
46 h_r = createHistogram(bn+TString("ns_r"), nBins_r, min_r, max_r, "r", "N_{s}");
47 h_beta = createHistogram(bn+TString("ns_beta"), nBins_beta, min_beta, max_beta, "#beta_{s}", "N_{s}");
48 h_xy = createHistogram(bn+TString("ns_xy"), 40, -10.0, 10.0, 40, -10.0, 10.0, "x", "y", "N_{s}");
49}

52void RadialBoostHistos::loadHistograms(TFile * inputFile)
53{
54 if (!inputFile)
55 {
56 if (reportFatal()) cout << "-Fatal- Attempting to load RadialBoostHistos from an invalid file pointer" << endl;
57 return;
58 }
59 //RadialBoostConfiguration & ac = *(RadialBoostConfiguration*) getConfiguration();
60 TString bn = getHistoBaseName();
61 h_phi = loadH1(inputFile,bn+TString("ns_phi") );
62 h_r = loadH1(inputFile,bn+TString("ns_r") );
63 h_beta = loadH1(inputFile,bn+TString("ns_beta") );
64 h_xy = loadH2(inputFile,bn+TString("ns_xy") );
65 return;
66}
