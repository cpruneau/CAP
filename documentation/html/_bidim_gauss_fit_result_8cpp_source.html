352std::tuple<double,double> BidimGaussFitResult::extractGeneralizedRMSError(TH1 *h)
353{
354 int NLOOPS_RMS = 10000;
355 double norm = double(NLOOPS_RMS);
356 int npoints = h->GetNbinsX();
357 cout << endl
358 << " Low edge :" << h->GetXaxis()->GetBinLowEdge(1) << endl
359 << " High edge :" << h->GetXaxis()->GetBinUpEdge(npoints) << endl;
360 double *point = new double[npoints];
361 double *point2 = new double[npoints];
362 double *val = new double[npoints];
363 double *vale = new double[npoints];
364 for (int ip = 0; ip < npoints; ++ip)
365 {
366 point[ip] = h->GetXaxis()->GetBinCenter(ip+1);
367 point2[ip] = point[ip]*point[ip];
368 val[ip] = h->GetBinContent(ip+1);
369 vale[ip] = h->GetBinError(ip+1);
370 }
371 double rms = 0.0;
372 double rms2 = 0.0;
373 for (int i = 0; i < NLOOPS_RMS; ++i)
374 {
375 double sumw = 0.0;
376 double sumwx = 0.0;
377 double sumwx2 = 0.0;
378 for (int ip = 0; ip < npoints; ++ip)
379 {
380 double w = gRandom->Gaus(val[ip],vale[ip]);
381 sumw += w;
382 sumwx += w*point[ip];
383 sumwx2 += w*point2[ip];
384 }
385 double _rms = (sumw != 0) ? TMath::Max(sumwx2/sumw - sumwx/sumw*sumwx/sumw,0.0) : 0.0;
386 rms += TMath::Sqrt(_rms);
387 rms2 += _rms;
388 }
389 delete [] vale;
390 delete [] val;
391 delete [] point2;
392 delete [] point;
393 rms /= norm;
394 rms2 /= norm;
395 rms2 = (rms2>0) ? TMath::Sqrt(rms2-rms*rms) : 0.0;
396 return std::make_tuple(rms,rms2);
397}

399std::tuple<double,double,double,double> BidimGaussFitResult::extractGeneralizedRMSandRMSError(TH2 *h,int xlowlimbin,int ylowlimbin)
400{
401 /* from the histogram we extract the rms on both dimensions */
402 /* longitudinally we restrict ourselves to the passed range */
403 /* azimuthally we restrict ourselves to the near side */
404 TH2* forRMS = (TH2*) h->Clone(TString::Format("%s_forRMS", h->GetName()));
405 int firstybin = ylowlimbin;
406 int lastybin = h->GetYaxis()->FindBin(-h->GetYaxis()->GetBinCenter(firstybin));
407 int firstxbin = xlowlimbin;
408 int lastxbin = h->GetXaxis()->FindBin(-h->GetXaxis()->GetBinCenter(firstxbin));
409 forRMS->GetXaxis()->SetRange(firstxbin,lastxbin);
410 forRMS->GetYaxis()->SetRange(firstybin,lastybin);
411 auto forRMSx = forRMS->ProjectionX(TString::Format("%s_px",forRMS->GetName()));
412 auto forRMSy = forRMS->ProjectionY(TString::Format("%s_py",forRMS->GetName()));
413 double rmsx = forRMSx->GetRMS();
414 double rmsmc_x,rmsemc_x;
415 std::tie(rmsmc_x,rmsemc_x) = extractGeneralizedRMSError(forRMSx);
416 double rmsy = forRMSy->GetRMS();
417 double rmsmc_y,rmsemc_y;
418 std::tie(rmsmc_y,rmsemc_y) = extractGeneralizedRMSError(forRMSy);
419 delete forRMSy;
420 delete forRMSx;
421 delete forRMS;
422
423 cout << endl
424 << " X: RMS: " << rmsx << " MC RMS: " << rmsmc_x << " DeltaRMS: " << rmsemc_x << endl
425 << " Y: RMS: " << rmsy << " MC RMS: " << rmsmc_y << " DeltaRMS: " << rmsemc_y << endl;
426 return std::make_tuple(rmsx,rmsemc_x,rmsy,rmsemc_y);
427}

429double BidimGaussFitResult::extractGeneralizedRMS(TF1* f, int sigmaIndex, int betaIndex)
430{
431 double sigma = f->GetParameter(sigmaIndex);
432 double beta = f->GetParameter(betaIndex);
433 return TMath::Sqrt(sigma*sigma*TMath::Gamma(3.0/beta)/TMath::Gamma(1.0/beta));
434}

436double BidimGaussFitResult::extractGeneralizedRMSError(TF1* f, TMatrixDSym& cov, int sigmaIndex, int betaIndex)
437{
438 double sigma = f->GetParameter(sigmaIndex);
439 double beta = f->GetParameter(betaIndex);
440 double sigmaDer = TMath::Sqrt(TMath::Gamma(3./beta)/TMath::Gamma(1./beta));
441 TF1* tmp = new TF1("tmp","TMath::Sqrt(TMath::Gamma(3./x)/TMath::Gamma(1./x))",1,2);
442 double betaDer = sigma*tmp->Derivative(beta);
443 double rmsError =
444 TMath::Power(sigmaDer * f->GetParError(sigmaIndex), 2) +
445 TMath::Power(betaDer * f->GetParError(betaIndex), 2) +
446 2.0 * sigmaDer * betaDer * cov(sigmaIndex, betaIndex);
447 delete tmp;
448 return TMath::Sqrt(rmsError);
449}
