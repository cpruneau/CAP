21// ============================================================================================================
22// 16 parameters function
23// ============================================================================================================
24// 0 : a : peak amplitude
25// 1 : omegaEta : longitudinal peak width (eta)
26// 2 : omegaPhi : azimuthal peak witdh (phi)
27// 3 : gammaEta : generalized gaussian exponent -- eta
28// 4 : gammaPhi : generalized gaussian exponent -- eta
29// 5 : a0 : constant offset
30// 6 : a1 : first harmonic coefficient
31// 7 : a2 : second harmonic coefficient
32// 8 : a3 : third harmonic coefficient
33// 9 : a4 : fourth harmonic coefficient
34// 10 : a5 : fift harmonic coefficient
35// 11 : a6 : sixth harmonic coefficient
36// 12 : a2LinEta : secong harmonic coefficient * fabs(eta)
37// 13 : a3LinEta : third harmonic coefficient * fabs(eta)
38// 14 : a2SqEta : second harmonic coefficient * square(fabs(eta))
39// 15 : a3SqEta : third harmonic coefficient * square(fabs(eta))
40// ============================================================================================================
41double FlowPlusGen2DGaussianFunction(double *x, double *par)
42{
43 double eta = x[0];
44 double etaSq = eta*eta;
45 double phi = x[1];
46 // harmonic components
47 double cos1Phi = TMath::Cos(phi);
48 double cos2Phi = TMath::Cos(2.0*phi);
49 double cos3Phi = TMath::Cos(3.0*phi);
50 double cos4Phi = TMath::Cos(4.0*phi);
51 double cos5Phi = TMath::Cos(5.0*phi);
52 double cos6Phi = TMath::Cos(6.0*phi);
53 // generalized gaussian peak
54 double norm1 = par[3]/2.0/par[1]/TMath::Gamma(1.0/par[3]);
55 double norm2 = par[4]/2.0/par[2]/TMath::Gamma(1.0/par[4]);
56 double arg1 = TMath::Abs(eta/par[1]);
57 double arg2a = TMath::Abs(phi/par[2]);
58 double arg2b = TMath::Abs((phi-TMath::TwoPi())/par[2]);
59 double peakX = norm1 * TMath::Exp(-(TMath::Power(arg1,par[3]) ));
60 double peakY = norm2 * ( TMath::Exp(-(TMath::Power(arg2a,par[4]))) + TMath::Exp(-(TMath::Power(arg2b,par[4]))));
61 double result = par[0]*peakX*peakY;
62 result += par[5];
63 result += par[6]*cos1Phi;
64 result += par[7]*cos2Phi;
65 result += par[8]*cos3Phi;
66 result += par[9]*cos4Phi;
67 result += par[10]*cos5Phi;
68 result += par[11]*cos6Phi;
69 /* the eta dependency of vn */
70 result += par[12]*eta*cos2Phi;
71 result += par[13]*eta*cos3Phi;
72 result += par[14]*etaSq*cos2Phi;
73 result += par[15]*etaSq*cos3Phi;
74 return result;
75}

77// ============================================================================================================
78// 11 parameters function
79// ============================================================================================================
80// 0 : a0 : constant offset
81// 1 : a1 : first harmonic coefficient
82// 2 : a2 : second harmonic coefficient
83// 3 : a3 : third harmonic coefficient
84// 4 : a4 : fourth harmonic coefficient
85// 5 : a5 : fift harmonic coefficient
86// 6 : a6 : sixth harmonic coefficient
87// 7 : a2LinEta : secong harmonic coefficient * fabs(eta)
88// 8 : a3LinEta : third harmonic coefficient * fabs(eta)
89// 9 : a2SqEta : second harmonic coefficient * square(fabs(eta))
90// 10 : a3SqEta : third harmonic coefficient * square(fabs(eta))
91// ============================================================================================================
92double Flow2DFunction(double *x, double *par)
93{
94 double eta = TMath::Abs(x[0]);
95 double etaSq = eta*eta;
96 double phi = x[1];
97 double cos1Phi = TMath::Cos(phi);
98 double cos2Phi = TMath::Cos(2.0*phi);
99 double cos3Phi = TMath::Cos(3.0*phi);
100 double cos4Phi = TMath::Cos(4.0*phi);
101 double cos5Phi = TMath::Cos(5.0*phi);
102 double cos6Phi = TMath::Cos(6.0*phi);
103 double result;
104 /* the flow part */
105 result = par[0];
106 result += par[1]*cos1Phi;
107 result += par[2]*cos2Phi;
108 result += par[3]*cos3Phi;
109 result += par[4]*cos4Phi;
110 result += par[5]*cos5Phi;
111 result += par[6]*cos6Phi;
112 /* the eta dependency of vn */
113 result += par[7]*eta*cos2Phi;
114 result += par[8]*eta*cos3Phi;
115 result += par[9]*etaSq*cos2Phi;
116 result += par[10]*etaSq*cos3Phi;
117 return result;
118}

120// ============================================================================================================
121// 5 parameters function
122// ============================================================================================================
123// 0 : a : peak amplitude
124// 1 : omegaEta : longitudinal peak width (eta)
125// 2 : omegaPhi : azimuthal peak witdh (phi)
126// 3 : gammaEta : generalized gaussian exponent -- eta
127// 4 : gammaPhi : generalized gaussian exponent -- etadouble Gen2DGaussianFunction(double *x, double *par)
128// 5 : a0 : constant offset
129double Gen2DGaussianFunction(double *x, double *par)
130{
131 double eta = x[0];
132 double phi = x[1];
133 double norm1 = par[3]/2.0/par[1]/TMath::Gamma(1.0/par[3]);
134 double norm2 = par[4]/2.0/par[2]/TMath::Gamma(1.0/par[4]);
135 double arg1 = TMath::Abs(eta/par[1]);
136 double arg2a = TMath::Abs(phi/par[2]);
137 double arg2b = TMath::Abs((phi-TMath::TwoPi())/par[2]);
138 double peakX = norm1 * TMath::Exp(-(TMath::Power(arg1,par[3]) ));
139 double peakY = norm2 * ( TMath::Exp(-(TMath::Power(arg2a,par[4]))) + TMath::Exp(-(TMath::Power(arg2b,par[4]))));
140 double result = par[0]*peakX*peakY;
141 result += par[5]; // allow for a residual offset
142 return result;
143}

513// Carry out a 2D fit of the given data at large delta-eta where flow (if present) should dominate
514// =========================================================================================================
515void BidimGaussFitter::flowFit(TH2 *dataHist, BidimGaussFitConfiguration & fitConfig)
516{
517 flowFitResult.reset();
518 initializeFlowFitParameters(flowFitFct,fitConfig,1.0);
519 for (int iLoop = 0; iLoop < fitConfig.nFitLoops; iLoop++)
520 {
521 flowFitResult.resultPtr = dataHist->Fit(flowFitFct, fitConfig.fitOption, "");
522 if (!flowFitResult.resultPtr)
523 {
524 if (reportWarning("BidimGaussFitter","2D Gauss Fit","flowFit(const BidimGaussConfig &fitConfig,const TH2 *data)") )
525 {
526 cout << endl << "Flow fit was UNSUCCESSFUL after iLoop==" << iLoop << endl;
527 }
528 return;
529 }
530 if (flowFitResult.resultPtr->IsValid()) break;
531 if (iLoop==fitConfig.nFitLoops)
532 {
533 if (reportWarning("BidimGaussFitter","2D Gauss Fit","flowFit(const BidimGaussConfig &fitConfig,const TH2 *data)") )
534 {
535 cout << endl << "Flow fit was UNSUCCESSFUL after iLoop==" << iLoop << endl;
536 }
537 return;
538 }
539 }
540 flowFitResult.success = true;
541 flowFitResult.saveResults(flowFitFct);
542 if (reportInfo("BidimGaussFitter","2D Gauss Fit","flowFit(...)"))
543 {
544 flowFitResult.printResults(cout);
545 }
546}

1345 {
