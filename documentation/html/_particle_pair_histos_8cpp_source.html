228void ParticlePairHistos::fill(vector<ParticleDigit*> & particle1, vector<ParticleDigit*> & particle2, bool same, double weight)
229{
230 double nPairs = 0;
231 int iG;
232
233 for (unsigned int iPart_1=0; iPart_1<particle1.size(); iPart_1++)
234 {
235 float pt1 = particle1[iPart_1]->pt;
236 float dpt1 = pt1; // - avgPt1
237 unsigned int iPt_1 = particle1[iPart_1]->iPt;
238 unsigned int iPhi_1 = particle1[iPart_1]->iPhi;
239 unsigned int iEta_1 = particle1[iPart_1]->iEta;
240 unsigned int iY_1 = particle1[iPart_1]->iY;
241
242 for (unsigned int iPart_2=(same?iPart_1+1: 0); iPart_2<particle2.size(); iPart_2++)
243 {
244 float pt2 = particle2[iPart_2]->pt;
245 float dpt2 = pt2; // - avgPt2
246 unsigned int iPt_2 = particle2[iPart_2]->iPt;
247 unsigned int iPhi_2 = particle2[iPart_2]->iPhi;
248 unsigned int iEta_2 = particle2[iPart_2]->iEta;
249 unsigned int iY_2 = particle2[iPart_2]->iY;
250
251
252 nPairs += 1.0;
253 iG = h_n2_ptpt->GetBin(iPt_1,iPt_2);
254 h_n2_ptpt->AddBinContent(iG, weight);
255
256 iG = h_n2_phiPhi->GetBin(iPhi_1,iPhi_2);
257 h_n2_phiPhi->AddBinContent(iG, weight);
258 if (fillP2)
259 {
260 h_DptDpt_phiPhi ->AddBinContent(iG,weight*dpt1*dpt2);
261 }
262 if (same)
263 {
264 nPairs++;
265 iG = h_n2_ptpt->GetBin(iPt_2,iPt_1);
266 h_n2_ptpt->AddBinContent(iG, weight);
267 iG = h_n2_phiPhi->GetBin(iPhi_2,iPhi_1);
268 h_n2_phiPhi->AddBinContent(iG, weight);
269 if (fillP2)
270 {
271 h_DptDpt_phiPhi->AddBinContent(iG,weight*pt1*pt2);
272 }
273 }
274
275 if (fillEta)
276 {
277 // delta-eta maps onto a 2n-1 range i.e., 0 to 2n-2
278 int iDeltaEta = iEta_1-iEta_2 + nBins_eta-1;
279 int iDeltaPhi = iPhi_1-iPhi_2;
280 if (iDeltaPhi < 0) iDeltaPhi += nBins_phi;
281
282 iG = h_n2_etaEta->GetBin(iEta_1,iEta_2);
283 h_n2_etaEta->AddBinContent(iG,weight);
284 if (fillP2)
285 {
286 h_DptDpt_etaEta->AddBinContent(iG,weight*pt1*pt2);
287 }
288
289 iG = h_n2_DetaDphi->GetBin(iDeltaEta+1,iDeltaPhi+1);
290 h_n2_DetaDphi->AddBinContent(iG,weight);
291 if (fillP2)
292 {
293 h_DptDpt_DetaDphi->AddBinContent(iG,weight*pt1*pt2);
294 }
295 if (same)
296 {
297 // delta-eta maps onto a 2n-1 range i.e., 0 to 2n-2
298 iDeltaEta = iEta_2-iEta_1 + nBins_eta-1;
299 iDeltaPhi = iPhi_2-iPhi_1;
300 if (iDeltaPhi < 0) iDeltaPhi += nBins_phi;
301 iG = h_n2_etaEta->GetBin(iEta_2,iEta_1);
302 h_n2_etaEta->AddBinContent(iG,weight);
303 if (fillP2)
304 {
305 h_DptDpt_etaEta->AddBinContent(iG,weight*pt1*pt2);
306 }
307
308 iG = h_n2_DetaDphi->GetBin(iDeltaEta+1,iDeltaPhi+1);
309 h_n2_DetaDphi->AddBinContent(iG,weight);
310 if (fillP2)
311 {
312 h_DptDpt_DetaDphi->AddBinContent(iG,weight*pt1*pt2);
313 }
314 }
315 }
316
317 if (fillY)
318 {
319 int iDeltaY = iY_1-iY_2 + nBins_y-1;
320 int iDeltaPhi = iPhi_1-iPhi_2;
321 if (iDeltaPhi < 0) iDeltaPhi += nBins_phi;
322
323 iG = h_n2_yY->GetBin(iY_1,iY_2);
324 h_n2_yY->AddBinContent(iG,weight);
325 if (fillP2)
326 {
327 h_DptDpt_yY->AddBinContent(iG,weight*pt1*pt2);
328 }
329
330 iG = h_n2_DyDphi->GetBin(iDeltaY+1,iDeltaPhi+1);
331 h_n2_DyDphi ->AddBinContent(iG,weight);
332 if (fillP2)
333 {
334 h_DptDpt_DyDphi->AddBinContent(iG,weight*pt1*pt2);
335 }
336 if (same)
337 {
338 int iDeltaY = iY_2-iY_1 + nBins_y-1;
339 int iDeltaPhi = iPhi_2-iPhi_1;
340 if (iDeltaPhi < 0) iDeltaPhi += nBins_phi;
341
342 iG = h_n2_yY->GetBin(iY_1,iY_2);
343 h_n2_yY->AddBinContent(iG,weight);
344 if (fillP2)
345 {
346 h_DptDpt_yY->AddBinContent(iG,weight*pt1*pt2);
347 }
348
349 iG = h_n2_DyDphi->GetBin(iDeltaY+1,iDeltaPhi+1);
350 h_n2_DyDphi ->AddBinContent(iG,weight);
351 if (fillP2)
352 {
353 h_DptDpt_DyDphi->AddBinContent(iG,weight*pt1*pt2);
354 }
355 }
356 }
357 }
358 }
359
360 // Update number of entries
361 h_n2_ptpt->SetEntries(h_n2_ptpt->GetEntries()+nPairs);
362 h_n2_phiPhi->SetEntries(h_n2_phiPhi->GetEntries()+nPairs);
363 if (fillP2)
364 {
365 h_DptDpt_phiPhi->SetEntries(h_DptDpt_phiPhi->GetEntries()+nPairs);
366 }
367 if (fillEta)
368 {
369 h_n2_etaEta->SetEntries(h_n2_etaEta->GetEntries()+nPairs);
370 h_n2_DetaDphi->SetEntries(h_n2_DetaDphi->GetEntries()+nPairs);
371 if (fillP2)
372 {
373 h_DptDpt_etaEta->SetEntries(h_DptDpt_etaEta->GetEntries()+nPairs);
374 h_DptDpt_DetaDphi->SetEntries(h_DptDpt_DetaDphi->GetEntries()+nPairs);
375 }
376 }
377
378 if (fillY)
379 {
380 h_n2_yY->SetEntries(h_n2_yY->GetEntries()+nPairs);
381 h_n2_DyDphi->SetEntries(h_n2_DyDphi->GetEntries()+nPairs);
382 if (fillP2)
383 {
384 h_DptDpt_yY->SetEntries(h_DptDpt_yY->GetEntries()+nPairs);
385 h_DptDpt_DyDphi->SetEntries(h_DptDpt_DyDphi->GetEntries()+nPairs);
386 }
387 }
388 h_n2->Fill(double(nPairs),weight);
389}
