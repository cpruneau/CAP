58TCanvas * CanvasCollection::createCanvas(const TString & canvasName, const CanvasConfiguration & configuration, int inc)
59{
60 int xInc = inc*getCollectionSize();
61 TCanvas * canvas = new TCanvas(canvasName,
62 canvasName,
63 xInc+configuration.getValueInt("windowXPosition"),
64 configuration.getValueInt("windowYPosition"),
65 configuration.getValueInt("windowWidth"),
66 configuration.getValueInt("windowHeight") );
67 canvas->SetLogx( configuration.getValueBool("windowLogX") );
68 canvas->SetLogy( configuration.getValueBool("windowLogY") );
69 canvas->SetLogz( configuration.getValueBool("windowLogZ") );
70 canvas->SetRightMargin( configuration.getValueDouble("windowRightMargin") );
71 canvas->SetLeftMargin( configuration.getValueDouble("windowLeftMargin") );
72 canvas->SetBottomMargin(configuration.getValueDouble("windowBottomMargin") );
73 canvas->SetTopMargin( configuration.getValueDouble("windowTopMargin") );
74 canvas->SetTheta( configuration.getValueDouble("windowTheta") );
75 canvas->SetPhi( configuration.getValueDouble("windowPhi") );
76
77 canvas->SetFillColor( configuration.getValueInt("windowFillColor") );
78 canvas->SetFillStyle( configuration.getValueInt("windowFillStyle") );
79 canvas->SetBorderSize( configuration.getValueDouble("windowBorderSize") );
80 canvas->SetBorderMode( configuration.getValueDouble("windowBorderMode") );
81 append(canvas);
82
83 return canvas;
84}
