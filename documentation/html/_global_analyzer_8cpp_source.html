153void GlobalAnalyzer::execute()
154{
155 // if (reportDebug("GlobalAnalyzer",getName(),"execute()"))
156 // {
157 // }
158 incrementEventProcessed();
159 Event & event = * getEventStream(0);
160 // count eventStreams used to fill histograms and for scaling at the end...
161 resetParticleCounters();
162 unsigned int nEventFilters = eventFilters.size();
163 unsigned int nParticleFilters = particleFilters.size();
164 unsigned int nParticles = event.getNParticles();
165
166 for (unsigned int iEventFilter=0; iEventFilter<nEventFilters; iEventFilter++ )
167 {
168 if (!eventFilters[iEventFilter]->accept(event)) continue;
169 incrementEventAccepted(iEventFilter);
170 if (nParticles<1) continue;
171
172 n.assign(nParticleFilters,0.0);
173 e.assign(nParticleFilters,0.0);
174 q.assign(nParticleFilters,0.0);
175 s.assign(nParticleFilters,0.0);
176 b.assign(nParticleFilters,0.0);
177 for (unsigned int iParticle=0; iParticle<event.getNParticles(); iParticle++)
178 {
179 incrementParticlesCounted();
180 Particle & particle = * event.getParticleAt(iParticle);
181 for (unsigned int iParticleFilter=0; iParticleFilter<nParticleFilters; iParticleFilter++ )
182 {
183 //cout << iParticleFilter << " " << particle.getType().getName() << endl;
184 if (particleFilters[iParticleFilter]->accept(particle))
185 {
186 incrementParticlesAccepted();
187 ParticleType & type = particle.getType();
188 TLorentzVector & momentum = particle.getMomentum();
189 n[iParticleFilter]++;
190 e[iParticleFilter] += momentum.E();
191 q[iParticleFilter] += type.getCharge();
192 s[iParticleFilter] += type.getStrange();
193 b[iParticleFilter] += type.getBaryon();
194 }
195 }
196 }
197 if (iEventFilter==0 && setEvent)
198 {
199 EventProperties * ep = event.getEventProperties();
200 ep->fill(n,e,q,s,b);
201 }
202 GlobalHistos * globalHistos = (GlobalHistos * ) histograms[iEventFilter];
203 globalHistos->fill(n,e,q,s,b,1.0);
204 }
205}
