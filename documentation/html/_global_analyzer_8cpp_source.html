148void GlobalAnalyzer::execute()
149{
150//
151// if (reportStart(__FUNCTION__))
152// ;
153 incrementTaskExecuted();
154 Event & event = * getEventStream(0);
155 // count eventStreams used to fill histograms and for scaling at the end...
156 // resetParticleCounters();
157 unsigned int nEventFilters = eventFilters.size();
158 unsigned int nParticleFilters = particleFilters.size();
159 unsigned int nParticles = event.getNParticles();
160 resetNParticlesAcceptedEvent();
161 for (unsigned int iEventFilter=0; iEventFilter<nEventFilters; iEventFilter++ )
162 {
163 if (!eventFilters[iEventFilter]->accept(event)) continue;
164 incrementNEventsAccepted(iEventFilter);
165 if (nParticles<1) continue;
166
167 n.assign(nParticleFilters,0.0);
168 e.assign(nParticleFilters,0.0);
169 q.assign(nParticleFilters,0.0);
170 s.assign(nParticleFilters,0.0);
171 b.assign(nParticleFilters,0.0);
172 for (unsigned int iParticle=0; iParticle<event.getNParticles(); iParticle++)
173 {
174 Particle & particle = * event.getParticleAt(iParticle);
175 for (unsigned int iParticleFilter=0; iParticleFilter<nParticleFilters; iParticleFilter++ )
176 {
177 //cout << iParticleFilter << " " << particle.getType().getName() << endl;
178 if (particleFilters[iParticleFilter]->accept(particle))
179 {
180 incrementNParticlesAccepted(iEventFilter,iParticleFilter);
181 // // incrementParticlesAccepted();
182 ParticleType & type = particle.getType();
183 TLorentzVector & momentum = particle.getMomentum();
184 n[iParticleFilter]++;
185 e[iParticleFilter] += momentum.E();
186 q[iParticleFilter] += type.getCharge();
187 s[iParticleFilter] += type.getStrange();
188 b[iParticleFilter] += type.getBaryon();
189 }
190 }
191 }
192 if (iEventFilter==0 && setEvent)
193 {
194 EventProperties * ep = event.getEventProperties();
195 ep->fill(n,e,q,s,b);
196 }
197 GlobalHistos * globalHistos = (GlobalHistos * ) histograms[iEventFilter];
198 globalHistos->fill(n,e,q,s,b,1.0);
199 }
200}
