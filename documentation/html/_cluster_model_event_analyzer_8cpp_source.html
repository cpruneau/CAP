89void ClusterModelEventAnalyzer::execute()
90{
91 if (reportDebug(__FUNCTION__)) cout << "ClusterModelEventAnalyzer::analyze(...) Starting" << endl;
92 incrementTaskExecuted();
93 if (!cmEvent && reportError())
94 {
95 cout << "ClusterModelEventAnalyzer::execute() Pointer to Event is null." << endl;
96 return;
97 }
98 if (!cmEventFilter->accept(*cmEvent)) return;
99// cmEvent->printProperties(cout);
100 int nCells = cmEvent->nCells;
101 if (reportDebug(__FUNCTION__)) cout << "ClusterModelEventAnalyzer::analyze(...) nCells: " << nCells << endl;
102 for (int iCell=0; iCell<nCells; iCell++)
103 {
104 Cell * cell = cmEvent->getCellAt(iCell);
105 cellHistos->fill(cell);
106 for (int iCluster=0; iCluster<cell->nClusters; iCluster++)
107 {
108 clusterHistos->fill( cmEvent->getClusterAt(iCluster) );
109 }
110 }
111 if (reportDebug(__FUNCTION__)) cout << "ClusterModelEventAnalyzer::analyze(...) Completed." << endl;
112}
