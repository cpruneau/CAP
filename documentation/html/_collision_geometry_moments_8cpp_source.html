144void CollisionGeometryMoments::calculate()
145{
146 if (m0>2)
147 {
148 meanX = m1x/m0;
149 meanY = m1x/m0;
150 varX = m2x2/m0 - m1x*m1x;
151 varY = m2y2/m0 - m1y*m1y;
152 varXY = m2xy/m0 - m1x*m1y;
153 epsDenom = varX + varY;
154 if (epsDenom>0)
155 {
156 epsX = (varY - varX)/epsDenom;
157 epsY = 2*varXY/epsDenom;
158 epsMod = sqrt(epsX*epsX + epsY*epsY);
159 psi2 = atan2(epsY,epsX);
160 }
161 }
162 else
163 {
164 meanX = -1.0E100;
165 meanY = -1.0E100;
166 varX = -1.0E100;
167 varY = -1.0E100;
168 varXY = -1.0E100;
169 }
