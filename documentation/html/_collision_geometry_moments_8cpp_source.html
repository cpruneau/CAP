148void CollisionGeometryMoments::calculate()
149{
150 if (m0>2)
151 {
152 meanX = m1x/m0;
153 meanY = m1x/m0;
154 varX = m2x2/m0 - m1x*m1x;
155 varY = m2y2/m0 - m1y*m1y;
156 varXY = m2xy/m0 - m1x*m1y;
157 epsDenom = varX + varY;
158 if (epsDenom>0)
159 {
160 epsX = (varY - varX)/epsDenom;
161 epsY = 2*varXY/epsDenom;
162 epsMod = sqrt(epsX*epsX + epsY*epsY);
163 psi2 = atan2(epsY,epsX);
164 }
165 }
166 else
167 {
168 meanX = -1.0E100;
169 meanY = -1.0E100;
170 varX = -1.0E100;
171 varY = -1.0E100;
172 varXY = -1.0E100;
173 }
