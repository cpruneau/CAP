148void EventProperties::fill(vector<double> & n, // number of particles accepted by filter #i
149 vector<double> & e, // total energy of particles accepted by filter #i
150 vector<double> & q, // total charge of particles accepted by filter #i
151 vector<double> & s, // total strangeness of particles accepted by filter #i
152 vector<double> & b)
153{
154 int nFilters = n.size();
155 for (int iFilter=0; iFilter<nFilters; iFilter++)
156 {
157 nFiltered.push_back( n[iFilter] );
158 eFiltered.push_back( e[iFilter] );
159 qFiltered.push_back( q[iFilter] );
160 sFiltered.push_back( s[iFilter] );
161 bFiltered.push_back( b[iFilter] );
162 }
163}

165void EventProperties::fillSpherocity(vector<double> & s0, vector<double> & s1)
166{
167 for (unsigned int iFilter=0; iFilter<s0.size(); iFilter++) s0Filtered.push_back( s0[iFilter] );
168 for (unsigned int iFilter=0; iFilter<s1.size(); iFilter++) s1Filtered.push_back( s1[iFilter] );
169}

181 output << " aTarget : " << aTarget << endl; // mass number target
182 output << " nPartTarget : " << nPartTarget << endl; // number of participants target
183 output << " nPartTotal : " << nPartTotal << endl; // total number of participants
184 output << " nBinaryTotal : " << nBinaryTotal << endl; // total number of binary collisions
185 output << " impactParameter : " << impactParameter << endl; // nucleus-nucleus center distance in fm
186 output << " centrality : " << centrality << endl; // fraction cross section value
187 output << " multiplicity : " << multiplicity << endl; // nominal multiplicity in the reference range
188 output << " other : " << other << endl; // other value of interest
189 nFilters = nFiltered.size();
190 output << " nFilters : " << nFilters << endl;
191 for (int k=0;k<nFilters;k++)
192 {
193 output
194 << setw(5) << k << " "
195 << scientific << setw(15)<< setprecision(5) << nFiltered[k]
196 << scientific << setw(15)<< setprecision(5) << eFiltered[k]
197 << scientific << setw(15)<< setprecision(5) << qFiltered[k]
