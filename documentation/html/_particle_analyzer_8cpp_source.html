155void ParticleAnalyzer::execute()
156{
157 incrementTaskExecuted();
158 Event & event = *eventStreams[0];
159 //Is this event accepted by this task's event filters?
160 bool analyzeThisEvent = false;
161 resetNParticlesAcceptedEvent();
162 vector<unsigned int> eventFilterPassed;
163 for (int iEventFilter=0; iEventFilter<nEventFilters; iEventFilter++ )
164 {
165 if (!eventFilters[iEventFilter]->accept(event)) continue;
166 incrementNEventsAccepted(iEventFilter);
167 eventFilterPassed.push_back(iEventFilter);
168 analyzeThisEvent = true;
169 }
170 if (!analyzeThisEvent) return;
171
172 // The event is accepted at least one event filter
173 // but it may not have pairs. If so skip out.
174 // Doing the checks in this order guarantees the accepted
175 // event count is correct for normalization purposes.
176 unsigned int nParticles = event.getNParticles();
177 if (nParticles<1) return;
178
179 vector<double> nAccepted(nParticleFilters,0.0);
180 vector<double> totalEnergy(nParticleFilters,0.0);
181
182 if (eventFilterPassed.size()>1 || nParticleFilters>1)
183 {
184 // multiple events filters or particle filters
185 // let us first reduce the number of particles to look at
186 // and generate sublists for each particle filter
187 // and use them to do the histo filling.
188
189 Factory<ParticleDigit> * factory = ParticleDigit::getFactory();
190 factory->reset();
191 int index;
192 int iEventFilter = 0;
193// float pt,e;
194// int iPt, iPhi, iEta, iY;
195 ParticleHistos * histos = (ParticleHistos *) baseSingleHistograms[0];
196 for (int iParticleFilter=0; iParticleFilter<nParticleFilters; iParticleFilter++ ) filteredParticles[iParticleFilter].clear();
197
198 resetNParticlesAcceptedEvent();
199 for (unsigned int iParticle=0; iParticle<nParticles; iParticle++)
200 {
201 Particle & particle = * event.getParticleAt(iParticle);
202 float pt,e;
203 int iPt, iPhi, iEta, iY;
204 ParticleDigit * pd;
205 bool digitized = false;
206 for (int iParticleFilter=0; iParticleFilter<nParticleFilters; iParticleFilter++ )
207 {
208 if (particleFilters[iParticleFilter]->accept(particle))
209 {
210 incrementNParticlesAccepted(iEventFilter,iParticleFilter);
211 if (!digitized)
212 {
213 TLorentzVector & momentum = particle.getMomentum();
214 pt = momentum.Pt();
215 e = momentum.E();
216 iPt = histos->getPtBinFor(pt);
217 iPhi = histos->getPhiBinFor(momentum.Phi());
218 iEta = histos->getEtaBinFor(momentum.Eta());
219 iY = histos->getYBinFor(momentum.Rapidity());
220 pd = factory->getNextObject();
221 pd->iY = iY;
222 pd->iEta = iEta;
223 pd->iPt = iPt;
224 pd->iPhi = iPhi;
225 pd->pt = pt;
226 pd->e = e;
227 digitized = true; // so no need to digitize this particle again...
228 }
229 if (digitized && ( iPt>0 || iPhi>0 || iEta>=0 || iY>=0))
230 {
231 filteredParticles[iParticleFilter].push_back(pd);
232 }
233 } // particle accepted by filter
234 } // filter loop
235 } //particle loop
236
237 // use the filter particles to fill the histos for the accepted event filters
238 for (unsigned int jEventFilter=0; jEventFilter<eventFilterPassed.size(); jEventFilter++ )
239 {
240 iEventFilter = eventFilterPassed[jEventFilter];
241 for (int iParticleFilter=0; iParticleFilter<nParticleFilters; iParticleFilter++ )
242 {
243 index = iParticleFilter+iEventFilter*nParticleFilters;
244 ParticleHistos * histos = (ParticleHistos *) baseSingleHistograms[index];
245 histos->fill(filteredParticles[iParticleFilter],1.0);
246 } // iParticleFilter loop
247 } // jEventFilter loop
248 }
249 else
250 {
251 // 1 event filter and 1 particle filter: just scan the particles...
252 // no need for sublists.
253 int iParticleFilter = 0;
254 int iEventFilter = eventFilterPassed[0];
255 int index = iParticleFilter+iEventFilter*nParticleFilters;
256 ParticleHistos * histos = (ParticleHistos *) baseSingleHistograms[index];
257 for (unsigned int iParticle=0; iParticle<nParticles; iParticle++)
258 {
259 Particle & particle = * event.getParticleAt(iParticle);
260 //TLorentzVector & momentum = particle.getMomentum();
261 if (particleFilters[iParticleFilter]->accept(particle))
262 {
263 incrementNParticlesAccepted(0,0);
264 //cout << "ParticlePairAnalyzer::execute() accepted:" << endl;
265 nAccepted[iParticleFilter]++;
266 totalEnergy[iParticleFilter] += particle.getMomentum().E();
267 histos->fill(particle,1.0);
268 }
269 }
270 histos->fillMultiplicity(nAccepted[iParticleFilter],totalEnergy[iParticleFilter],1.0);
271 }
272// all done with this event...
273}
