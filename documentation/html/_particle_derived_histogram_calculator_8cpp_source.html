179void ParticleDerivedHistogramCalculator::execute()
180{
181 TString fct = "execute()";
182 if (reportStart(fct))
183 { }
184 unsigned int nEventFilters = eventFilters.size();
185 unsigned int nParticleFilters = particleFilters.size();
186 if (reportInfo(fct))
187 {
188 cout << endl;
189 cout << "Computing derived histograms for: " << endl;
190 cout << " nEventFilters: " << nEventFilters << endl;
191 cout << " nParticleFilters: " << nParticleFilters << endl;
192 }
193 ParticleHistos * baseHistos;
194 ParticleDerivedHistos * derivedHistos;
195 unsigned index;
196
199 for (unsigned int iEventFilter=0; iEventFilter<nEventFilters; iEventFilter++ )
200 {
201 for (unsigned int iParticleFilter=0; iParticleFilter<nParticleFilters; iParticleFilter++ )
202 {
203 index = iEventFilter*nParticleFilters + iParticleFilter;
204 baseHistos = (ParticleHistos *) baseSingleHistograms[index];
205 derivedHistos = (ParticleDerivedHistos *) derivedSingleHistograms[index];
206 derivedHistos->calculateDerivedHistograms(baseHistos);
207 }
208 }
209 if (reportEnd(fct))
210 { }
