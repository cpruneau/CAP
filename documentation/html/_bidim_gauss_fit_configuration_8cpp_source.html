476void BidimGaussFitConfiguration::setExtraConfigParams(TH2 * h)
477{
478 const TAxis * xAxisHist = h->GetXaxis();
479 bwdDeltaEtaOuterLimitBin = xAxisHist->FindBin(-deltaEtaOuterLimit);
480 bwdDeltaEtaLimitBin = xAxisHist->FindBin(-deltaEtaLimit);
481 fwdDeltaEtaOuterLimitBin = xAxisHist->FindBin(deltaEtaOuterLimit);
482 fwdDeltaEtaLimitBin = xAxisHist->FindBin(deltaEtaLimit);
483 deltaEtaLimit = xAxisHist->GetBinUpEdge(fwdDeltaEtaLimitBin);
484 deltaEtaOuterLimit = xAxisHist->GetBinUpEdge(fwdDeltaEtaOuterLimitBin);
485 const TAxis * yAxisHist = h->GetYaxis();
486 deltaPhiLowLimitBin = yAxisHist->FindBin(deltaPhiLowLimit) + 1;
487 deltaPhiHighLimitBin = yAxisHist->FindBin(deltaPhiHighLimit) - 1;
488 deltaPhiLowLimit = yAxisHist->GetBinLowEdge(deltaPhiLowLimitBin);
489 deltaPhiHighLimit = yAxisHist->GetBinUpEdge(deltaPhiHighLimitBin);
490 deltaPhiHighestVal = yAxisHist->GetBinUpEdge(yAxisHist->GetNbins());
491 if (useExcludeCentralRegion)
492 {
493 valueInPatchBorder = excludeCentralRegion(deltaEtaExclusionRegion,h);
494 }
495}
496
497
498// ==============================================================================
499// Determine the low and high bins (in eta) to be used as an exclusion region in the
500// bidim gauss fit.
501// If excludedBins[iCent] < 0, use the value of deltaEtaExclusionRegion to determine
502// the bins to be excluded. Otherwise, use the number of bins specified in the
503// array excludedBins for the given "centrality" class.
504// ==============================================================================
505void BidimGaussFitConfiguration::excludedEtaBins(double deltaEtaExclusionRegion,
506 const TH2 *hist,
507 const int *excludedBins,
508 int iCent,
509 int &lowBin, int &highBin)
510{
511 const TAxis * xAxisHist = hist->GetXaxis();
512 if (excludedBins[iCent] < 0)
513 {
514 lowBin = xAxisHist->FindBin(-deltaEtaExclusionRegion);
515 highBin = xAxisHist->FindBin( deltaEtaExclusionRegion);
516 }
517 else
518 {
519 lowBin = xAxisHist->FindBin(0.0) - excludedBins[iCent];
520 highBin = xAxisHist->FindBin(0.0) + excludedBins[iCent];
521 }
522}
523
524// ==============================================================================
525// Determine the low and high bins (in phi) to be used as an exclusion region in the
526// bidim gauss fit.
527// If excludedBins[iCent] < 0, use the value of deltaEtaExclusionRegion to determine
528// the bins to be excluded. Otherwise, use the number of bins specified in the
529// array excludedBins for the given "centrality" class.
530// ==============================================================================
531void BidimGaussFitConfiguration::excludedPhiBins(double deltaEtaExclusionRegion,
532 const TH2 *hist,
533 const int *excludedBins,
534 int iCent,
535 int &lowBin,
536 int &highBin)
537{
538 const TAxis * yAxisHist = hist->GetYaxis();
539 if (excludedBins[iCent] < 0)
540 {
541 lowBin = yAxisHist->FindBin(-deltaEtaExclusionRegion);
542 highBin = yAxisHist->FindBin(deltaEtaExclusionRegion);
543 }
544 else {
545 lowBin = yAxisHist->FindBin(0.0) - excludedBins[iCent];
546 highBin = yAxisHist->FindBin(0.0) + excludedBins[iCent];
547 }
548}

550double BidimGaussFitConfiguration::excludeCentralRegion(double deltaEtaExclusionRegion,TH2 *hist)
551{
552 const TAxis * xAxisHist = hist->GetXaxis();
553 const TAxis * yAxisHist = hist->GetYaxis();
554 bool usedRegion = kTRUE;
555
556 // work variables - at the end, we save these into in patch variables...
557 int lowBinEta;
558 int highBinEta;
559 int lowBinPhi;
560 int highBinPhi;
561
562 switch (collisionSystem)

648 for (int biny = lowBinPhi; biny <= highBinPhi; biny++)

651 float y = float(abs(biny-yzero));

676 for (int biny = lowBinPhi; biny <= highBinPhi; biny++)
677 {
678 if (biny == lowBinPhi) valueInPatchBorder = TMath::Max(valueInPatchBorder,hist->GetBinContent(binx,biny));
679 else if (biny == highBinPhi) valueInPatchBorder = TMath::Max(valueInPatchBorder,hist->GetBinContent(binx,biny));
680 else
681 {
682 if (binx == lowBinEta || binx == highBinEta)
683 {
684 valueInPatchBorder = TMath::Max(valueInPatchBorder,hist->GetBinContent(binx,biny));
685 }
686 }
687 hist->SetBinError(binx, biny, 1e6);
688 }
689 }
690 Info("fitDeltaPhi2DOneFunction","NOTE : Skipping bins at (%d, %d) at (%d) = (0) in delta eta",
691 lowBinEta, highBinEta, xAxisHist->FindBin(0.0));
692 Info("fitDeltaPhi2DOneFunction","NOTE : Skipping bins at (%d, %d) at (%d) = (0) in delta phi",
693 lowBinPhi, highBinPhi, yAxisHist->FindBin(0.0));
694 }
695
696 lowBinEtaInPatch = lowBinEta;
697 highBinEtaInPatch = highBinEta;
698 lowBinPhiInPatch = lowBinPhi;
699 highBinPhiInPatch = highBinPhi;
