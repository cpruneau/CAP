47EnhancedGraph::EnhancedGraph(const TString & graphName,
48 const TString & xTitle,
49 const TString & yTitle,
50 double minX, double maxX,
51 double minY, double maxY,
52 int n, double * x, double *y, double *ex, double *ey,
53 double * yM, double * yP,
54 int opt,
55 bool absolute)
56: TH1D(graphName,graphName, 100, minX, maxX),
57nPoints(n),
58xPtr(x),
59exPtr(ex),
60yPtr(y),
61eyPtr(ey),
62ySysMPtr(yM),
63ySysPPtr(yP),
64yM(0),
65yP(0),
66sysAbsolute(absolute),
67option(opt)
68{
69 SetXTitle(xTitle);
70 SetYTitle(yTitle);
71 SetMinimum(minY);
72 SetMaximum(maxY);
73 // graph w/ stat error
74 g = new TGraphErrors(n, x, y, ex, ey);
75
76 switch (option)
77 {
78 case 0:
79 yM = new double[nPoints];
80 yP = new double[nPoints];
81 for (int k=0; k<n; k++)
82 {
83 yM[k] = y[k] + ySysMPtr[k];
84 yP[k] = y[k] + ySysPPtr[k];
85 }
86 gP = new TGraph(n, x, yP);
87 gM = new TGraph(n, x, yM);
88 gSys = 0;
89 break;
90 case 1:
91 gP = 0;
92 gM = 0;
93 yM = new double[nPoints];
94 yP = new double[nPoints];
95 for (int k=0; k<n; k++)
96 {
97 yM[k] = ySysMPtr[k];
98 yP[k] = ySysPPtr[k];
99 }
100 gSys = new TGraphAsymmErrors(n, x, y, ex, ex, yM, yP);
101 break;
102 }
103
104 isVisible = true;
105}
