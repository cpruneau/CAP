52EnhancedGraph::EnhancedGraph(const TString & graphName,
53 const TString & xTitle,
54 const TString & yTitle,
55 double minX, double maxX,
56 double minY, double maxY,
57 int n, double * x, double *y, double *ex, double *ey,
58 double * yM, double * yP,
59 int opt,
60 bool absolute)
61: TH1D(graphName,graphName, 100, minX, maxX),
62nPoints(n),
63xPtr(x),
64exPtr(ex),
65yPtr(y),
66eyPtr(ey),
67ySysMPtr(yM),
68ySysPPtr(yP),
69yM(0),
70yP(0),
71sysAbsolute(absolute),
72option(opt)
73{
74 SetXTitle(xTitle);
75 SetYTitle(yTitle);
76 SetMinimum(minY);
77 SetMaximum(maxY);
78 // graph w/ stat error
79 g = new TGraphErrors(n, x, y, ex, ey);
80
81 switch (option)
82 {
83 case 0:
84 yM = new double[nPoints];
85 yP = new double[nPoints];
86 for (int k=0; k<n; k++)
87 {
88 yM[k] = y[k] + ySysMPtr[k];
89 yP[k] = y[k] + ySysPPtr[k];
90 }
91 gP = new TGraph(n, x, yP);
92 gM = new TGraph(n, x, yM);
93 gSys = 0;
94 break;
95 case 1:
96 gP = 0;
97 gM = 0;
98 yM = new double[nPoints];
99 yP = new double[nPoints];
100 for (int k=0; k<n; k++)
101 {
102 yM[k] = ySysMPtr[k];
103 yP[k] = ySysPPtr[k];
104 }
105 gSys = new TGraphAsymmErrors(n, x, y, ex, ex, yM, yP);
106 break;
107 }
108
109 isVisible = true;
110}
