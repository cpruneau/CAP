50double TwoDCubic(double *x, double *par)
51{
52 double xx = x[0];
53 double yy = x[1];
54
55 double v = 0;
56 for (int iX=0; iX<3; iX++)
57 {
58 for (int iY=0; iY<3; iY++)
59 {
60 int index = 3*iX + iY;
61 v += par[index]*TMath::Power(xx,iX)*TMath::Power(yy,iY);
62 }
63 }
64 return v;
65}
