69void NucleusGenerator::initialize()
70{
71
72 if (reportStart(__FUNCTION__))
73 ;
74 gType = configuration.getValueInt("generatorType");
75 nR = configuration.getValueInt("nRadiusBins");
76 minR = configuration.getValueDouble("minimumRadius");
77 maxR = configuration.getValueDouble("maximumRadius");
78 parA = configuration.getValueDouble("parA");
79 parB = configuration.getValueDouble("parB");;
80 parC = configuration.getValueDouble("parC");
81 useRecentering = configuration.getValueInt("useRecentering");
82 useNucleonExclusion = configuration.getValueInt("useNucleonExclusion");
83 exclusionRadius = configuration.getValueInt("exclusionRadius"); // fm
84 exclusionRadiusSq = exclusionRadius*exclusionRadius;
85 double dr = (maxR-minR)/double(nR);
86 double r = minR + dr/2.0;
87 double density;
88 double profile;
89
90 if (rDensity) delete rDensity; rDensity = nullptr;
91 if (rProfile) delete rProfile; rProfile = nullptr;
92 if (rProfileGen) delete rProfileGen; rProfileGen = nullptr;
93
94 TString baseName;
95 TString histoName;
96 switch (gType)
97 {
98 case 0: baseName += "_Uniform"; break;
99 case 1: baseName += "_WoodsSaxon"; break;
100 case 2: baseName += "_Exponential"; break;
101 case 3: baseName += "_Gaussian"; break;
102 case 4: baseName += "_DoubleGaussian"; break;
103 }
104
105
106 histoName = baseName + "_rDensity";
107 rDensity = new TH1D(histoName,histoName,nR,minR, maxR);
108 histoName = baseName + "_rProfile";
109 rProfile = new TH1D(histoName,histoName,nR,minR, maxR);
110 histoName = baseName + "_rProfileGen";
111 rProfileGen = new TH1D(histoName,histoName,nR,minR, maxR);
112
113
114 for (int iR=0; iR<nR; iR++)
115 {
116 double r2 = r*r;
117 switch (gType)
118 {
119 case 0: // uniform hard sphere
120 density = (r*r*r<parA)? 1.0 : 0.0;
121 break;
122 case 1: // Woods-Saxon/Fermi
123 density = 1.0/(1.0+exp((r-parA)/parB) );
124 break;
125 case 2: // exponential
126 density = exp(-r/parA);
127 break;
128 case 3: // gaussian
129 density = exp(-r2/2.0/parA/parA);
130 break;
131 case 4: //double-gaussian
132 density = (1.0-parC)*exp(-r2/parA/parA)/parA/parA/parA;
133 density += parC*exp(-r2/parB/parB)/parB/parB/parB;
134 break;
135 }
136 profile = r2*density;
137 rDensity->SetBinContent(iR+1,density);
138 rDensity->SetBinError(iR+1,0.0);
139 rProfile->SetBinContent(iR+1,profile);
140 rProfile->SetBinError(iR+1,0.0);
141 r += dr;
142 }
143 if (reportEnd(__FUNCTION__))
144 ;
145}

152void NucleusGenerator::generate(Nucleus & nucleus, double xShift)
153{
154 double r, cosTheta, phi;
155 TLorentzVector position(0.0,0.0,0.0,0.0);
156 //nucleus.reset(); already handled by the collision geometry
157 unsigned int iNucleon = 0;
158 unsigned int nNucleons = nucleus.getNNucleons();
159 while (iNucleon < nNucleons)
160 {
161 Particle * nucleon = nucleus.getNucleonAt(iNucleon);
162 generate(r, cosTheta, phi);
163 nucleon->setRCosThetaPhiT(r,cosTheta,phi,0.0);
164 //nucleon->printProperties(cout);
165 int sanityCheck = 0;
166 if (useNucleonExclusion)
167 {
168 bool reject = false;
169 for (unsigned int jNucleon=0; jNucleon<iNucleon; jNucleon++)
170 {
171 Particle * otherNucleon = nucleus.getNucleonAt(jNucleon);
172 if (nucleon->distanceXYZSq(otherNucleon) < exclusionRadiusSq)
173 {
174 reject = true;
175 break;
176 }
177 }
178 if (reject)
179 {
180 sanityCheck++;
181 if (sanityCheck>200)
182 {
183 cout << "Nucleon rejected > 200 times" << endl;
184 exit(1);
185 }
186 //cout << "reject sanityCheck:" << sanityCheck << endl;
187 continue;
188 }
189 }
190 position += nucleon->getPosition();
191 if (iNucleon<nucleus.getNProtons())
192 nucleon->setType(ParticleType::getProtonType());
193 else
194 nucleon->setType(ParticleType::getNeutronType());
195 iNucleon++;
196 }
197 // center of mass, recenter
198 position *= 1.0/double(nNucleons);
199 TLorentzVector shift(xShift -position.X(),-position.Y(),-position.Z(),0.0);
200
201 for (unsigned int iNucleon=0; iNucleon<nNucleons; iNucleon++)
202 {
203 Particle * nucleon = nucleus.getNucleonAt(iNucleon);
204 nucleon->shift(shift);
205 }
206
207}
