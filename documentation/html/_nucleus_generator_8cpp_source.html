52void NucleusGenerator::initialize(const TString & _generatorTypeName,
53 int _generatorType,
54 double _parA, double _parB, double _parC,
55 int _nR, double _minR, double _maxR,
56 bool _useRecentering,
57 bool _useNucleonExclusion,
58 double _exclusionRadius)
59{
60 nR = _nR;
61 minR = _minR;
62 maxR = _maxR;
63 parA = _parA;
64 parB = _parB;
65 parC = _parC;
66 generatorTypeName = _generatorTypeName;
67 generatorType = _generatorType;
68 useRecentering = _useRecentering;
69 useNucleonExclusion = _useNucleonExclusion;
70 exclusionRadius = _exclusionRadius; // fm
71 exclusionRadiusSq = exclusionRadius*exclusionRadius;
72 if (rDensity) delete rDensity; rDensity = nullptr;
73 if (rProfile) delete rProfile; rProfile = nullptr;
74 if (rProfileGen) delete rProfileGen; rProfileGen = nullptr;
75
76 switch (generatorType)
77 {
78 case Uniform: generatorTypeName += "_Uniform"; break;
79 case WoodsSaxon: generatorTypeName += "_WoodsSaxon"; break;
80 case Exponential: generatorTypeName += "_Exponential"; break;
81 case Gaussian: generatorTypeName += "_Gaussian"; break;
82 case DoubleGaussian: generatorTypeName += "_DoubleGaussian"; break;
83 }
84
85 TString histoName;
86 histoName = generatorTypeName + "_rDensity";
87 rDensity = new TH1D(histoName,histoName,nR,minR, maxR);
88 histoName = generatorTypeName + "_rProfile";
89 rProfile = new TH1D(histoName,histoName,nR,minR, maxR);
90 histoName = generatorTypeName + "_rProfileGen";
91 rProfileGen = new TH1D(histoName,histoName,nR,minR, maxR);
92
93 double dr = (maxR-minR)/double(nR);
94 double r = minR + dr/2.0;
95 double density, profile;
96 for (int iR=0; iR<nR; iR++)
97 {
98 double r2 = r*r;
99 switch (generatorType)
100 {
101 case Uniform: // uniform hard sphere
102 density = (r*r*r<parA)? 1.0 : 0.0;
103 break;
104 case WoodsSaxon: // Woods-Saxon/Fermi
105 density = 1.0/(1.0+exp((r-parA)/parB) );
106 break;
107 case Exponential: // exponential
108 density = exp(-r/parA);
109 break;
110 case Gaussian: // gaussian
111 density = exp(-r2/2.0/parA/parA);
112 break;
113 case DoubleGaussian: //double-gaussian
114 density = (1.0-parC)*exp(-r2/parA/parA)/parA/parA/parA;
115 density += parC*exp(-r2/parB/parB)/parB/parB/parB;
116 break;
117 }
118 profile = r2*density;
119 rDensity->SetBinContent(iR+1,density);
120 rDensity->SetBinError(iR+1,0.0);
121 rProfile->SetBinContent(iR+1, profile);
122 rProfile->SetBinError(iR+1,0.0);
123 r += dr;
124 }
125}

127void NucleusGenerator::generate(Nucleus & nucleus, double xShift)
128{
129 double r, cosTheta, phi;
130 TLorentzVector position(0.0,0.0,0.0,0.0);
131 //nucleus.reset(); already handled by the collision geometry
132 unsigned int iNucleon = 0;
133 unsigned int nNucleons = nucleus.getNNucleons();
134 while (iNucleon < nNucleons)
135 {
136 Particle * nucleon = nucleus.getNucleonAt(iNucleon);
137 generate(r, cosTheta, phi);
138 nucleon->setRCosThetaPhiT(r,cosTheta,phi,0.0);
139 //nucleon->printProperties(cout);
140 int sanityCheck = 0;
141 if (useNucleonExclusion)
142 {
143 bool reject = false;
144 for (unsigned int jNucleon=0; jNucleon<iNucleon; jNucleon++)
145 {
146 Particle * otherNucleon = nucleus.getNucleonAt(jNucleon);
147 if (nucleon->distanceXYZSq(otherNucleon) < exclusionRadiusSq)
148 {
149 reject = true;
150 break;
151 }
152 }
153 if (reject)
154 {
155 sanityCheck++;
156 if (sanityCheck>200)
157 {
158 cout << "Nucleon rejected > 200 times" << endl;
159 exit(1);
160 }
161 //cout << "reject sanityCheck:" << sanityCheck << endl;
162 continue;
163 }
164 }
165 position += nucleon->getPosition();
166 if (iNucleon<nucleus.getNProtons())
167 nucleon->setType(ParticleType::getProtonType());
168 else
169 nucleon->setType(ParticleType::getNeutronType());
170 iNucleon++;
171 }
172 // center of mass, recenter
173 position *= 1.0/double(nNucleons);
174 TLorentzVector shift(xShift -position.X(),-position.Y(),-position.Z(),0.0);
175
176 for (unsigned int iNucleon=0; iNucleon<nNucleons; iNucleon++)
177 {
178 Particle * nucleon = nucleus.getNucleonAt(iNucleon);
179 nucleon->shift(shift);
180 }
181
182}
