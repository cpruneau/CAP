77void CollisionGeometry::addNNCollision(Particle* nucleonA,
78 Particle* nucleonB)
79{
80
81 Particle interaction;
82 interaction.setParents(nucleonA,nucleonB);
83
84 //cout << " A: x= " << nucleonA->getPosition().X() << " B: x= " << nucleonB->getPosition().X() << " I: x= " << interaction.getPosition().X() << endl;
85
86 nnInteractions.push_back(interaction);
87
88 if (!nucleonA->isWounded())
89 {
90 TLorentzVector & positionA = nucleonA->getPosition();
91 participantMoments.fill(positionA.X() ,positionA.Y());
92 nucleonA->setWounded(true);
93 }
94 if (!nucleonB->isWounded())
95 {
96 TLorentzVector & positionB = nucleonB->getPosition();
97 participantMoments.fill(positionB.X() ,positionB.Y());
98 nucleonB->setWounded(true);
99 }
100 TLorentzVector & positionInt = interaction.getPosition();
101 binaryMoments.fill(positionInt.X(),positionInt.Y());
102}
