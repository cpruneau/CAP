78void GaussianGeneratorTask::execute()
79{
80 incrementTaskExecuted();
81 Event & event = * eventStreams[0];
82
83 int nParticles = event.getNParticles();
84 for (int itrack = 0; itrack < nParticles; itrack++)
85 {
86 Particle *particle1 = event.getParticleAt(itrack);
87 for (int jtrack = itrack+1; jtrack < nParticles; jtrack++)
88 {
89 Particle *particle2 = event.getParticleAt(jtrack);
90 double deltaeta = particle1->getMomentum().Eta() - particle2->getMomentum().Eta();
91 double deltaphi = TVector2::Phi_mpi_pi(particle1->getMomentum().Phi() - particle2->getMomentum().Phi());
92 double weight; weight= profile->Eval(deltaeta,deltaphi) + 1.0;
93 }
94 }
