70void GaussianGeneratorTask::execute()
71{
72 Event & event = * eventStreams[0];
73
74 int nParticles = event.getNParticles();
75 for (int itrack = 0; itrack < nParticles; itrack++)
76 {
77 Particle *particle1 = event.getParticleAt(itrack);
78 for (int jtrack = itrack+1; jtrack < nParticles; jtrack++)
79 {
80 Particle *particle2 = event.getParticleAt(jtrack);
81 double deltaeta = particle1->getMomentum().Eta() - particle2->getMomentum().Eta();
82 double deltaphi = TVector2::Phi_mpi_pi(particle1->getMomentum().Phi() - particle2->getMomentum().Phi());
83 double weight; weight= profile->Eval(deltaeta,deltaphi) + 1.0;
84 }
85 }
