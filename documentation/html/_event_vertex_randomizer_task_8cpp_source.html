75void EventVertexRandomizerTask::execute()
76{
77
78 if (reportStart(__FUNCTION__))
79 ;
80 incrementTaskExecuted();
81 double eventX = gRandom->Gaus(rConversion*xAvg, rConversion*xRms);
82 double eventY = gRandom->Gaus(rConversion*yAvg, rConversion*yRms);
83 double eventZ = gRandom->Gaus(rConversion*zAvg, rConversion*zRms);
84 double eventT = gRandom->Gaus(tConversion*tAvg, tConversion*tRms);
85 Event * event = eventStreams[0];
86 unsigned int nParticles = event->getNParticles();
87 for (unsigned int iParticle = 0; iParticle < nParticles; iParticle++)
88 {
89 Particle * particle = event->getParticleAt(iParticle);
90 particle->shift(eventX,eventY,eventZ,eventT);
91 }
