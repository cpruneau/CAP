152void GlobalHistos::fill(vector<double> & n, vector<double> & e, vector<double> & q, vector<double> & s, vector<double> & b, double weight)
153{
154 unsigned int nParticleFilters = particleFilters.size();
155 unsigned int nAll = 0; // all
156 unsigned int nND = 0; // non diagonal only
157 for (unsigned int k1=0; k1<nParticleFilters; k1++)
158 {
159 h_n[k1]->Fill(n[k1],weight);
160 h_e[k1]->Fill(e[k1],weight);
161 h_q[k1]->Fill(q[k1],weight);
162 h_s[k1]->Fill(s[k1],weight);
163 h_b[k1]->Fill(b[k1],weight);
164 if ( fillCorrelationHistos )
165 {
166 //unsigned int n2 = nParticleFilters-k1;
167 for (unsigned int k2=k1; k2<nParticleFilters; k2++)
168 {
169 if (k1!=k2)
170 {
171 h_nVsN[nND]->Fill(n[k1],n[k2],weight);
172 h_eVsE[nND]->Fill(e[k1],e[k2],weight);
173 h_qVsQ[nND]->Fill(q[k1],q[k2],weight);
174 h_bVsB[nND]->Fill(b[k1],b[k2],weight);
175 nND++;
176 }
177 h_eVsN[nAll]->Fill(n[k1],e[k2],weight);
178 h_qVsN[nAll]->Fill(n[k1],q[k2],weight);
179 h_bVsN[nAll]->Fill(n[k1],b[k2],weight);
180 h_qVsE[nAll]->Fill(e[k1],q[k2],weight);
181 h_bVsE[nAll]->Fill(e[k1],b[k2],weight);
182 h_bVsQ[nAll]->Fill(q[k1],b[k2],weight);
183 nAll++;
184 }
185 }
