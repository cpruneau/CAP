109void GlobalHistos::loadHistograms(TFile * inputFile)
110{
111 if (reportStart(__FUNCTION__))
112 ;
113 if (!inputFile)
114 {
115 if (reportFatal(__FUNCTION__)) cout << "Attempting to load GlobalHistos from an invalid file pointer" << endl;
116 return;
117 }
118 const Configuration & configuration = getConfiguration();
119 fillCorrelationHistos = configuration.getValueBool("fillCorrelationHistos");
120
121
122 TString bn = getName();
123 TString name;
124 unsigned int nParticleFilters = particleFilters.size();
125 for (unsigned int k1=0; k1<nParticleFilters; k1++)
126 {
127 TString pfName1 = particleFilters[k1]->getName();

129 h_e.push_back( loadH1(inputFile, makeName(bn,pfName1,"e")) );
130 h_q.push_back( loadH1(inputFile, makeName(bn,pfName1,"q")) );

134 {

136 {
137 TString pfName2 = particleFilters[k2]->getName();
138 if (k1!=k2)
139 {
140 h_nVsN.push_back( loadH2(inputFile, makeName(bn,pfName1,"n",pfName2,"n")) );
141 h_eVsE.push_back( loadH2(inputFile, makeName(bn,pfName1,"e",pfName2,"e")) );
142 h_qVsQ.push_back( loadH2(inputFile, makeName(bn,pfName1,"q",pfName2,"q")) );
143 h_bVsB.push_back( loadH2(inputFile, makeName(bn,pfName1,"b",pfName2,"b")) );
144 }
145 h_eVsN.push_back( loadH2(inputFile, makeName(bn,pfName1,"n",pfName2,"e")) );
146 h_qVsN.push_back( loadH2(inputFile, makeName(bn,pfName1,"n",pfName2,"q")) );

164void GlobalHistos::fill(vector<double> & n, vector<double> & e, vector<double> & q, vector<double> & s, vector<double> & b, double weight)
165{
166 unsigned int nParticleFilters = particleFilters.size();
167 unsigned int nAll = 0; // all
168 unsigned int nND = 0; // non diagonal only
169 for (unsigned int k1=0; k1<nParticleFilters; k1++)
170 {
171 h_n[k1]->Fill(n[k1],weight);
172 h_e[k1]->Fill(e[k1],weight);
173 h_q[k1]->Fill(q[k1],weight);
174 h_s[k1]->Fill(s[k1],weight);
175 h_b[k1]->Fill(b[k1],weight);
176 if ( fillCorrelationHistos )
177 {
178 //unsigned int n2 = nParticleFilters-k1;
179 for (unsigned int k2=k1; k2<nParticleFilters; k2++)
180 {
181 if (k1!=k2)
182 {
183 h_nVsN[nND]->Fill(n[k1],n[k2],weight);
184 h_eVsE[nND]->Fill(e[k1],e[k2],weight);
185 h_qVsQ[nND]->Fill(q[k1],q[k2],weight);
186 h_bVsB[nND]->Fill(b[k1],b[k2],weight);
187 nND++;
188 }
189 h_eVsN[nAll]->Fill(n[k1],e[k2],weight);
190 h_qVsN[nAll]->Fill(n[k1],q[k2],weight);
191 h_bVsN[nAll]->Fill(n[k1],b[k2],weight);
192 h_qVsE[nAll]->Fill(e[k1],q[k2],weight);
193 h_bVsE[nAll]->Fill(e[k1],b[k2],weight);
194 h_bVsQ[nAll]->Fill(q[k1],b[k2],weight);
195 nAll++;
196 }
197 }
