92void TaskIterator::execute()
93{
94 timer.start();
95 initialize();
96 if (!isTaskOk()) return;
97 for (long iter=0; iter<nIterationRequested; iter++)
98 {
99 if (useParticles)
100 {
101 Particle::resetFactory();
102 Event::resetEventStreams();
103 }
104 unsigned int nSubTasks = subTasks.size();
105 for (unsigned int iTask=0; iTask<nSubTasks; iTask++) subTasks[iTask]->execute();
106 incrementTaskExecuted();
107 if (doPartialReports && getNTaskExecutedReset()%nIterationReported==0) cout << "Completed iteration # " << iter << endl;
108 if (doPartialSaves && getNTaskExecutedReset()%nIterationPartialSave==0 )
109 {
110 savePartial();
111 if (doSubsampleAnalysis) reset();
112 }
113
114 if (getNTaskExecuted() >= nIterationRequested ) break;
115 }
116 timer.stop();
117 finalize();
118}
