53void TaskIterator::run()
54{
55 timer.start();
56 nEventRequested = configuration.getValueLong( "nEventsRequested" );
57 nEventReported = configuration.getValueLong( "nEventReported" );
58 nEventPartialSave = configuration.getValueLong( "nEventPartialSave" );
59 doPartialSave = configuration.getValueBool( "doPartialSave" );
60 doSubsampleAnalysis = configuration.getValueBool( "doSubsampleAnalysis");
61
62 if (reportInfo("TaskIterator",getName(),"run(...)")) cout << "Running for nEvent: " << nEventRequested << endl;
63 postTaskOk();
64 initializeTasks();
65 if (!isTaskOk())
66 {
67 if (reportWarning("TaskIterator",getName(),"run(...)")) cout << "Initialization failed. Abort." << endl;
68 return;
69 }
70 nEventProcessed = 0;
71 if (reportInfo("TaskIterator",getName(),"run(...)")) cout << "Starting with nEventRequested:" << nEventRequested << endl;
72 for (long iEvent=1; iEvent<=nEventRequested; iEvent++)
73 {
74 executeTasks();
75 if (!isTaskOk())
76 {
77 if (reportWarning("TaskIterator",getName(),"run(...)")) cout << "Error detected after event:" << iEvent << endl;
78 break;
79 }
80 if (nEventProcessed%nEventReported==0 )
81 {
82 if (reportInfo("TaskIterator",getName(),"run(...)")) cout << "Completed event # " << iEvent << endl;
83 timer.stop();
84 timer.print(cout);
85 }
86 if ( (doSubsampleAnalysis||doPartialSave) && nEventProcessed%nEventPartialSave==0)
87 {
88 savePartialTasks();
89 if (!isTaskOk()) break;
90 if (doSubsampleAnalysis) resetTasks();
91 postTaskOk();
92 }
93 }
94 if (isTaskOk()) finalizeTasks();
95
96 if (isTaskOk() && doSubsampleAnalysis)
97 {
98 subsampleAnalysisTasks();
99 }
100
101 timer.stop();
102 if (reportInfo("TaskIterator",getName(),"run(...)"))
103 {
104 cout << endl;
105 cout << " Completed with status : " << getTaskStatusName() << endl;
106 cout << " Completed Events : " << nEventProcessed << endl;
107 cout << " Requested Events : " << nEventRequested << endl;
108 cout << " "; timer.print(cout);
109 cout << endl << endl<< endl << endl;
110 }
111}
