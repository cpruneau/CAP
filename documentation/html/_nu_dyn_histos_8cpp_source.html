40void NuDynHistos::createHistograms()
41{
42 TString bn = getHistoBaseName();
43 Configuration & configuration = getConfiguration();
44 nFilters = configuration.getValueInt("nFilters");
45 multiplicityType = configuration.getValueInt("multiplicityType");
46 pairOnly = configuration.getValueBool("pairOnly");
47 int nBins_mult = configuration.getValueInt("nBins_mult");
48 double min_mult = configuration.getValueInt("min_mult");
49 double max_mult = configuration.getValueInt("max_mult");
50
51 TString suffix = "";
52 TString xTitle = "";
53
54 switch (multiplicityType)
55 {
56 case 0: suffix = "_vsCent"; xTitle = "%"; break;
57 case 1: suffix = "_vsMult"; xTitle = "mult_{Tot}"; break;
58 case 2: suffix = "_vsMult"; xTitle = "mult_{acc}"; break;
59 }
60
61 h_eventStreams = createHistogram(bn+TString("NeventStreams"),10,0.0,10, "Streams","n_{Events}");
62 h_eventStreams_vsMult = createHistogram(bn+TString("NeventStreams")+suffix,nBins_mult,min_mult, max_mult, xTitle,"n_{Events}");
63 for (unsigned int i1=0; i1<nFilters; i1++)
64 {
65 h_f1.push_back( createProfile(makeName(bn+"f1",i1), 10,0.0,10, "Streams", makeName("f_{1}^{",i1,"}")) );
66 h_f1_vsMult.push_back( createProfile(makeName(bn+TString("f1")+suffix,i1),nBins_mult,min_mult,max_mult,xTitle, makeName("f_{1}^{",i1,"}") ) );
67 for (unsigned int i2=i1; i2<nFilters; i2++)
68 {
69 h_f2.push_back( createProfile(makeName(bn+"f2",i1,i2), 10,0.0,10, "Streams", makeName("f_{2}^{",i1,i2,"}")) );
70 h_f2_vsMult.push_back( createProfile(makeName(bn+TString("f2")+suffix,i1,i2),nBins_mult,min_mult,max_mult,xTitle,makeName("f_{2}^{",i1,i2,"}")) );
71 if (!pairOnly)
72 {
73 for (unsigned int i3=i2; i3<nFilters; i3++)
74 {
75 h_f3.push_back( createProfile(makeName(bn+"f3",i1,i2,i3), 10,0.0,10, "Streams", makeName("f_{3}^{",i1,i2,i3,"}")) );
76 h_f3_vsMult.push_back( createProfile(makeName(bn+TString("f3")+suffix,i1,i2,i3),nBins_mult,min_mult,max_mult,xTitle,makeName("f_{3}^{",i1,i2,i3,"}")) );
77 for (unsigned int i4=i3; i4<nFilters; i4++)
78 {
79 h_f4.push_back( createProfile(makeName(bn+"f4",i1,i2,i3,i4), 10,0.0,10, "Streams", makeName("f_{4}^{",i1,i2,i3,i4,"}")));
80 h_f4_vsMult.push_back( createProfile(makeName(bn+TString("f4")+suffix,i1,i2,i3,i4),nBins_mult,min_mult,max_mult,xTitle,makeName("f_{4}^{",i1,i2,i3,i4,"}")));
81 }
82 }
83 } // !pairOnly
84 }
85 }

89void NuDynHistos::loadHistograms(TFile * inputFile)
90{
91 if (!inputFile)
92 {
93 if (reportFatal()) cout << "-Fatal- Attempting to load NuDynHistos from an invalid file pointer" << endl;
94 return;
95 }
96 Configuration & configuration = getConfiguration();
97 nFilters = configuration.getValueInt("nFilters");
98 multiplicityType = configuration.getValueInt("multiplicityType");
99 pairOnly = configuration.getValueBool("pairOnly");
100
101 TString bn = getHistoBaseName();
102 TString suffix = "";
103 switch ( multiplicityType )
104 {
105 case 0: suffix = "_vsCent"; break;
106 case 1: suffix = "_vsMult"; break;
107 case 2: suffix = "_vsMult"; break;
108 }
109 h_eventStreams = loadH1(inputFile,bn+TString("NeventStreams"));
110 h_eventStreams_vsMult = loadH1(inputFile,bn+TString("NeventStreams")+suffix);
111
112 for (unsigned int i1=0; i1<nFilters; i1++)
113 {
114 h_f1.push_back(loadProfile(inputFile,makeName(bn+"f1",i1)));
115 h_f1_vsMult.push_back(loadProfile(inputFile,makeName(bn+TString("f1")+suffix,i1)));
116 for (unsigned int i2=i1; i2<nFilters; i2++)
117 {
118 h_f2.push_back(loadProfile(inputFile,makeName(bn+"f2",i1,i2)));
119 h_f2_vsMult.push_back(loadProfile(inputFile,makeName(bn+TString("f2")+suffix,i1,i2)));
120 if (!pairOnly)
121 {
122 for (unsigned int i3=i2; i3<nFilters; i3++)
123 {
124 h_f3.push_back(loadProfile(inputFile,makeName(bn+"f3",i1,i2,i3)));
125 h_f3_vsMult.push_back(loadProfile(inputFile,makeName(bn+TString("f3")+suffix,i1,i2,i3)));
126 for (unsigned int i4=i3; i4<nFilters; i4++)
127 {
128 h_f4.push_back(loadProfile(inputFile,makeName(bn+"f4",i1,i2,i3,i4)));

132 } // !pairOnly

138void NuDynHistos::fill(double mult, vector<double> & nAccepted, double weight)
139{
140 h_eventStreams->Fill(mult, weight);
141 h_eventStreams_vsMult->Fill(mult, weight);
142 double fill;
143
144 int k1 = 0;
145 int k2 = 0;
146 int k3 = 0;
147 int k4 = 0;
148 for (unsigned int i1=0; i1<nFilters; i1++)
149 {
150 fill = nAccepted[i1];
151 h_f1[k1]->Fill(mult,fill,weight);
152 h_f1_vsMult[k1]->Fill(mult,fill,weight);
153 for (unsigned int i2=i1; i2<nFilters; i2++)
154 {
155 fill = nAccepted[i1]*(nAccepted[i2]-((i2==i1)?1:0));
156 h_f2[k2]->Fill(mult,fill,weight);
157 h_f2_vsMult[k2]->Fill(mult,fill,weight);
158 if (!pairOnly)
159 {
160 for (unsigned int i3=i2; i3<nFilters; i3++)
161 {
162 fill = nAccepted[i1]
163 * (nAccepted[i2]-((i2==i1)?1:0))
164 * (nAccepted[i3]-((i3==i1)?1:0)-((i3==i2)?1:0));
165 h_f3[k3]->Fill(mult,fill,weight);
166 h_f3_vsMult[k3]->Fill(mult,fill,weight);
167 k3++;
168 for (unsigned int i4=i3; i4<nFilters; i4++)
169 {
170 fill = nAccepted[i1]
171 * (nAccepted[i2]-((i2==i1)?1:0))
172 * (nAccepted[i3]-((i3==i1)?1:0)-((i3==i2)?1:0))
173 * (nAccepted[i4]-((i4==i1)?1:0)-((i4==i2)?1:0)-((i4==i3)?1:0)) ;
174 h_f4[k4]->Fill(mult,fill,weight);
175 h_f4_vsMult[k4]->Fill(mult,fill,weight);
176 k4++;
177 }
178 }
179 }
180 k2++;
181 }
182 k1++;
183 }
