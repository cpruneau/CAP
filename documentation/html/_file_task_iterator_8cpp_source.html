204 if (reportStart(__FUNCTION__))

215
216 for (unsigned int k=0; k<fileList.size(); k++)

218 TString name = fileList[k];

222 TString pattern = includePatterns[kInclude];
223 if (!name.Contains(pattern)) { include = false; break;}
224 }

226 {

230 {
231 TString pattern = excludePatterns[kExclude];
232 if (name.Contains(pattern))
233 {
234 include = false;
235 break;
236 }
237 }
238 if (include)
239 {
240 if (name.Contains(".root"))
241 {
242 int dot = name.First('.');
243 int len = name.Length();
244 if (dot==len-5 )
245 name.Remove(dot,len-dot);
246 }
247 TString check = pathName+name;
248 //cout << " CHECK:::::: " << check << endl;
249 selectedFileNames.push_back(pathName+name);
250 }
251 }
252 if (reportEnd(__FUNCTION__))
