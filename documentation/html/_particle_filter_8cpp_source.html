200bool ParticleFilter::accept(const Particle & particle)
201{
202 unsigned int nConditions = getNConditions();
203 if (nConditions<1) return true;
204 double value;
205 bool accepting;
206 ParticleType & type = particle.getType();
207
208 for (unsigned int k = 0; k<nConditions; k++)
209 {
210 Condition & condition = *(conditions[k]);
211 unsigned int filterType = condition.filterType;
212 unsigned int filterSubType = condition.filterSubtype;
213 switch (filterType)
214 {
215 case 0: // live or not to be considered at all
216 switch (filterSubType)
217 {
218 case 0: accepting = !particle.isLive(); break; // decayed or removed particles only
219 case 1: accepting = particle.isLive(); break; // undecayed particles only
220 case 2: accepting = 1; break; // all
221 }
222 break;
223
224 case 1: // type selection/filtering
225 accepting = condition.accept(double(type.getCharge()));
226 break;
227
228 case 2: // PDG Code
229 accepting = condition.accept(double(type.getPdgCode()));
230 break;
231
232 //case 3: // User Code
233 //accepting = condition.accept(double(type.getUserPid()));
234 //break;
235
236 case 4: // Type selection
237 switch (filterSubType)
238 {
239 case 0: accepting = type.isPhoton(); break;
240
241 case 10: accepting = type.isLepton(); break;
242 case 11: accepting = type.isElectron() || type.isPositron(); break;
243 case 12: accepting = type.isElectron(); break;
244 case 13: accepting = type.isPositron(); break;
245 case 14: accepting = type.isMuon() || type.isAntiMuon(); break;
246 case 15: accepting = type.isMuon(); break;
247 case 16: accepting = type.isAntiMuon(); break;
248 case 17: accepting = type.isTau() || type.isAntiTau(); break;
249 case 18: accepting = type.isTau(); break;
250 case 19: accepting = type.isAntiTau(); break;
251
252 case 40: accepting = type.isStrange(); break;
253 case 41: accepting = type.isStrangePlus(); break;
254 case 42: accepting = type.isStrangeMinus(); break;
255
256 case 50: accepting = type.isCharm(); break;
257 case 51: accepting = type.isCharmPlus(); break;
258 case 52: accepting = type.isCharmMinus(); break;
259
260 case 60: accepting = type.isBottom(); break;
261 case 61: accepting = type.isBottomPlus(); break;
262 case 62: accepting = type.isBottomMinus(); break;
263
264 case 70: accepting = type.isTop(); break;
265
266 case 1100: accepting = type.isHadron(); break; // any hadron
267 case 1101: accepting = type.isHadron() && type.isCharged(); break; // any charged hadron
268 case 1102: accepting =
269 type.isPionP() ||
270 type.isPionM() ||
271 type.isKaonP() ||
272 type.isKaonM() ||
273 type.isProton() ||
274 type.isAntiProton(); // pi+, pi-, K+, K-, p, pBar
275 break; // any charge hadron
276 case 1103: accepting =
277 type.isPionP() ||
278 type.isKaonP() ||
279 type.isProton(); // pi+, K+, p
280 break;
281 case 1104: accepting =
282 type.isPionM() ||
283 type.isKaonM() ||
284 type.isAntiProton(); // pi-, K-, pBar
285 break;
286
287 case 1110: accepting = type.isPion(); break;
288 case 1111: accepting = type.isPionP(); break;
289 case 1112: accepting = type.isPion0(); break;
290 case 1113: accepting = type.isPionM(); break;
291
292 case 1120: accepting = type.isKaon(); break;
293 case 1121: accepting = type.isKaonP(); break;
294 case 1122: accepting = type.isKaonM(); break;
295 case 1123: accepting = type.isKaon0(); break;
296 case 1124: accepting = type.isKaon0Bar(); break;
297 case 1125: accepting = type.isKaon0S(); break;
298 case 1126: accepting = type.isKaon0L(); break;
299
300 case 1200: accepting = type.isBaryon(); break;
301 case 1201: accepting = type.isBaryonPlus(); break;
302 case 1202: accepting = type.isBaryonMinus(); break;
303
304 case 1210: accepting = type.isProton() || type.isAntiProton(); break;
305 case 1211: accepting = type.isProton(); break;
306 case 1212: accepting = type.isAntiProton(); break;
307
308 case 1220: accepting = type.isLambda() || type.isAntiLambda(); break;
309 case 1221: accepting = type.isLambda(); break;
310 case 1222: accepting = type.isAntiLambda(); break;
311
312 case 1230: accepting = type.isSigmaP(); break;
313 case 1231: accepting = type.isSigma0(); break;
314 case 1232: accepting = type.isSigmaM(); break;
315 case 1241: accepting = type.isAntiSigmaP(); break;
316 case 1242: accepting = type.isAntiSigma0(); break;
317 case 1243: accepting = type.isAntiSigmaM(); break;
318 case 1244: accepting = type.isXi0(); break;
319 case 1245: accepting = type.isXiM(); break;
320 case 1246: accepting = type.isAntiXi0(); break;
321 case 1247: accepting = type.isAntiXiM(); break;
322 case 1248: accepting = type.isOmegaM(); break;
323 case 1249: accepting = type.isAntiOmegaM(); break;
324 }
325 break;
326
327 case 5: // kinematic selection/filtering
328 const TLorentzVector & momentum = particle.getMomentum();
329 switch (filterSubType)
330 {
331 case 0: value = momentum.P(); break; // momentum
332 case 1: value = momentum.Pt(); break; // transverse momentum
333 case 2: value = momentum.E(); break; // energy
334 case 3: value = momentum.Px(); break; // p_x
335 case 4: value = momentum.Py(); break; // p_y
336 case 5: value = momentum.Pz(); break; // p_z
337 case 6: value = momentum.Phi(); break; // phi azimuth
338 case 7: value = momentum.Eta(); break; // pseudo rapidity
339 case 8: value = momentum.Rapidity(); break; // rapidity
340 }
341 accepting = conditions[k]->accept(value);
342 break;
343 }
344 if (!accepting) return false;
345 }
346 return true;
347}
