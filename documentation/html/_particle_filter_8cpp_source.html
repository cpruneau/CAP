197bool ParticleFilter::accept(const Particle & particle)
198{
199 unsigned int nConditions = getNConditions();
200 if (nConditions<1) return true;
201 double value;
202 bool accepting;
203 ParticleType & type = particle.getType();
204
205 for (unsigned int k = 0; k<nConditions; k++)
206 {
207 Condition & condition = *(conditions[k]);
208 unsigned int filterType = condition.filterType;
209 unsigned int filterSubType = condition.filterSubtype;
210 switch (filterType)
211 {
212 case 0: // live or not to be considered at all
213 switch (filterSubType)
214 {
215 case 0: accepting = !particle.isLive(); break; // decayed or removed particles only
216 case 1: accepting = particle.isLive(); break; // undecayed particles only
217 case 2: accepting = 1; break; // all
218 }
219 break;
220
221 case 1: // type selection/filtering
222 accepting = condition.accept(double(type.getCharge()));
223 break;
224
225 case 2: // PDG Code
226 accepting = condition.accept(double(type.getPdgCode()));
227 break;
228
229 //case 3: // User Code
230 //accepting = condition.accept(double(type.getUserPid()));
231 //break;
232
233 case 4: // Type selection
234 switch (filterSubType)
235 {
236 case 0: accepting = type.isPhoton(); break;
237
238 case 10: accepting = type.isLepton(); break;
239 case 11: accepting = type.isElectron() || type.isPositron(); break;
240 case 12: accepting = type.isElectron(); break;
241 case 13: accepting = type.isPositron(); break;
242 case 14: accepting = type.isMuon() || type.isAntiMuon(); break;
243 case 15: accepting = type.isMuon(); break;
244 case 16: accepting = type.isAntiMuon(); break;
245 case 17: accepting = type.isTau() || type.isAntiTau(); break;
246 case 18: accepting = type.isTau(); break;
247 case 19: accepting = type.isAntiTau(); break;
248
249 case 40: accepting = type.isStrange(); break;
250 case 41: accepting = type.isStrangePlus(); break;
251 case 42: accepting = type.isStrangeMinus(); break;
252
253 case 50: accepting = type.isCharm(); break;
254 case 51: accepting = type.isCharmPlus(); break;
255 case 52: accepting = type.isCharmMinus(); break;
256
257 case 60: accepting = type.isBottom(); break;
258 case 61: accepting = type.isBottomPlus(); break;
259 case 62: accepting = type.isBottomMinus(); break;
260
261 case 70: accepting = type.isTop(); break;
262
263 case 1100: accepting = type.isHadron(); break; // any hadron
264 case 1101: accepting = type.isHadron() && type.isCharged(); break; // any charged hadron
265 case 1102: accepting =
266 type.isPionP() ||
267 type.isPionM() ||
268 type.isKaonP() ||
269 type.isKaonM() ||
270 type.isProton() ||
271 type.isAntiProton(); // pi+, pi-, K+, K-, p, pBar
272 break; // any charge hadron
273 case 1103: accepting =
274 type.isPionP() ||
275 type.isKaonP() ||
276 type.isProton(); // pi+, K+, p
277 break;
278 case 1104: accepting =
279 type.isPionM() ||
280 type.isKaonM() ||
281 type.isAntiProton(); // pi-, K-, pBar
282 break;
283
284 case 1110: accepting = type.isPion(); break;
285 case 1111: accepting = type.isPionP(); break;
286 case 1112: accepting = type.isPion0(); break;
287 case 1113: accepting = type.isPionM(); break;
288
289 case 1120: accepting = type.isKaon(); break;
290 case 1121: accepting = type.isKaonP(); break;
291 case 1122: accepting = type.isKaonM(); break;
292 case 1123: accepting = type.isKaon0(); break;
293 case 1124: accepting = type.isKaon0Bar(); break;
294 case 1125: accepting = type.isKaon0S(); break;
295 case 1126: accepting = type.isKaon0L(); break;
296
297 case 1200: accepting = type.isBaryon(); break;
298 case 1201: accepting = type.isBaryonPlus(); break;
299 case 1202: accepting = type.isBaryonMinus(); break;
300
301 case 1210: accepting = type.isProton() || type.isAntiProton(); break;
302 case 1211: accepting = type.isProton(); break;
303 case 1212: accepting = type.isAntiProton(); break;
304
305 case 1220: accepting = type.isLambda() || type.isAntiLambda(); break;
306 case 1221: accepting = type.isLambda(); break;
307 case 1222: accepting = type.isAntiLambda(); break;
308
309 case 1230: accepting = type.isSigmaP(); break;
310 case 1231: accepting = type.isSigma0(); break;
311 case 1232: accepting = type.isSigmaM(); break;
312 case 1241: accepting = type.isAntiSigmaP(); break;
313 case 1242: accepting = type.isAntiSigma0(); break;
314 case 1243: accepting = type.isAntiSigmaM(); break;
315 case 1244: accepting = type.isXi0(); break;
316 case 1245: accepting = type.isXiM(); break;
317 case 1246: accepting = type.isAntiXi0(); break;
318 case 1247: accepting = type.isAntiXiM(); break;
319 case 1248: accepting = type.isOmegaM(); break;
320 case 1249: accepting = type.isAntiOmegaM(); break;
321 }
322 break;
323
324 case 5: // kinematic selection/filtering
325 const TLorentzVector & momentum = particle.getMomentum();
326 switch (filterSubType)
327 {
328 case 0: value = momentum.P(); break; // momentum
329 case 1: value = momentum.Pt(); break; // transverse momentum
330 case 2: value = momentum.E(); break; // energy
331 case 3: value = momentum.Px(); break; // p_x
332 case 4: value = momentum.Py(); break; // p_y
333 case 5: value = momentum.Pz(); break; // p_z
334 case 6: value = momentum.Phi(); break; // phi azimuth
335 case 7: value = momentum.Eta(); break; // pseudo rapidity
336 case 8: value = momentum.Rapidity(); break; // rapidity
337 }
338 accepting = conditions[k]->accept(value);
339 break;
340 }
341 if (!accepting) return false;
342 }
343 return true;
344}
