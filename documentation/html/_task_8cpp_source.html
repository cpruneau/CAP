218void Task::initialize()
219{
220 if (reportStart("Task",getName(),"initialize()"))
221 { }
222 if (isTaskOk() && configuration.getValueBool( "useParticles") )
223 {
224 particleFactory = Particle::getFactory();
225 if (!particleFactory)
226 {
227 if (reportFatal("Task",getName(),"initialize()"))
228 {
229 cout << " Particle factory required but not available. Must abort." << endl;
230 }
231 postTaskFatal();
232 return;
233 }
234 masterCollection = ParticleTypeCollection::getMasterParticleCollection();
235 if (!masterCollection)
236 {
237 if (reportFatal("Task",getName(),"initialize()"))
238 {
239 cout << " Particle master collection required but not available. Must abort." << endl;
240 }
241 postTaskFatal();
242 return;
243 }
244 }
245 if (!isTaskOk())
246 {
247 if (reportError("Task",getName(),"initialize()"))
248 {
249 cout << "Error posted - Abort immediately." << endl;
250 }
251 exit(1);
252 }
253 if (configuration.getValueBool( "useEventStream0") ) addEventStream(Event::getEventStream(0));
254 if (configuration.getValueBool( "useEventStream1") ) addEventStream(Event::getEventStream(1));
255 if (configuration.getValueBool( "useEventStream2") ) addEventStream(Event::getEventStream(2));
256 if (configuration.getValueBool( "useEventStream3") ) addEventStream(Event::getEventStream(3));
257 if (reportInfo("Task",getName(),"initialize()"))
258 {
259 cout << " #event added streams: " << getNEventStreams() << endl;
260 }
261 if (configuration.getValueBool( "createHistograms") )
262 {
263 if (reportDebug("Task",getName(),"initialize()"))
264 cout << "Create EventCountHistograms" << endl;
265 unsigned int nEventFilters = eventFilters.size();
266 if (nEventFilters>0) nFilteredEventsAccepted.assign(nEventFilters,0.0);
267 eventCountHistos = new EventCountHistos(getName(),configuration,getReportLevel());
268 eventCountHistos->createHistograms();
269 if (reportDebug("Task",getName(),"initialize()")) cout << "Done with EventCountHistograms" << endl;
270 createHistograms();
271 }
272 nEventProcessed = 0;
273 nEventAccepted = 0;
274 if (configuration.getValueBool( "loadHistograms") )
275 {
276 loadHistograms();
277 }
278
279 if (reportEnd("Task",getName(),"initialize()"))
280 { }
281}

334void Task::subsampleAnalysis()
335{
336 if (reportStart("Task",getName(),"subsampleAnalysis()"))
337 { }
338 bool doSubsampleAnalysis = configuration.getValueBool( "doSubsampleAnalysis");
339 TString inputPathName = configuration.getValueString( "histoOutputPath" ); // where files were saved
340 TString inputFileName = configuration.getValueString( "histoOutputFileName"); // base name of files in a job
341 TString outputPathName = inputPathName; // where files were saved
342 TString outputFileName = inputFileName;
343 int numberOfPartialSaves = getSubSampleIndex(); // number of partial saves
344
345
346 if (!doSubsampleAnalysis)
347 {
348 if (reportInfo("Task",getName(),"subsampleAnalysis(...)"))
349 {
350 cout << "No subsample analysis requested for task:" << getName() << endl;
351 }
352 return;
353 }
354 else
355 {
356 if (reportInfo("Task",getName(),"subsampleAnalysis(...)"))
357 {
358 cout << "Subsample analysis requested for task:" << getName() << endl;
359 cout << " inputPathName:" << inputPathName << endl;
360 cout << " inputFileName:" << inputFileName << endl;
361 cout << " numberOfPartialSaves:" << numberOfPartialSaves << endl;
362 cout << " outputPathName:" << outputPathName << endl;
363 cout << " outputFileName:" << outputFileName << endl;
364 }
365 }
366 Configuration statConfig("subSampleConfig");
367 statConfig.addParameter( "forceHistogramsRewrite", true);
368 statConfig.addParameter( "nInputFile", numberOfPartialSaves-1);
369 statConfig.addParameter( "histoInputPath", inputPathName );
370 statConfig.addParameter( "histoOutputPath", outputPathName );
371 statConfig.addParameter( "histoOutputFileName", outputFileName+"_Sum" );
372 for (int k=0; k<=numberOfPartialSaves; k++ )
373 {
374 TString key("InputFile"); key += k;
375 TString name = inputFileName;
376 name += "_";
377 name += k;
378 cout << " k: " << k << " " << name << endl;
379 statConfig.addParameter(key,name);
380 }
381 SubSampleStatCalculator * subSampleStatCalculator = new SubSampleStatCalculator("DefaultSubSampleStatCalculator",statConfig,getReportLevel());
382 subSampleStatCalculator->execute();
383 delete subSampleStatCalculator;
384 if (reportEnd("Task",getName(),"subsampleAnalysis(...)"))
385 { }

514void Task::scaleHistograms()
515{
516 if (reportStart("Task",getName(),"scaleHistograms()"))
517 { }
518 double scalingFactor;
519 unsigned int nEventFilters = eventFilters.size();
520 unsigned int nParticleFilters = particleFilters.size();
521 int index;
522 if (reportInfo("Task",getName(),"scaleHistograms()"))
523 {
524 cout << endl;
525 cout << " Accepted number of events: " << nEventAccepted << endl;
526 cout << " nEventFilters: " << nEventFilters << endl;
527 cout << " nParticleFilters: " << nParticleFilters << endl;
528 cout << " baseSingleHistograms.size(): " << baseSingleHistograms.size() << endl;
529 cout << " basePairHistograms.size(): " << basePairHistograms.size() << endl;
530 cout << " derivedSingleHistograms.size(): " << derivedSingleHistograms.size() << endl;
531 cout << " derivedPairHistograms.size(): " << derivedPairHistograms.size() << endl;
532 cout << " combinedHistograms.size(): " << combinedHistograms.size() << endl;
533 cout << "-------------- Accumulated statistics: ----------------------------------" << endl;
534 printEventCount();
535 }
536 for (unsigned int iEventFilter=0; iEventFilter<nEventFilters; iEventFilter++ )
537 {
538 if (nFilteredEventsAccepted[iEventFilter]>1)
539 {
540 scalingFactor = 1.0/double(nFilteredEventsAccepted[iEventFilter]);
541 if (reportInfo("Task",getName(),"scaleHistograms()"))
542 {
543 cout << endl;
544 cout << " iEventFilter: " << iEventFilter<< endl;
545 cout << " nFilteredEventsAccepted[iEventFilter]: " << nFilteredEventsAccepted[iEventFilter]<< endl;
546 cout << " scalingFactor: " << scalingFactor << endl;
547 }
548 if (histograms.size()>0)
549 {
550 if (reportInfo("Task",getName(),"scaleHistograms()")) cout << "Scaling group named " << histograms[iEventFilter]->getName() << endl;
551 histograms[iEventFilter]->scale(scalingFactor);
552 }
553 if (baseSingleHistograms.size()>0)
554 {
555 for (unsigned int iParticleFilter1=0; iParticleFilter1<nParticleFilters; iParticleFilter1++ )
556 {
557 index = iEventFilter*nParticleFilters + iParticleFilter1;
558 if (reportInfo("Task",getName(),"scaleHistograms()"))cout << "Scaling group index:" << index << " named " << baseSingleHistograms[index]->getName() << endl;
559 baseSingleHistograms[index]->scale(scalingFactor);
560 }
561 }
562 if (basePairHistograms.size()>0)
563 {
564 for (unsigned int iParticleFilter1=0; iParticleFilter1<nParticleFilters; iParticleFilter1++ )
565 {
566 for (unsigned int iParticleFilter2=0; iParticleFilter2<nParticleFilters; iParticleFilter2++ )
567 {
568 index = iEventFilter*nParticleFilters*nParticleFilters+iParticleFilter1*nParticleFilters+iParticleFilter2;
569 if (reportInfo("Task",getName(),"scaleHistograms()"))cout << "Scaling group index:" << index << " named " << baseSingleHistograms[index]->getName() << endl;
570 basePairHistograms[index]->scale(scalingFactor);
571 }
572 }
573 }
574 }
575 else
576 {
577 if (reportWarning("Task",getName(),"scaleHistograms()"))
578 {
579 cout << endl;
580 cout << " iEventFilter: " << iEventFilter<< endl;
581 cout << " nFilteredEventsAccepted[iEventFilter]: " << nFilteredEventsAccepted[iEventFilter]<< endl;
582 cout << " no scaling performed: " << endl;
583 }
584 }
585 }
586 if (reportEnd("Task",getName(),"scaleHistograms()"))
587 { }
588}

652void Task::saveHistograms()
653{
654 TString fct = "saveHistograms()";
655 if (reportStart("Task",getName(),fct))
656 { }
657 //TString outputPath;
658 TString outputFileName;
659 TFile * outputFile = nullptr;
660
661 if (configuration.getValueBool("fileFromParent") )
662 {
663 // this is used by the FileIterator to set the file name to be read
664 outputFileName = configuration.getValueString("histogramOuputFileName");
665 }
666 else
667 {
668 // this is used normally to identify the file name to be used to save or load from.
669 outputFileName = configuration.getValueString( "histoOutputPath");
670 outputFileName += configuration.getValueString( "histoOutputFileName");
671 }
672
673 if (configuration.getValueBool( "doSubsampleAnalysis") || configuration.getValueBool( "doPartialSave") )
674 {
675 outputFileName += "_";
676 outputFileName += subSampleIndex++;
677 }
678 if (configuration.getValueBool( "forceHistogramsRewrite") || configuration.getValueBool("fileFromParent"))
679 {
680 if (reportInfo("Task",getName(),fct)) cout << "Opening output (RECREATE) file " << outputFileName << endl;
681 outputFile = openRootFile("",outputFileName,"RECREATE");
682 if (!outputFile)
683 {
684 if (reportError("Task",getName(),"saveHistograms()")) cout << "Could not open (RECREATE) file:" << outputFileName << endl;
685 postTaskError();
686 return;
687 }
688 }
689 else
690 {
691 if (reportInfo("Task",getName(),fct)) cout << "Opening output (NEW) file " << outputFileName << endl;
692 outputFile = openRootFile("",outputFileName,"NEW");
