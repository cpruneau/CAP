184void Task::initialize()
185{
186 if (reportStart(__FUNCTION__))
187 ;
188 bool useParticles = configuration.getValueBool("useParticles");
189 bool createHistos = configuration.getValueBool("createHistograms");
190 bool loadHistos = configuration.getValueBool("loadHistograms");
191 bool useEventStream0 = configuration.getValueBool("useEventStream0");
192 bool useEventStream1 = configuration.getValueBool("useEventStream1");
193 bool useEventStream2 = configuration.getValueBool("useEventStream2");
194 bool useEventStream3 = configuration.getValueBool("useEventStream3");
195 initializeTaskExecuted();
196 if (useParticles)
197 {
198 particleFactory = Particle::getFactory();
199 particleTypeCollection = ParticleTypeCollection::getMasterParticleCollection();
200 if (nParticleFilters>0 && nEventFilters>0)
201 {
202 initializeNEventsAccepted();
203 initializeNParticlesAccepted();
204 }
205 else
206 {
207 if (reportWarning(__FUNCTION__)) cout << "useParticles==true but nParticleFilters=0 or nEventFilters==0." << endl;
208 }
209 }
210 if (useEventStream0) addEventStream(Event::getEventStream(0));
211 if (useEventStream1) addEventStream(Event::getEventStream(1));
212 if (useEventStream2) addEventStream(Event::getEventStream(2));
213 if (useEventStream3) addEventStream(Event::getEventStream(3));
214 if (reportDebug(__FUNCTION__)) cout << " #event added streams: " << getNEventStreams() << endl;
215 if (createHistos && isTaskOk())
216 {
217 createEventCountHistograms();
218 createHistograms();
219 }
220 if (loadHistos && isTaskOk()) loadHistograms();
221 if (hasSubTasks() && isTaskOk()) initializeSubTasks();
222 if (reportEnd(__FUNCTION__))
223 ;
224}

470void Task::scaleHistograms()
471{
472 if (reportStart(__FUNCTION__))
473 ;
474 double scalingFactor;
475 int index;
476 for (int iEventFilter=0; iEventFilter<nEventFilters; iEventFilter++ )
477 {
478 long nEvents = nEventsAcceptedReset[iEventFilter];
479 if (nEvents>1)
480 {
481 scalingFactor = 1.0/double(nEvents);
482 if (reportInfo(__FUNCTION__))
483 {
484 cout << endl;
485 cout << " iEventFilter: " << iEventFilter<< endl;
486 cout << " nEventsAccepted[iEventFilter]: " << nEvents<< endl;
487 cout << " scalingFactor: " << scalingFactor << endl;
488 }
489 if (histograms.size()>0)
490 {
491 if (reportInfo(__FUNCTION__)) cout << "Scaling group named " << histograms[iEventFilter]->getName() << endl;
492 histograms[iEventFilter]->scale(scalingFactor);
493 }
494 if (baseSingleHistograms.size()>0)
495 {
496 for (int iParticleFilter1=0; iParticleFilter1<nParticleFilters; iParticleFilter1++ )
497 {
498 index = iEventFilter*nParticleFilters + iParticleFilter1;
499 if (reportInfo(__FUNCTION__))cout << "Scaling group index:" << index << " named " << baseSingleHistograms[index]->getName() << endl;
500 baseSingleHistograms[index]->scale(scalingFactor);
501 }
502 }
503 if (basePairHistograms.size()>0)
504 {
505 for (int iParticleFilter1=0; iParticleFilter1<nParticleFilters; iParticleFilter1++ )
506 {
507 for (int iParticleFilter2=0; iParticleFilter2<nParticleFilters; iParticleFilter2++ )
508 {
509 index = iEventFilter*nParticleFilters*nParticleFilters+iParticleFilter1*nParticleFilters+iParticleFilter2;
510 if (reportInfo(__FUNCTION__))cout << "Scaling group index:" << index << " named " << baseSingleHistograms[index]->getName() << endl;
511 basePairHistograms[index]->scale(scalingFactor);
512 }
513 }
514 }
515 }
516 else
517 {
518 if (reportWarning(__FUNCTION__))
519 {
520 cout << endl;
521 cout << " iEventFilter: " << iEventFilter<< endl;
522 cout << " nEventsAcceptedReset[iEventFilter]: " << nEventsAcceptedReset[iEventFilter]<< endl;
523 cout << " no scaling performed: " << endl;
524 }
525 }
526 }
527 if (reportEnd(__FUNCTION__))
528 ;
529}

588void Task::saveHistograms()
589{
590 if (reportStart(__FUNCTION__))
591 ;
592 bool useParticles = configuration.getValueBool("useParticles");
593 bool doSubsampleAnalysis = configuration.getValueBool("doSubsampleAnalysis");
594 bool doPartialSave = configuration.getValueBool("doPartialSaves");
595 bool forceHistogramsRewrite = configuration.getValueBool("forceHistogramsRewrite");
596 TString histoOutputPath = configuration.getValueString("histoOutputPath");
597 TString histoOutputFileName = configuration.getValueString("histoOutputFileName");
598 TString histoOutputDataName = configuration.getValueString("histoOutputDataName");
599 TString histoOutputAnalyzerName = configuration.getValueString("histoOutputAnalyzerName");
600
601// histoOutputFileName = removeRootExtension(histoOutputFileName);
602
603 histoOutputFileName = histoOutputDataName;
604 histoOutputFileName += "_";
605 histoOutputFileName += histoOutputAnalyzerName;
606 if (doSubsampleAnalysis || doPartialSave )
607 {
608 histoOutputFileName += "_";
609 histoOutputFileName += subSampleIndex++;
610 }
611 if (reportInfo(__FUNCTION__))
612 {
613 cout << endl;
614 cout << " nTaskExecutedReset : " << nTaskExecutedReset << endl;
615 cout << " nTaskExecuted : " << nTaskExecuted << endl;
616 cout << " histoOutputPath : " << histoOutputPath << endl;
617 cout << " histoOutputFileName : " << histoOutputFileName << endl;
618 }
619 TFile * outputFile;
620 if (forceHistogramsRewrite)
621 outputFile = openRootFile(histoOutputPath,histoOutputFileName,"RECREATE");
622 else
623 outputFile = openRootFile(histoOutputPath,histoOutputFileName,"NEW");
624 if (!outputFile) return;
625 if (useParticles) saveEventCountHistograms(outputFile);
626 if (useParticles) writeNEventsAccepted(outputFile);
627 if (useParticles) writeNEexecutedTask(outputFile);
628 saveHistograms(outputFile);
629 outputFile->Close();
630 if (reportEnd(__FUNCTION__))
631 ;
632}

692long Task::readParameter(TFile * inputFile, const TString & parameterName)
693{
694 if (reportStart(__FUNCTION__))
695 ;
696 TParameter<Long64_t> *par = (TParameter<Long64_t> *) inputFile->Get(parameterName);
697 if (!par)
698 {
699 if (reportError(__FUNCTION__)) cout << "Parameter not found." << endl;
700 postTaskError();
701 return 1.0;
702 }
703 double value = par->GetVal();
704 delete par;
705 if (reportDebug(__FUNCTION__)) cout << "Parameter named " << parameterName << " has value : " << value << endl;
706 return value;

709TFile * Task::openRootFile(const TString & inputPath, const TString & fileName, const TString & ioOption)
710{
711 if (reportStart(__FUNCTION__))
712 ;
713 TString inputFileName = inputPath;
714
715 // make sure that if an inputPath is given, it ends with '/'
716 int slash = inputFileName.First('/');
717 int len = inputFileName.Length();
