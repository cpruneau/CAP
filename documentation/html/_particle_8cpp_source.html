199void Particle::boostRapidity(double boost)
200{
201 double rapidity = momentum.Rapidity();
202 rapidity += boost;
203 double px = momentum.Px();
204 double py = momentum.Py();
205 double pz = momentum.Pz();
206 double e = momentum.E();
207 double mt = sqrt(e*e - pz*pz);
208 pz = mt * sinh(rapidity);
209 e = mt * cosh(rapidity);
210 momentum.SetPxPyPzE (px,py,pz,e);
211 unsigned int nChildren = children.size();
212 for (unsigned int iChildren=0; iChildren<nChildren; iChildren++)
213 {
214 children[iChildren]->boostRapidity(boost);
215 }
216}

219double Particle::distanceXYZSq(Particle * otherParticle) const
220{
221 double dx = position.X() - otherParticle->position.X();
222 double dy = position.Y() - otherParticle->position.Y();
223 double dz = position.Z() - otherParticle->position.Z();
224 return dx*dx + dy*dy + dz*dz;
225}

299// Particle Interaction n->X
300void Particle::setParents(const vector<Particle*> & newParents)
301{
302 parents.clear();
303 Particle * parent = newParents[0];
304 momentum = parent->getMomentum();
305 position = parent->getPosition();
306 unsigned int nParents = newParents.size();
307 for (unsigned int iParent=1;iParent<nParents;iParent++)
308 {
309 parent = newParents[iParent];
310 momentum += parent->getMomentum();
311 position += parent->getPosition();
312 parents.push_back(parent);
313 }
314 position *= 1.0/double(nParents);
315}
