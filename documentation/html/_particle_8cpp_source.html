206void Particle::boostRapidity(double boost)
207{
208 double rapidity = momentum.Rapidity();
209 rapidity += boost;
210 double px = momentum.Px();
211 double py = momentum.Py();
212 double pz = momentum.Pz();
213 double e = momentum.E();
214 double mt = sqrt(e*e - pz*pz);
215 pz = mt * sinh(rapidity);
216 e = mt * cosh(rapidity);
217 momentum.SetPxPyPzE (px,py,pz,e);
218 unsigned int nChildren = children.size();
219 for (unsigned int iChildren=0; iChildren<nChildren; iChildren++)
220 {
221 children[iChildren]->boostRapidity(boost);
222 }
223}

226double Particle::distanceXYZSq(Particle * otherParticle) const
227{
228 double dx = position.X() - otherParticle->position.X();
229 double dy = position.Y() - otherParticle->position.Y();
230 double dz = position.Z() - otherParticle->position.Z();
231 return dx*dx + dy*dy + dz*dz;
232}

306// Particle Interaction n->X
307void Particle::setParents(const vector<Particle*> & newParents)
308{
309 parents.clear();
310 Particle * parent = newParents[0];
311 momentum = parent->getMomentum();
312 position = parent->getPosition();
313 unsigned int nParents = newParents.size();
314 for (unsigned int iParent=1;iParent<nParents;iParent++)
315 {
316 parent = newParents[iParent];
317 momentum += parent->getMomentum();
318 position += parent->getPosition();
319 parents.push_back(parent);
320 }
321 position *= 1.0/double(nParents);
322}
