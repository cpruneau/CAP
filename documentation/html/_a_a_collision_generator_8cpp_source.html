53void AACollisionGenerator::execute()
54{
55
56 if (reportStart(__FUNCTION__))
57 ;
58 incrementTaskExecuted();
59 unsigned int nParticles = eventStreams[0]->getNParticles()
60 for (unsigned int iParticle = 0; iParticle < nParticles; iParticle++)
61 {
62 Particle * particle = event->getParticleAt(iParticle)
63 if (particle->isInteraction() && particle->isLive() )
64 {
65 nnCollisionGenerator->generate(particle);
66 particle->setLive(false); // the interaction has been generated
67 }
68 }
69}
