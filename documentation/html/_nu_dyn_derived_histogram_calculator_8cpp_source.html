122void NuDynDerivedHistogramCalculator::execute()
123{
124
125 if (reportStart(__FUNCTION__))
126 ;
127 incrementTaskExecuted();
128 unsigned int nEventFilters = eventFilters.size();
129 unsigned int nParticleFilters = particleFilters.size();
130 if (reportInfo(__FUNCTION__))
131 {
132 cout << endl;
133 cout << "Computing derived histograms for: " << endl;
134 cout << " nEventFilters: " << nEventFilters << endl;
135 cout << " nParticleFilters: " << nParticleFilters << endl;
136 }
137 NuDynHistos * baseHistos;
138 NuDynDerivedHistos * derivedHistos;
139 unsigned index;
140
143 for (unsigned int iEventFilter=0; iEventFilter<nEventFilters; iEventFilter++ )
144 {
145 for (unsigned int iParticleFilter=0; iParticleFilter<nParticleFilters; iParticleFilter++ )
146 {
147 index = iEventFilter*nParticleFilters + iParticleFilter;
148 baseHistos = (NuDynHistos *) baseSingleHistograms[index];
149 derivedHistos = (NuDynDerivedHistos *) derivedSingleHistograms[index];
150 derivedHistos->calculateDerivedHistograms(baseHistos);
151 }
152 }
153 if (reportEnd(__FUNCTION__))
154 { }
