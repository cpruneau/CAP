181void MeasurementPerformanceSimulator::execute()
182{
183
184 if (reportStart(__FUNCTION__))
185 ;
186 incrementTaskExecuted();
187 Event & genEvent = *eventStreams[0];
188 Event & recoEvent = *eventStreams[1];
189 unsigned int nParticles = genEvent.getNParticles();
190 unsigned int firstPartFilter;
191 unsigned int lastPartFilter;
192 TLorentzVector recoMomentum;
193 for (unsigned int iEventFilter=0; iEventFilter<nEventFilters; iEventFilter++ )
194 {
195 if (!eventFilters[iEventFilter]->accept(genEvent)) continue;
196 incrementNEventsAccepted(iEventFilter);
197 if (nParticles<1) return;
198
199 if (allEventsUseSameFilters)
200 {
201 firstPartFilter = 0;
202 lastPartFilter = nParticleFilters;
203 }
204 else
205 {
206 firstPartFilter = iEventFilter * nParticleFilters;
207 lastPartFilter = firstPartFilter + nParticleFilters;
208 }
209 for (unsigned int iParticleFilter=firstPartFilter; iParticleFilter<lastPartFilter; iParticleFilter++ )
210 {
211 ParticleFilter * particleFilter = particleFilters[iParticleFilter];
212 ParticlePerformanceSimulator * simulator = (ParticlePerformanceSimulator *) baseSingleHistograms[iParticleFilter];
213 for (unsigned int iParticle=0; iParticle<nParticles; iParticle++)
214 {
215 Particle * genParticle = genEvent.getParticleAt(iParticle);
216 if (particleFilter->accept(*genParticle))
217 {
218 TLorentzVector & genMomentum = genParticle->getMomentum();
219 if (!simulator->accept(genMomentum)) continue;
220 ParticleType * type = genParticle->getTypePtr();
221 TLorentzVector & genPosition = genParticle->getPosition();
222 simulator->smearMomentum(genMomentum,recoMomentum);
223 Particle * recoParticle = particleFactory->getNextObject();
224 // we dont smear the position for now...
225 recoParticle->set(type,recoMomentum,genPosition,true);
226 recoParticle->setTruth(genParticle);
227 recoEvent.add(recoParticle);
228 } // particle accepted by filter
229 } //particle loop
230 } // particle filter loop
231 } // event filter loop
232}
