51// cout << "Nucleus::defineAs(unsigned int z, unsigned int a)" << endl;

82// Keeps the nucleus defined but resets all the positions.
83void Nucleus::reset()
84{
85 momentum.SetPxPyPzE (0.0,0.0,0.0,0.0);
86 position.SetXYZT (0.0,0.0,0.0,0.0);
87 parents.clear();
88
89 for (unsigned int iNucleon=0; iNucleon<children.size(); iNucleon++)
90 {
91 children[iNucleon]->reset();
92 children[iNucleon]->setWounded(false);;
93 }
94 live = true;
95}
