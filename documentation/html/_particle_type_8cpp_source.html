48ParticleType::ParticleType(int pdgCode_in, TString & name_in, TString & title_in,
49 double mass_in,
50 double width_in, int gSpin_in, int baryon_in, int strange_in,
51 int charm_in, int bottom_in, int gIsospin_in, int charge_in)
52:
53name(name_in),
54title(title_in),
55privateCode(-1), // Code used in this package only (convenience)
56pdgCode(pdgCode_in),
57mass(mass_in),
58width(width_in),
59gSpin(gSpin_in),
60gIsospin(gIsospin_in),
61baryon(baryon_in),
62strange(strange_in),
63charm(charm_in),
64bottom(bottom_in),
65leptonElectron(0),
66leptonMuon(0),
67leptonTau(0),
68charge(charge_in),
69stable(true),
70weakStable(true),
71statistics(0),
72decayModes(),
73decayRndmSelector(nullptr)
74{
75 // for now, we do not differentiate between stable and weakStable
76 if (width<1.0E-4)
77 {
78 stable = true;
79 weakStable = true;
80 }
81 if (abs(baryon)==1 ||
82 abs(leptonElectron)==1 ||
83 abs(leptonMuon)==1 ||
84 abs(leptonTau)==1)
85 {
86 statistics = 1; // Fermi-Dirac
87 }
88 else
89 {
90 statistics = -1; // Bose-Einstein
91 }
92}

155void ParticleType::addDecayMode(double branchingRatio,
156 std::vector<int> children)
157{
158 ParticleDecayMode decayMode;
159 decayMode.setBranchingRatio(branchingRatio);
160 for (int k=0; k<int(children.size()); k++)
161 {
162 decayMode.addChild(children[k]);
163 }
164 decayModes.push_back(decayMode);
165 if (decayModes.size() > 1)
166 {
167 stable = false;
168 weakStable = false;
169 }
170 else
171 {
172 if (pdgCode != decayModes[0].getChildPdgCode(0))
173 {
174 stable = false;
175 weakStable = false;
176 }
177 }
178}

210void ParticleType::setupDecayGenerator()
211{
212 int nDecayModes = decayModes.size();
213 vector<double> decayBranchingRatios;
214 for (int k=0; k<nDecayModes; k++)
215 {
216 decayBranchingRatios.push_back(decayModes[k].getBranchingRatio());
217 }
218 decayRndmSelector = new SelectionGenerator(decayBranchingRatios);
219}
220
221ParticleDecayMode & ParticleType::generateDecayMode()
222{
223 if (!decayRndmSelector)
224 {
225 exit(0);
226 }
227 int index = decayRndmSelector->generate();
228 if (index<0)
229 {
230 exit(0);
231 }
232 else if (index>= int(decayModes.size()))
233 {
234 index = decayModes.size()-1;
235 }
236 return decayModes[index];
237}

239ostream & ParticleType::printProperties(ostream & os)
240{
241 //scientific
242 os << fixed << setw(10) << setprecision(8) << pdgCode; // Monte-Carlo number according PDG
243 os << setw(20) << name; // ParticleType name
244 os << setw(20) << title; // ParticleType title
245 os << scientific << setw(20) << setprecision(4) << mass; // ParticleType mass (GeV)
246 os << scientific << setw(20) << setprecision(4) << width; // decay width
247 os << fixed << setw(7) << setprecision(3) << gSpin; // spin degeneracy
248 os << fixed << setw(5) << setprecision(3) << baryon; // baryon number
249 os << fixed << setw(5) << setprecision(3) << strange; // strangeness
250 os << fixed << setw(5) << setprecision(3) << charm; // charmness
251 os << fixed << setw(5) << setprecision(3) << bottom; // bottomness
252 os << fixed << setw(5) << setprecision(3) << gIsospin; // isospin degeneracy
253 os << fixed << setw(5) << setprecision(3) << charge; // charge
254 os << fixed << setw(5) << setprecision(6) << decayModes.size(); // amount of decays listed for this resonance
255 os << fixed << setw(5) << setprecision(3) << stable; // defines whether this ParticleType is stable
256 os << fixed << setw(5) << setprecision(3) << weakStable; // defines whether this ParticleType is stable
257 os << fixed << setw(5) << setprecision(3) << statistics;
258 os << endl;
259 for (int k=0;k<int(decayModes.size());k++)
260 {
261 if (pdgCode != decayModes[0].getChildPdgCode(0))
262 {
263 cout << " " << k << " ";
264 decayModes[k].printProperties(os);
265 }
266 }
267 return os;
268}

270ostream & ParticleType::printDecayProperties(ostream & os)
271{
272 TString latex;
273 latex = "$";
274 latex += title;
275 latex += "$";
276 os << " PDG:" << fixed << setw(10) << setprecision(8) << pdgCode; // Monte-Carlo number according PDG
277 os << setw(20) << latex;
278 os << " $m=$ " << scientific << setw(20) << setprecision(4) << mass << " GeV/$c^2$";
279 os << " $#Gamma=$ " << scientific << setw(20) << setprecision(4) << width << " GeV/$c^2$ \\\\"; // decay width
280 os << endl;
281 for (int k=0;k<int(decayModes.size());k++)
282 {
283 if (pdgCode != decayModes[0].getChildPdgCode(0))
284 {
285 os << " Br(xx)= " << scientific << setw(20) << setprecision(4) << (100.0*decayModes[k].getBranchingRatio()) << " ";
286 os << " $"<< title << "#rightarrow ";
287 int nChildren = decayModes[k].getNChildren();
288 for (int iChild=0; iChild<nChildren; iChild++)
289 {
290 ParticleType & type = decayModes[k].getChildType(iChild);
291 os << type.getTitle();
292 if (iChild < nChildren-1) os << "+";
293 }
294 os << "$ \\\\"<< endl;
295 }
296 }
297 return os;
298}
