87DataGraph::DataGraph(const TString _graphName,
88 const TString _xTitle,
89 const TString _yTitle,
90 const TString _legendText,
91 double _xMin, double _xMax,
92 double _yMin, double _yMax,
93 TH1 * histo, int _opt)
94: TH1D(_graphName,_graphName, 100, _xMin, _xMax),
95legendText(_legendText),
96nPoints(0),
97x(nullptr),
98exLow(nullptr),
99exHigh(nullptr),
100y(nullptr),
101eyLow(nullptr),
102eyHigh(nullptr),
103eeyLow(nullptr),
104eeyHigh(nullptr),
105option(_opt)
106{
107 Fill(-100.0);
108 SetName(_graphName);
109 SetXTitle(_xTitle);
110 SetYTitle(_yTitle);
111 SetMinimum(_yMin);
112 SetMaximum(_yMax);
113
114 nPoints = histo->GetNbinsX();
115 x = new double[nPoints];
116 exLow = new double[nPoints];
117 exHigh = new double[nPoints];
118 y = new double[nPoints];
119 eyLow = new double[nPoints];
120 eyHigh = new double[nPoints];
121 double w, ey;
122 for (unsigned int k=0; k<nPoints; k++)
123 {
124 x[k] = histo->GetXaxis()->GetBinCenter(k+1);
125 w = 0.5 * histo->GetXaxis()->GetBinWidth(k+1);
126 exLow[k] = w;
127 exHigh[k] = w;
128 y[k] = histo->GetBinContent(k+1);
129 ey = histo->GetBinError(k+1);
130 eyLow[k] = ey;
131 eyHigh[k] = ey;
132 }
133
134 TGraph * g;
135 g = new TGraphAsymmErrors (nPoints, x, y, exLow, exHigh, eyLow, eyHigh);
136 graphs.push_back(g);
137 visible = true;
138}

262void DataGraph::printToFile(ostream & output)
263{
264 output << "========================================================================================" << endl;
265 output << " Graph Name:" << GetName() << endl;
266 output << " Graph Title:" << GetName() << endl;
267 output << " nPoints:" << nPoints << endl;
268 for (int k=0; k<nPoints; k++)
269 {
270
271 output << "x:" << x[k];
272 if (exLow && !exHigh) output << "+-" << exLow[k];
273 if (exLow && exHigh) output << "-" << exLow[k];
274 if (exLow && exHigh) output << "+" << exHigh[k];
275 if (eyLow && !eyHigh) output << " (stat) +- " << eyLow[k];
276 if (eyLow && eyHigh) output << " (stat) - " << eyLow[k];
277 if (eeyLow && eyHigh) output << " (stat) + " << eyHigh[k];
278 if (eeyLow && !eeyHigh) output << " (sys) +- " << eeyLow[k];
279 if (eeyLow && eeyHigh) output << " (sys) - " << eeyLow[k];
280 if (eeyLow && eeyHigh) output << " (sys) + " << eeyHigh[k];
281 }
282 output << "========================================================================================" << endl;
283}

291 double yMin, double yMax,
292 TFile * inputFile, TString folderName, TString h1Name, TString h1e1Name, TString h1e2Name, TString g1Name )
293{
294 if (!inputFile)
295 {
296 cout << "<E> DataGraph::loadGraph(...) inputFile is a nullptr" << endl;
297 return nullptr;
298 }
299//else
300// {
301// cout << "<I> DataGraph::loadGraph() Loading from file: " << inputFile->GetName() << endl;
302//
303// }
304
305 TString name;
306 name = folderName; name += "/"; name += h1Name;
307 TH1 * h1 = (TH1*)inputFile->Get(name);
308 if (!h1)
309 {
310 cout << "<E> DataGraph::loadGraph() Could not load histogram: " << name << " from file:" << inputFile->GetName() << endl;
311 return nullptr;
312 }
313 name = folderName; name += "/"; name += h1e1Name;
314 TH1 * h1e1 = (TH1*)inputFile->Get(name);
315 if (!h1e1)
316 {
317 cout << "<E> DataGraph::loadGraph() Could not load histogram called: " << name << " from file:" << inputFile->GetName() << endl;
318 return nullptr;
319 }
320 name = folderName; name += "/"; name += h1e2Name;
321 TH1 * h1e2 = (TH1*)inputFile->Get(name);
322 if (!h1e2)
323 {
324 cout << "<E> DataGraph::loadGraph() Could not load histogram called: " << name << " from file:" << inputFile->GetName() << endl;
325 return nullptr;
326 }
327 name = folderName; name += "/"; name += g1Name;
328 TGraphAsymmErrors* g = (TGraphAsymmErrors*)inputFile->Get(name);
329 if (!g)
330 {
331 cout << "<E> DataGraph::loadGraph() Could not load graph called: " << name << " from file:" << inputFile->GetName() << endl;
332 return nullptr;
333 }
334 //cout << "<I> DataGraph::loadGraph() Data loaded..." << endl;
335
336 int n = h1->GetNbinsX();
337 double * x = new double[n];
338 double * exLow = new double[n];
339 double * exHigh = new double[n];
340 double * y = new double[n];
341 double * eyLow = new double[n];
342 double * eyHigh = new double[n];
343 double * eeyLow = new double[n];
344 double * eeyHigh = new double[n];
345 for (int k=0; k<n; k++)
346 {
347 x[k] = g->GetPointX(k);
348 exLow[k] = g->GetErrorXlow(k);
349 exHigh[k] = g->GetErrorXhigh(k);
350 y[k] = g->GetPointY(k);
351 eyLow[k] = g->GetErrorYlow(k);
352 eyHigh[k] = g->GetErrorYhigh(k);
353 eeyLow[k] = h1e1->GetBinContent(k);
354 eeyHigh[k] = h1e2->GetBinContent(k);
355 }
356 DataGraph * dg = new DataGraph(graphName, xTitle, yTitle, legendText, xMin, xMax, yMin, yMax, n,
357 x, exLow, exHigh,y,eyLow,eyHigh,eeyLow,eeyHigh,0);
358 return dg;
359}

361DataGraph * DataGraph::loadGraph(const TString graphName,
362 const TString xTitle,
363 const TString yTitle,
364 const TString legendText,
365 double xMin, double xMax,
366 double yMin, double yMax,
367 TFile * inputFile, TString hName, double scale, int rebin)
368{
369 if (!inputFile)
370 {
371 cout << "<E> DataGraph::loadGraph(...) inputFile is a nullptr" << endl;
372 return nullptr;
373 }
374//else
375// {
376// cout << "<I> DataGraph::loadGraph() Loading from file: " << inputFile->GetName() << endl;
377//
378// }
379
380 TString name;
381 name = hName;
382 TH1 * histo = (TH1*)inputFile->Get(name);
383 if (!histo)
384 {
385 cout << "<E> DataGraph::loadGraph() Could not load histogram: " << name << " from file:" << inputFile->GetName() << endl;
386 return nullptr;
387 }
388
389 if (rebin>1)
390 {
391 histo->Rebin(rebin);
392 histo->Scale(1.0/double(rebin));
393 }
394
395
396 int n = histo->GetNbinsX();
397 double * x = new double[n];
398 double * exLow = new double[n];
399 double * exHigh = new double[n];
400 double * y = new double[n];
401 double * eyLow = new double[n];
402 double * eyHigh = new double[n];
403 double w, ey;
404 for (int k=0; k<n; k++)
405 {
406 x[k] = histo->GetXaxis()->GetBinCenter(k+1);
407 w = 0.5 * histo->GetXaxis()->GetBinWidth(k+1);
408 exLow[k] = w;
409 exHigh[k] = w;
410 y[k] = scale * histo->GetBinContent(k+1);
411 ey = scale * histo->GetBinError(k+1);
412 eyLow[k] = ey;
413 eyHigh[k] = ey;
414 }
415 DataGraph * dg = new DataGraph(graphName, xTitle, yTitle, legendText, xMin, xMax, yMin, yMax, n,
416 x, exLow, exHigh,y,eyLow,eyHigh);
417 return dg;
418}
