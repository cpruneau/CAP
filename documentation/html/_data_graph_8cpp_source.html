92DataGraph::DataGraph(const TString _graphName,

95 const TString _legendText,

99: TH1D(_graphName,_graphName, 100, _xMin, _xMax),
100legendText(_legendText),
101nPoints(0),
102x(nullptr),
103exLow(nullptr),
104exHigh(nullptr),
105y(nullptr),
106eyLow(nullptr),
107eyHigh(nullptr),
108eeyLow(nullptr),
109eeyHigh(nullptr),
110option(_opt)
111{
112 Fill(-100.0);
113 SetName(_graphName);
114 SetXTitle(_xTitle);
115 SetYTitle(_yTitle);
116 SetMinimum(_yMin);
117 SetMaximum(_yMax);
118
119 nPoints = histo->GetNbinsX();
120 x = new double[nPoints];
121 exLow = new double[nPoints];
122 exHigh = new double[nPoints];
123 y = new double[nPoints];
124 eyLow = new double[nPoints];
125 eyHigh = new double[nPoints];
126 double w, ey;
127 for (unsigned int k=0; k<nPoints; k++)
128 {
129 x[k] = histo->GetXaxis()->GetBinCenter(k+1);
130 w = 0.5 * histo->GetXaxis()->GetBinWidth(k+1);
131 exLow[k] = w;
132 exHigh[k] = w;
133 y[k] = histo->GetBinContent(k+1);
134 ey = histo->GetBinError(k+1);
135 eyLow[k] = ey;
136 eyHigh[k] = ey;
137 }
138
139 TGraph * g;
140 g = new TGraphAsymmErrors (nPoints, x, y, exLow, exHigh, eyLow, eyHigh);
141 graphs.push_back(g);
142 visible = true;

267void DataGraph::printToFile(ostream & output)
268{
269 output << "========================================================================================" << endl;
270 output << " Graph Name:" << GetName() << endl;
271 output << " Graph Title:" << GetName() << endl;
272 output << " nPoints:" << nPoints << endl;
273 for (int k=0; k<nPoints; k++)
274 {
275
276 output << "x:" << x[k];
277 if (exLow && !exHigh) output << "+-" << exLow[k];
278 if (exLow && exHigh) output << "-" << exLow[k];
279 if (exLow && exHigh) output << "+" << exHigh[k];
280 if (eyLow && !eyHigh) output << " (stat) +- " << eyLow[k];
281 if (eyLow && eyHigh) output << " (stat) - " << eyLow[k];
282 if (eeyLow && eyHigh) output << " (stat) + " << eyHigh[k];
283 if (eeyLow && !eeyHigh) output << " (sys) +- " << eeyLow[k];
284 if (eeyLow && eeyHigh) output << " (sys) - " << eeyLow[k];
285 if (eeyLow && eeyHigh) output << " (sys) + " << eeyHigh[k];

366DataGraph * DataGraph::loadGraph(const TString graphName,
367 const TString xTitle,
368 const TString yTitle,
369 const TString legendText,
370 double xMin, double xMax,
371 double yMin, double yMax,
372 TFile * inputFile, TString hName, double scale, int rebin)
373{
374 if (!inputFile)
375 {
376 cout << "<E> DataGraph::loadGraph(...) inputFile is a nullptr" << endl;
377 return nullptr;
378 }
379//else
380// {
381// cout << "<I> DataGraph::loadGraph() Loading from file: " << inputFile->GetName() << endl;
382//
383// }
384
385 TString name;
386 name = hName;
387 TH1 * histo = (TH1*)inputFile->Get(name);
388 if (!histo)
389 {
390 cout << "<E> DataGraph::loadGraph() Could not load histogram: " << name << " from file:" << inputFile->GetName() << endl;
391 return nullptr;
392 }
393
394 if (rebin>1)
395 {
396 histo->Rebin(rebin);
397 histo->Scale(1.0/double(rebin));
398 }
399
400
401 int n = histo->GetNbinsX();
402 double * x = new double[n];
403 double * exLow = new double[n];
404 double * exHigh = new double[n];
405 double * y = new double[n];
406 double * eyLow = new double[n];
407 double * eyHigh = new double[n];
408 double w, ey;
409 for (int k=0; k<n; k++)
410 {
411 x[k] = histo->GetXaxis()->GetBinCenter(k+1);
412 w = 0.5 * histo->GetXaxis()->GetBinWidth(k+1);
413 exLow[k] = w;
414 exHigh[k] = w;
415 y[k] = scale * histo->GetBinContent(k+1);
416 ey = scale * histo->GetBinError(k+1);
417 eyLow[k] = ey;
418 eyHigh[k] = ey;
419 }
420 DataGraph * dg = new DataGraph(graphName, xTitle, yTitle, legendText, xMin, xMax, yMin, yMax, n,
421 x, exLow, exHigh,y,eyLow,eyHigh);
422 return dg;
