165void CollisionGeometryHistograms::fill(Event & event, double weight)
166{
167
168 double impactPar = event.getImpactParameter();
169 double nPart = event.getNParticipants();
170 double nBinary = event.getNBinaryCollisions();
171 double xSect = event.getCrossSection();
172
173// if (reportInfo("CollisionGeometryHistograms",getName(),"loadHistograms(TFile * inputFile)"))
174// {
175// cout<< " impactPar :" << impactPar << endl;
176// cout<< " nPart :" << nPart << endl;
177// cout<< " nBinary :" << nBinary << endl;
178// cout<< " xSect :" << xSect << endl;
179// }
180
181 // Impact parameter, cross-section, nParticipants, nBinaries, etc.
182 h_nAcceptedVsB ->Fill(impactPar, weight);
183 h_nPartVsB ->Fill(impactPar, nPart, weight);
184 h_nBinaryVsB ->Fill(impactPar, nBinary, weight);
185 h_nPartVsB_Prof ->Fill(impactPar, nPart, weight);
186 h_nPartSqVsB_Prof ->Fill(impactPar, nPart*nPart, weight);
187 h_nBinaryVsB_Prof ->Fill(impactPar, nBinary, weight);
188 h_nBinarySqVsB_Prof ->Fill(impactPar, nBinary*nBinary, weight);
189
190
191 h_nPart ->Fill(nPart, weight);
192 h_nBinaryVsNPart_Prof ->Fill(nPart, nBinary, weight);
193 h_nBinarySqVsNPart_Prof ->Fill(nPart, nBinary*nBinary, weight);
194 h_bVsNPart_Prof ->Fill(nPart, impactPar, weight);
195 h_bSqVsNPart_Prof ->Fill(nPart, impactPar*impactPar, weight);
196
197 h_nBinary ->Fill(nBinary, weight);
198 h_nPartVsNBinary_Prof ->Fill(nBinary, nPart, weight);
199 h_nPartSqVsNBinary_Prof ->Fill(nBinary, nPart*nPart, weight);
200 h_bVsNBinary_Prof ->Fill(nBinary, impactPar, weight);
201 h_bSqVsNBinary_Prof ->Fill(nBinary, impactPar*impactPar, weight);
202
203 h_crossSection ->Fill(xSect, weight);
204 h_nPartVsXsect_Prof ->Fill(xSect, nPart, weight);
205 h_nPartSqVsXsect_Prof ->Fill(xSect, nPart*nPart, weight);
206 h_nBinaryVsXsect_Prof ->Fill(xSect, nBinary, weight);
207 h_nBinarySqVsXsect_Prof ->Fill(xSect, nBinary*nBinary, weight);
208 h_bVsXsect_Prof ->Fill(xSect, impactPar, weight);
209 h_bSqVsXsect_Prof ->Fill(xSect, impactPar*impactPar, weight);
210
211 // Geometry
212
213 const vector<Particle*> interactions = event.getNucleonNucleonInteractions();
214 for (unsigned int k=0;k<interactions.size(); k++)
215 {
216 const TLorentzVector & position = interactions[k]->getPosition();
217 h_xyDistInteractions->Fill( position.X(), position.Y() );
218 }
219
220 Nucleus & nucleusA = event.getNucleusA();
221 Nucleus & nucleusB = event.getNucleusB();
222 vector<Particle*> nucleonsA = nucleusA.getChildren();
223 vector<Particle*> nucleonsB = nucleusB.getChildren();
224 for (unsigned int k=0;k<nucleonsA.size(); k++)
225 {
226 const TLorentzVector & position = nucleonsA[k]->getPosition();
227 h_xyDistNucleons->Fill( position.X(), position.Y() );
228 }
229 for (unsigned int k=0;k<nucleonsB.size(); k++)
230 {
231 const TLorentzVector & position = nucleonsB[k]->getPosition();
232 h_xyDistNucleons->Fill( position.X(), position.Y() );
233 }
234}

300void CollisionGeometryHistograms::calculateRms(TProfile * h1, TProfile * h1Sq, TH1* h1Rms, TH1* h1Omega, TH1* h1R2)
301{
302 if (reportStart("CollisionGeometryHistograms",getName(),"calculateRms(...)"))
303 ;
304 double v, ev, v2, ev2, rms, erms, omega, eomega, R2, eR2;
305 int nBins = h1->GetNbinsX();
306 for (int iBin=1; iBin<nBins; iBin++)
307 {
308 v = h1->GetBinContent(iBin);
309 ev = 0.0;
310 v2 = h1Sq->GetBinContent(iBin);
311 ev2 = 0.0;
312 rms = v2 - v*v;
313 if (rms>0) rms = sqrt(rms);
314 erms = 0.0;
315
316 if (v>0)
317 {
318 omega = rms*rms/v;
319 eomega = 0.0;
320 R2 = (v2-v)/v/v - 1.0;
321 eR2 = 0;
322 }
323 else
324 {
325 omega = 0.0;
326 eomega = 0.0;
327 R2 = 0.0;
328 eR2 = 0;
329 }
330
331 //cout << " iBin: " << iBin << " v:" << v << " v2:" << v2 << " rms:" << rms << " omega:" << omega << " R2:" << R2 << endl;
332
333 h1Rms->SetBinContent(iBin, rms);
334 h1Rms->SetBinError (iBin, erms);
335
336 h1Omega->SetBinContent(iBin, omega);
337 h1Omega->SetBinError(iBin, eomega);
338
339 h1R2->SetBinContent(iBin, R2);
340 h1R2->SetBinError(iBin, eR2);
341 }
342 if (reportEnd("CollisionGeometryHistograms",getName(),"calculateRms(...)"))
343 ;
