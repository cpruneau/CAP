217void CollisionGeometryHistograms::fill(Event & event, double weight)
218{
219 EventProperties & ep = *event.getEventProperties();
220 double impactPar = ep.impactParameter;
221 double nParticipants = ep.nParticipantsTotal;
222 double nBinary = ep.nBinaryTotal;
223 double xSect = ep.fractionalXSection;
224
225// if (reportInfo("CollisionGeometryHistograms",getName(),"loadHistograms(TFile * inputFile)"))
226// {
227// cout<< " impactPar :" << impactPar << endl;
228// cout<< " nParticipants :" << nParticipants << endl;
229// cout<< " nBinary :" << nBinary << endl;
230// cout<< " xSect :" << xSect << endl;
231// }
232
233 // Impact parameter, cross-section, nParticipants, nBinaries, etc.
234 h_nAcceptedVsB ->Fill(impactPar, weight);
235 h_nPartVsB ->Fill(impactPar, nParticipants, weight);
236 h_nBinaryVsB ->Fill(impactPar, nBinary, weight);
237 h_nPartVsB_Prof ->Fill(impactPar, nParticipants, weight);
238 h_nPartSqVsB_Prof ->Fill(impactPar, nParticipants*nParticipants, weight);
239 h_nBinaryVsB_Prof ->Fill(impactPar, nBinary, weight);
240 h_nBinarySqVsB_Prof ->Fill(impactPar, nBinary*nBinary, weight);
241
242
243 h_nPart ->Fill(nParticipants, weight);
244 h_nBinaryVsNPart_Prof ->Fill(nParticipants, nBinary, weight);
245 h_nBinarySqVsNPart_Prof ->Fill(nParticipants, nBinary*nBinary, weight);
246 h_bVsNPart_Prof ->Fill(nParticipants, impactPar, weight);
247 h_bSqVsNPart_Prof ->Fill(nParticipants, impactPar*impactPar, weight);
248
249 h_nBinary ->Fill(nBinary, weight);
250 h_nPartVsNBinary_Prof ->Fill(nBinary, nParticipants, weight);
251 h_nPartSqVsNBinary_Prof ->Fill(nBinary, nParticipants*nParticipants, weight);
252 h_bVsNBinary_Prof ->Fill(nBinary, impactPar, weight);
253 h_bSqVsNBinary_Prof ->Fill(nBinary, impactPar*impactPar, weight);
254
255 h_crossSection ->Fill(xSect, weight);
256 h_nPartVsXsect_Prof ->Fill(xSect, nParticipants, weight);
257 h_nPartSqVsXsect_Prof ->Fill(xSect, nParticipants*nParticipants, weight);
258 h_nBinaryVsXsect_Prof ->Fill(xSect, nBinary, weight);
259 h_nBinarySqVsXsect_Prof ->Fill(xSect, nBinary*nBinary, weight);
260 h_bVsXsect_Prof ->Fill(xSect, impactPar, weight);
261 h_bSqVsXsect_Prof ->Fill(xSect, impactPar*impactPar, weight);
262
263 // Geometry
264
265 const vector<Particle*> interactions = event.getNucleonNucleonInteractions();
266 for (unsigned int k=0;k<interactions.size(); k++)
267 {
268 const TLorentzVector & position = interactions[k]->getPosition();
269 h_xyDistInteractions->Fill( position.X(), position.Y() );
270 }
271
272 Nucleus & nucleusA = event.getNucleusA();
273 Nucleus & nucleusB = event.getNucleusB();
274 vector<Particle*> nucleonsA = nucleusA.getChildren();
275 vector<Particle*> nucleonsB = nucleusB.getChildren();
276 for (unsigned int k=0;k<nucleonsA.size(); k++)
277 {
278 const TLorentzVector & position = nucleonsA[k]->getPosition();
279 h_xyDistNucleons->Fill( position.X(), position.Y() );
280 }
281 for (unsigned int k=0;k<nucleonsB.size(); k++)
282 {
283 const TLorentzVector & position = nucleonsB[k]->getPosition();
284 h_xyDistNucleons->Fill( position.X(), position.Y() );
285 }
286}

342void CollisionGeometryHistograms::calculateRms(TProfile * h1, TProfile * h1Sq, TH1* h1Rms, TH1* h1Omega, TH1* h1R2)
343{
344 if (reportStart("CollisionGeometryHistograms",getName(),"calculateRms(...)"))
345 ;
346 double v, vSq, rms, erms, omega, eomega, R2, eR2;
347 int nBins = h1->GetNbinsX();
348 for (int iBin=1; iBin<nBins; iBin++)
349 {
350 v = h1->GetBinContent(iBin);
351 vSq = h1Sq->GetBinContent(iBin);
352 rms = vSq - v*v;
353 if (rms>0) rms = sqrt(rms);
354 erms = 0.0;
355
356 if (v>0)
357 {
358 omega = rms*rms/v;
359 eomega = 0.0;
360 R2 = (vSq-v)/v/v - 1.0;
361 eR2 = 0;
362 }
363 else
364 {
365 omega = 0.0;
366 eomega = 0.0;
367 R2 = 0.0;
368 eR2 = 0;
369 }
370
371 //cout << " iBin: " << iBin << " v:" << v << " v2:" << v2 << " rms:" << rms << " omega:" << omega << " R2:" << R2 << endl;
372
373 h1Rms->SetBinContent(iBin, rms);
374 h1Rms->SetBinError (iBin, erms);
375
376 h1Omega->SetBinContent(iBin, omega);
377 h1Omega->SetBinError(iBin, eomega);
378
379 h1R2->SetBinContent(iBin, R2);
380 h1R2->SetBinError(iBin, eR2);
381 }
382 if (reportEnd("CollisionGeometryHistograms",getName(),"calculateRms(...)"))
383 ;
384}
