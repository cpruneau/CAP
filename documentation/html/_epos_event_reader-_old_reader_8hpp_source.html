88 // Generate an event
90 Event * generate()
91 {
92 if (debugLevel>0) cout << "EposEventReader::generate() Started" << endl;
93
94 if (!fChain)
95 {
96 cout << "-FATAL- EposEventReader::generate() no TChain available" << endl;
97 }
98
99 bool seekingEvent = true;
100 while (seekingEvent)
101 {
102 // load another event from the root file/TTree
103 Long64_t ientry = LoadTree(jentry);
104
105 // returning a null point is an indication that
106 // there are no more events in the file or stack of files.
107 if (ientry < 0) return 0;
108
109 nb = fChain->GetEntry(jentry++); nbytes += nb;
110 if (nb == 0) return 0;
111
112 if(Impact > generatorConfiguration->impactParameterMinimum &&
113 Impact <= generatorConfiguration->impactParameterMaximum) seekingEvent = false;
114 }
115
116 if (debugLevel>0) cout << "-DEBUG- EposEventReader::generate() new event b: " << Impact << ", ntracks: " << Mult << endl;
117
119 int thePid;
120 double charge;
121
122 event->reset();
123 particleFactory->reset();
124 int particleAccepted = 0;
125
126 // load particles from TTree and copy those that are selected into event
127 Particle aParticle; /* we use a particle object to extract angular information for indexes extraction */
128 Particle *particle; /* this is the actual particle which will be stored */
129 for (int iPart1=0; iPart1<Mult; iPart1++)
130 {
131 //if(ist[iPart1]!=0) continue;
132 //cout << "========== part1: " << iPart1 << endl;
133
134 thePid = PID[iPart1];
135 if ( thePid == 120 || thePid == 130 || thePid ==1120 )
136 charge = 1;
137 else if (thePid ==-120 || thePid ==-130 || thePid==-1120)
138 charge = -1;
139 else
140 charge = 0;
141 // only accept charged particles
142 if (charge==0) continue;
143
144 aParticle.pid = thePid;
145 aParticle.charge = charge;
146 aParticle.setPxPyPzE(Px[iPart1],Py[iPart1],Pz[iPart1],E[iPart1]);
147 aParticle.ixEtaPhi = -1;
148 aParticle.ixYPhi = -1;
149
150 if (aParticle.pt<generatorConfiguration->ptMinimum || aParticle.pt>generatorConfiguration->ptMaximum) continue;
151 if (aParticle.y<generatorConfiguration->yMinimum || aParticle.y>generatorConfiguration->yMaximum) continue;
152
153 particle = particleFactory->getObject();
154 particle->Copy(aParticle);
155
156 //particle->printProperties(cout);
157 particleAccepted++;
158 }
159 event->nParticles = particleAccepted;
160
161 if (debugLevel>0) cout << "-DEBUG- EposEventReader::generate() No of accepted Particles: "<<particleAccepted<<endl;
162 //event->printProperties(cout);
163 if (debugLevel>0) cout << "-DEBUG- EposEventReader::generate() event completed!" << endl;
164 return event;
165 }
