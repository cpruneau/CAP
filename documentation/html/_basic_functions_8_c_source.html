471// Double Gaussian Distribution (mu1, sigma1, mu2, sigma2)
473double DoubleGaussian(double *xx, double *par)
474{
475 double x = xx[0];
476 double mu1 = par[0];
477 double sigma1 = par[1];
478 double mu2 = par[2];
479 double sigma2 = par[3];
480 double f = par[4];
481 double v1;
482 double v2;
483 double v;
484 double arg1 = (x-mu1)/sigma1;
485 v1 = exp(-0.5*arg1*arg1);
486 double arg2 = (x-mu2)/sigma2;
487 v2 = exp(-0.5*arg2*arg2);
488 v = (f*v1+(1-f)*v2)/(sqrt(2*3.1415927)*(f*sigma1+(1-f)*sigma2));
489 return v;
490}

515// Asymetric Gaussian Distribution (mu.sigma1, sigma2)
517double AssymetricGaussian(double *xx, double *par)
518{
519 double x = xx[0];
520 double mu = par[0];
521 double sigma1 = par[1];
522 double sigma2 = par[2];
523 double v;
524 double arg;
525 if (x<mu)
526 {
527 arg = (x-mu)/sigma1;
528 v = 2.*exp(-0.5*arg*arg)/sqrt(2*3.1415927)/(sigma1+sigma2);

532 arg = (x-mu)/sigma2;
533 v = 2.*exp(-0.5*arg*arg)/sqrt(2*3.1415927)/(sigma1+sigma2);

565double ChiSquare(double *xx, double *par)
566{
567 double z = xx[0];
568 double n = par[0];
569 double arg = -0.5*n*log(2) -log( TMath::Gamma(n/2)) + (0.5*n - 1)*log(z) - z/2;
570 double v = exp(arg);
571 return v;

635// Weibull Distribution
637double Weibull(double *xx, double *par)
638{
639 double x = xx[0];
640 if (x<0) return 0;
641 double sigma = par[0];
642 double eta = par[1];
643 if (sigma<=0) return 0;
644 double arg = x/sigma;
645 double lnarg = log(arg);
646 double v = exp(log(eta)-log(sigma)+(eta-1)*lnarg - exp(eta*lnarg));
647 return v;
648}

655{

695double CorrelationModel(double *xx, double *par)
696{
697 double eta1 = xx[0];
698 double eta2 = xx[1];
699 double sigmaAvg = par[0];
700 double sigmaDif = par[1];
701 double dEta = eta1-eta2;
702 double etaAvg = 0.5*(eta1+eta2);
703 double arg3 = dEta/sigmaDif;
704 double arg4 = etaAvg/sigmaAvg;
705 return 1000*exp(-0.5*arg3*arg3)*exp(-0.5*arg4*arg4)/(2.*3.1415927*sigmaDif*sigmaAvg);
706}

732// Calculate the dE/dx vs beta*gamma
733// x: beta*gamma
734// k1: constant = Z/A
735// k2: constant
736// K/A = 0.307075 MeV g^-1 cm^2 for A=1 g mol^-1
737// mass_e = mass of electron in MeV.
738double BetheBlochBetaGamma(double *xx, double *par)
739{
740 double K = 0.307075;
741 double mass_e = 0.5109;
742 double z2 = 1;//par[0]; // square of charge of projectile
743 double Z = 29;
744 double A = 63;
745 double ZA = Z/A; //par[1]; // Z/A of target Copper
746 //double I = 0.000322;//copper par[2]; // average ionizaion potential 322 for Cu
747 double I = 10*Z/1000000.; //
748 double I2 = I*I;
749 double mass_A = 0.938*A;
750 double massRatio = mass_e;
751
752 double betaGamma = xx[0];
753 double betaGammaSq = betaGamma*betaGamma;
754 double betaSq = betaGammaSq/(1+betaGammaSq);
755 double gamma = betaGamma/sqrt(betaSq);
756 double Tmax = 2*mass_e*betaGammaSq/(1+2*gamma*massRatio+massRatio*massRatio);
757 double v = K*z2*ZA*(0.5*log(2*mass_e*betaGammaSq*Tmax/I2) - betaSq)/betaSq;
758 return v;
759}

762// p: momentum
763// x: bet*gamma
764// k1: constant
765// k2: constant
766//
767double BetheBlochP(double *xx, double *par)
768{
769 double K = 0.307075;
770 double mass_e = 0.5109;
771 double z2 = 1;//par[0]; // square of charge of projectile
772 double Z = 29;
773 double A = 63;
774 double ZA = Z/A; //par[1]; // Z/A of target Copper
775 //double I = 0.000322;//copper par[2]; // average ionizaion potential 322 for Cu
776 double I = 10*Z/1000000.; //
777 double I2 = I*I;
778 double mass_A = 0.938*A;
779 double massRatio = mass_e;
780
781 double p = xx[0];
782 double mass_proj = par[0];
783 double betaGamma = p/mass_proj;
784 double betaGammaSq = betaGamma*betaGamma;
785 double betaSq = betaGammaSq/(1+betaGammaSq);
786 double gamma = betaGamma/sqrt(betaSq);
787 double Tmax = 2*mass_e*betaGammaSq/(1+2*gamma*massRatio+massRatio*massRatio);
788 double v = K*z2*ZA*(0.5*log(2*mass_e*betaGammaSq*Tmax/I2) - betaSq)/betaSq;
789 return v/1.36; // units of MIP
790
791
792}

795// Calculate a relative error for a product of two quantities (assumed
796// uncorrelated.
797double relError(double v1, double ev1, double v2, double ev2)
798{
799 double re1, re2;
800 if (v1>0)
801 re1 = ev1/v1;
802 else
803 re1 = 0;
804 if (v2>0)
805 re2 = ev2/v2;
806 else
807 re2 = 0;
808 return sqrt(re1*re1+re2*re2);
